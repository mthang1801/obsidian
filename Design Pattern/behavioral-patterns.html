💡 Behavioral Patterns là gì?
Mẫu thiết kế hành vi — Giao tiếp giữa các đối tượng

Behavioral Patterns tập trung vào cách các đối tượng giao tiếp và phân phối trách nhiệm. Không quan tâm đến cách tạo ra object (Creational) hay cấu trúc (Structural), mà quan tâm đến flow của dữ liệu, logic, và control giữa các components. Trong Go, behavioral patterns thường được hiện thực qua interface + function type thay vì class inheritance.

Go không có class, không có kế thừa — nhưng composition + interface làm cho behavioral patterns thậm chí còn clean hơn so với Java/C++.

01
Strategy Pattern
Định nghĩa họ thuật toán, đóng gói từng cái, hoán đổi được tại runtime
BEHAVIORAL
💡 Định nghĩa — GoF
Strategy — Tách biệt thuật toán khỏi context sử dụng

Strategy cho phép bạn định nghĩa một họ các thuật toán, đặt mỗi cái vào một class riêng và làm chúng hoán đổi được. Context chỉ biết interface của Strategy, không biết concrete implementation. Trong Go, đây là pattern tự nhiên nhất vì interface + dependency injection là idiom cốt lõi.

Vấn đề giải quyết: Khi có nhiều biến thể thuật toán, tránh dùng switch/if-else dài → Strategy tách từng thuật toán ra, dễ thêm/xóa mà không ảnh hưởng client.

Hình 1.1: Kiến trúc mô phỏng Strategy Pattern trong Go — Context giao tiếp gián tiếp qua Interface để linh hoạt thay thế các Strategy tại runtime
Strategy Pattern — UML & Flow

📝 Example 1 — Payment Strategy (Basic)

GOpayment_strategy.goBASIC
// Strategy Pattern: hệ thống thanh toán đa phương thức
// Context: Order không biết cụ thể thanh toán thế nào
// Strategy: PaymentMethod interface

type PaymentMethod interface {
    Pay(amount float64) error
    Name() string
}

// Concrete strategies
type CreditCard struct{ Number, CVV, Expiry string }

func (c *CreditCard) Pay(amount float64) error {
    fmt.Printf("💳 Credit card [****%s] charged $%.2f\n", c.Number[len(c.Number)-4:], amount)
    return nil
}
func (c *CreditCard) Name() string { return "Credit Card" }

type PayPal struct{ Email string }

func (p *PayPal) Pay(amount float64) error {
    fmt.Printf("🅿️  PayPal [%s] charged $%.2f\n", p.Email, amount)
    return nil
}
func (p *PayPal) Name() string { return "PayPal" }

type Crypto struct{ WalletAddr, Coin string }

func (c *Crypto) Pay(amount float64) error {
    fmt.Printf("₿  %s wallet [%s...] paid $%.2f\n", c.Coin, c.WalletAddr[:8], amount)
    return nil
}
func (c *Crypto) Name() string { return c.Coin }

// Context — không biết payment cụ thể
type Order struct {
    ID      string
    Amount  float64
    payment PaymentMethod // injected strategy
}

func (o *Order) SetPayment(p PaymentMethod) { o.payment = p }

func (o *Order) Checkout() error {
    if o.payment == nil { return errors.New("no payment method set") }
    fmt.Printf("Order %s checkout via %s\n", o.ID, o.payment.Name())
    return o.payment.Pay(o.Amount)
}

// Usage
func main() {
    order := &Order{ID: "ORD-001", Amount: 299.99}
    order.SetPayment(&CreditCard{Number: "4111111111111234"})
    order.Checkout() // Credit card charged

    order.SetPayment(&PayPal{Email: "user@example.com"})
    order.Checkout() // PayPal charged — đổi strategy, Order không đổi
}

📝 Example 2 — Sort Strategy với Generics (Advanced)

GOsort_strategy.goADVANCED
// Strategy + Go 1.18 Generics: flexible sorter
// Đổi sorting algorithm tại runtime: QuickSort, MergeSort, HeapSort

type SortStrategy[T any] interface {
    Sort(data []T, less func(a, b T) bool)
    Name() string
}

// BubbleSort — dùng khi data nhỏ, nearly sorted
type BubbleSort[T any] struct{}

func (b BubbleSort[T]) Sort(data []T, less func(T, T) bool) {
    n := len(data)
    for i := 0; i < n-1; i++ {
        swapped := false
        for j := 0; j < n-i-1; j++ {
            if less(data[j+1], data[j]) {
                data[j], data[j+1] = data[j+1], data[j]
                swapped = true
            }
        }
        if !swapped { break } // early exit nếu already sorted
    }
}
func (b BubbleSort[T]) Name() string { return "BubbleSort" }

// QuickSort — general purpose
type QuickSort[T any] struct{}

func (q QuickSort[T]) Sort(data []T, less func(T, T) bool) {
    var qs func(lo, hi int)
    qs = func(lo, hi int) {
        if lo >= hi { return }
        pivot := data[hi]; i := lo
        for j := lo; j < hi; j++ {
            if less(data[j], pivot) { data[i], data[j] = data[j], data[i]; i++ }
        }
        data[i], data[hi] = data[hi], data[i]
        qs(lo, i-1); qs(i+1, hi)
    }
    qs(0, len(data)-1)
}
func (q QuickSort[T]) Name() string { return "QuickSort" }

// Context: Sorter
type Sorter[T any] struct {
    strategy SortStrategy[T]
}

func NewSorter[T any](s SortStrategy[T]) *Sorter[T] { return &Sorter[T]{strategy: s} }

func (s *Sorter[T]) Sort(data []T, less func(T, T) bool) {
    log.Printf("Sorting %d items with %s", len(data), s.strategy.Name())
    s.strategy.Sort(data, less)
}

// Auto-select strategy based on data size (Strategy Factory)
func AutoSorter[T any](size int) *Sorter[T] {
    if size < 20 { return NewSorter[T](BubbleSort[T]{}) }
    return NewSorter[T](QuickSort[T]{})
}

📝 Example 3 — Compression Strategy (Production, Expert)

GOcompression_strategy.goEXPERT
// Production-grade: Compression strategy với metrics, context, streaming
// Tích hợp với io.Writer — pluggable vào bất kỳ pipeline nào

import (
    "compress/gzip"; "compress/zlib"
    "context"; "io"; "time"
)

type CompressResult struct {
    BytesIn, BytesOut int64
    Duration          time.Duration
    Ratio             float64
}

type CompressionStrategy interface {
    Compress(ctx context.Context, src io.Reader, dst io.Writer) (CompressResult, error)
    Decompress(ctx context.Context, src io.Reader, dst io.Writer) error
    MimeType() string
    Level() int
}

// GzipStrategy
type GzipStrategy struct{ level int }

func NewGzip(level int) *GzipStrategy { return &GzipStrategy{level: level} }

func (g *GzipStrategy) Compress(ctx context.Context, src io.Reader, dst io.Writer) (CompressResult, error) {
    start := time.Now()
    w, err := gzip.NewWriterLevel(dst, g.level)
    if err != nil { return CompressResult{}, err }
    defer w.Close()

    // context-aware copy — cancel gracefully
    done := make(chan error, 1)
    var n int64
    go func() {
        n, err = io.Copy(w, src)
        done <- err
    }()
    select {
    case <-ctx.Done(): return CompressResult{}, ctx.Err()
    case err = <-done:
    }
    if err != nil { return CompressResult{}, err }
    elapsed := time.Since(start)
    return CompressResult{BytesIn: n, Duration: elapsed}, nil
}

func (g *GzipStrategy) Decompress(_ context.Context, src io.Reader, dst io.Writer) error {
    r, err := gzip.NewReader(src)
    if err != nil { return err }
    defer r.Close()
    _, err = io.Copy(dst, r)
    return err
}
func (g *GzipStrategy) MimeType() string { return "application/gzip" }
func (g *GzipStrategy) Level() int    { return g.level }

// Context: FileCompressor — đổi strategy tại runtime
type FileCompressor struct {
    strategy CompressionStrategy
    mu       sync.RWMutex
}

func (fc *FileCompressor) SetStrategy(s CompressionStrategy) {
    fc.mu.Lock(); defer fc.mu.Unlock()
    fc.strategy = s
}

func (fc *FileCompressor) CompressFile(ctx context.Context, src, dst string) (CompressResult, error) {
    fc.mu.RLock(); s := fc.strategy; fc.mu.RUnlock()
    in, _ := os.Open(src)
    defer in.Close()
    out, _ := os.Create(dst)
    defer out.Close()
    return s.Compress(ctx, in, out)
}
✅

Go Tip: Thay vì struct với method, đôi khi function type là Strategy đơn giản hơn: type SortFunc func([]int). Dùng khi strategy chỉ cần 1 method và không mang state.

02
Observer Pattern
Pub/Sub — Notify nhiều subscriber khi state của subject thay đổi
BEHAVIORAL
💡 Định nghĩa — GoF
Observer — Định nghĩa quan hệ 1-nhiều giữa các đối tượng

Observer định nghĩa quan hệ phụ thuộc one-to-many giữa đối tượng: khi Subject thay đổi state, tất cả Observers được notify và tự cập nhật. Đây là nền tảng của Event-Driven Architecture, reactive programming, và Pub/Sub systems. Trong Go, observer có thể dùng channel thay vì callback để tận dụng goroutine.

Hình 2.1: Kiến trúc mô phỏng Observer Pattern trong Go — Subject quản lý danh sách đăng ký và tự động thông báo tới toàn bộ Observers khi có thay đổi trạng thái
Observer Pattern — Subject ↔ Observer Communication

📝 Example 1 — Event Bus (Advanced)

GOevent_bus.goADVANCED
// Observer Pattern: Generic Event Bus
// Thread-safe, typed events, unsubscribe support

type EventType string

const (
    EventOrderCreated  EventType = "order.created"
    EventOrderShipped  EventType = "order.shipped"
    EventUserSignedUp  EventType = "user.signed_up"
)

type Event struct {
    Type    EventType
    Payload any
    At      time.Time
}

type Handler func(Event)

type EventBus struct {
    mu       sync.RWMutex
    handlers map[EventType]map[string]Handler // type → {id → handler}
}

func NewEventBus() *EventBus {
    return &EventBus{handlers: make(map[EventType]map[string]Handler)}
}

// Subscribe: đăng ký handler, trả về unsubscribe func
func (b *EventBus) Subscribe(t EventType, h Handler) (func()) {
    b.mu.Lock(); defer b.mu.Unlock()
    id := fmt.Sprintf("%p", &h)
    if b.handlers[t] == nil {
        b.handlers[t] = make(map[string]Handler)
    }
    b.handlers[t][id] = h
    return func() { // unsubscribe closure
        b.mu.Lock(); defer b.mu.Unlock()
        delete(b.handlers[t], id)
    }
}

// Publish: notify tất cả handlers (async bằng goroutine)
func (b *EventBus) Publish(e Event) {
    e.At = time.Now()
    b.mu.RLock()
    handlers := make([]Handler, 0)
    for _, h := range b.handlers[e.Type] { handlers = append(handlers, h) }
    for _, h := range b.handlers["*"] { handlers = append(handlers, h) } // wildcard
    b.mu.RUnlock()

    for _, h := range handlers {
        h := h // capture loop variable
        go func() {
            defer func() { if r := recover(); r != nil { log.Printf("handler panic: %v", r) } }()
            h(e)
        }()
    }
}

// Usage
func main() {
    bus := NewEventBus()

    unsub := bus.Subscribe(EventOrderCreated, func(e Event) {
        fmt.Printf("📧 Email: new order %v\n", e.Payload)
    })
    defer unsub() // cleanup on exit

    bus.Subscribe(EventOrderCreated, func(e Event) {
        fmt.Printf("📊 Analytics: order %v at %v\n", e.Payload, e.At)
    })

    bus.Publish(Event{Type: EventOrderCreated, Payload: map[string]any{"id": "ORD-001"}})
}

📝 Example 2 — Stock Market Observer với Channel (Expert)

GOstock_observer.goEXPERT
// Channel-based Observer: Go-idiomatic approach
// Subject gửi qua channel, observer chạy goroutine độc lập
// Non-blocking với buffer, graceful shutdown với context

type StockTick struct {
    Symbol    string
    Price     float64
    Volume    int64
    Timestamp time.Time
}

type StockObserver interface {
    OnTick(tick StockTick)
    ID() string
}

type StockMarket struct {
    mu        sync.RWMutex
    observers map[string]chan StockTick
    done      chan struct{}
}

func NewStockMarket() *StockMarket {
    return &StockMarket{
        observers: make(map[string]chan StockTick),
        done:      make(chan struct{}),
    }
}

func (sm *StockMarket) Register(obs StockObserver) {
    sm.mu.Lock(); defer sm.mu.Unlock()
    ch := make(chan StockTick, 64) // buffered: không block publisher
    sm.observers[obs.ID()] = ch
    go func() { // goroutine per observer
        for { select {
            case tick := <-ch: obs.OnTick(tick)
            case <-sm.done:   return
        }}
    }()
}

func (sm *StockMarket) Broadcast(tick StockTick) {
    sm.mu.RLock(); defer sm.mu.RUnlock()
    for _, ch := range sm.observers {
        select {
        case ch <- tick: // non-blocking send
        default: log.Println("observer slow, dropping tick")
        }
    }
}

func (sm *StockMarket) Shutdown() { close(sm.done) }

// Concrete observer: price alert
type PriceAlert struct{
    id       string
    symbol   string
    threshold float64
    notify   func(StockTick)
}
func (p *PriceAlert) ID() string { return p.id }
func (p *PriceAlert) OnTick(t StockTick) {
    if t.Symbol == p.symbol && t.Price >= p.threshold {
        p.notify(t)
    }
}
03
Command Pattern
Đóng gói request thành object — undo/redo, queuing, macro
BEHAVIORAL
💡 Định nghĩa — GoF
Command — Biến request thành object độc lập

Command đóng gói một request/action như một object — bao gồm: đối tượng được gọi, method, và tham số. Điều này cho phép: queue request, log history, undo/redo operations. Trong Go, function type là Command tự nhiên nhất, nhưng struct cần thiết khi command cần lưu state để undo.

Hình 3.2: Kiến trúc mô phỏng Command Pattern trong Go — Đóng gói yêu cầu thành đối tượng Command độc lập để quản lý hàng đợi và thực hiện xử lý bất đồng bộ
Command Pattern — Invoker · Command · Receiver

📝 Example 1 — Text Editor Undo/Redo (Advanced)

GOeditor_command.goADVANCED
// Command Pattern: Text Editor với full undo/redo stack
// Mỗi operation (Insert, Delete, Format) là 1 Command object

type Command interface {
    Execute() error
    Undo() error
    Description() string
}

// Receiver: TextEditor
type TextEditor struct {
    content []rune
    cursor  int
}

func (e *TextEditor) Insert(pos int, text string) {
    runes := []rune(text)
    e.content = append(e.content[:0], append(
        append(e.content[:pos:pos], runes...),
        e.content[pos:]...,
    )...)
    e.cursor = pos + len(runes)
}

func (e *TextEditor) Delete(pos, length int) string {
    deleted := string(e.content[pos : pos+length])
    e.content = append(e.content[:pos], e.content[pos+length:]...)
    e.cursor = pos
    return deleted
}

func (e *TextEditor) Content() string { return string(e.content) }

// ConcreteCommand: InsertCommand
type InsertCommand struct {
    editor   *TextEditor
    position int
    text     string
}

func (c *InsertCommand) Execute() error {
    c.editor.Insert(c.position, c.text); return nil
}
func (c *InsertCommand) Undo() error {
    c.editor.Delete(c.position, len([]rune(c.text))); return nil
}
func (c *InsertCommand) Description() string {
    return fmt.Sprintf("Insert %q at %d", c.text, c.position)
}

// Invoker: CommandHistory (Undo/Redo stack)
type CommandHistory struct {
    done   []Command
    undone []Command
}

func (h *CommandHistory) Execute(cmd Command) error {
    if err := cmd.Execute(); err != nil { return err }
    h.done = append(h.done, cmd)
    h.undone = h.undone[:0] // clear redo stack
    return nil
}

func (h *CommandHistory) Undo() error {
    if len(h.done) == 0 { return errors.New("nothing to undo") }
    n := len(h.done) - 1
    cmd := h.done[n]; h.done = h.done[:n]
    if err := cmd.Undo(); err != nil { return err }
    h.undone = append(h.undone, cmd)
    return nil
}

func (h *CommandHistory) Redo() error {
    if len(h.undone) == 0 { return errors.New("nothing to redo") }
    n := len(h.undone) - 1
    cmd := h.undone[n]; h.undone = h.undone[:n]
    if err := cmd.Execute(); err != nil { return err }
    h.done = append(h.done, cmd)
    return nil
}

📝 Example 2 — HTTP Command Queue với Worker Pool (Expert)

GOcommand_queue.goEXPERT
// Command Queue: async task processing với retries, priority, cancellation
// Kết hợp Command pattern + Worker Pool

type Priority int
const (Low Priority = iota; Normal; High; Critical)

type AsyncCommand struct {
    ID       string
    Priority Priority
    MaxRetry int
    Timeout  time.Duration
    Fn       func(ctx context.Context) error
    OnDone   func(error)
}

type CommandQueue struct {
    queue   chan *AsyncCommand
    workers int
    wg      sync.WaitGroup
    ctx     context.Context
    cancel  context.CancelFunc
}

func NewCommandQueue(workers, capacity int) *CommandQueue {
    ctx, cancel := context.WithCancel(context.Background())
    q := &CommandQueue{
        queue: make(chan *AsyncCommand, capacity),
        workers: workers, ctx: ctx, cancel: cancel,
    }
    for i := 0; i < workers; i++ {
        q.wg.Add(1)
        go q.worker()
    }
    return q
}

func (q *CommandQueue) worker() {
    defer q.wg.Done()
    for { select {
        case cmd, ok := <-q.queue:
            if !ok { return }
            q.execute(cmd)
        case <-q.ctx.Done(): return
    }}
}

func (q *CommandQueue) execute(cmd *AsyncCommand) {
    var err error
    for attempt := 0; attempt <= cmd.MaxRetry; attempt++ {
        ctx, cancel := context.WithTimeout(q.ctx, cmd.Timeout)
        err = cmd.Fn(ctx)
        cancel()
        if err == nil { break }
        if attempt < cmd.MaxRetry {
            time.Sleep(time.Duration(1<100 * time.Millisecond) // exp backoff
        }
    }
    if cmd.OnDone != nil { cmd.OnDone(err) }
}

func (q *CommandQueue) Enqueue(cmd *AsyncCommand) error {
    select {
    case q.queue <- cmd: return nil
    default: return errors.New("queue full")
    }
}

func (q *CommandQueue) Shutdown() {
    q.cancel(); close(q.queue); q.wg.Wait()
}
04
Template Method Pattern
Định nghĩa bộ khung thuật toán — các bước có thể override
BEHAVIORAL
💡 Định nghĩa — GoF
Template Method — Skeleton trong base, chi tiết trong subtype

Template Method định nghĩa bộ khung (skeleton) của một thuật toán trong base class/struct, và cho phép các subtype override các bước cụ thể mà không thay đổi cấu trúc tổng thể. Trong Go (không có kế thừa), ta dùng embedding + interface hooks hoặc function callbacks để inject các bước có thể customize.

Hình 4.1: Kiến trúc mô phỏng Template Method Pattern trong Go — Định nghĩa bộ khung thuật toán cố định và cho phép tùy biến linh hoạt các bước cụ thể thông qua interface hooks
Template Method — Cấu trúc Pipeline với Hooks

📝 Example 1 — Data Processing Pipeline (Advanced)

GOdata_pipeline.goADVANCED
// Template Method: Data Processing Pipeline
// Base struct định nghĩa skeleton, hooks được inject qua interface

type DataHooks interface {
    Parse(raw []byte) (any, error)
    Process(data any) (any, error)
    Format(result any) ([]byte, error)
}

// Base processor — cố định structure: validate → parse → process → format → save
type BaseProcessor struct {
    hooks    DataHooks
    maxBytes int
    logger   *log.Logger
}

func NewProcessor(hooks DataHooks) *BaseProcessor {
    return &BaseProcessor{hooks: hooks, maxBytes: 10 << 20, logger: log.Default()}
}

// Template method — cấu trúc cố định
func (p *BaseProcessor) Run(ctx context.Context, raw []byte) ([]byte, error) {
    // Step 1: Validate (fixed)
    if len(raw) == 0              { return nil, errors.New("empty input") }
    if len(raw) > p.maxBytes       { return nil, fmt.Errorf("input too large: %d bytes", len(raw)) }
    if err := ctx.Err(); err != nil { return nil, err }

    // Step 2: Parse (hook)
    p.logger.Printf("parsing %d bytes", len(raw))
    data, err := p.hooks.Parse(raw)
    if err != nil { return nil, fmt.Errorf("parse: %w", err) }

    // Step 3: Process (hook)
    result, err := p.hooks.Process(data)
    if err != nil { return nil, fmt.Errorf("process: %w", err) }

    // Step 4: Format (hook)
    out, err := p.hooks.Format(result)
    if err != nil { return nil, fmt.Errorf("format: %w", err) }

    p.logger.Printf("done: output %d bytes", len(out))
    return out, nil
}

// ConcreteHooks: JSONHooks
type JSONHooks struct{}

func (j JSONHooks) Parse(raw []byte) (any, error) {
    var m map[string]any
    return m, json.Unmarshal(raw, &m)
}
func (j JSONHooks) Process(data any) (any, error) {
    m := data.(map[string]any)
    m["processed_at"] = time.Now().Format(time.RFC3339)
    return m, nil
}
func (j JSONHooks) Format(result any) ([]byte, error) {

📝 Example 1 — Tree Iterator (Advanced)

GOtree_iterator.goADVANCED
// Iterator Pattern: BST in-order traversal
// 3 cách implement: interface, channel, closure

type TreeNode struct {
    Val   int
    Left, Right *TreeNode
}

// Approach 1: Interface-based Iterator (eager — preloads all)
type Iterator[T any] interface {
    HasNext() bool
    Next() T
    Reset()
}

type BSTIterator struct {
    nodes []int
    idx   int
}

func NewBSTIterator(root *TreeNode) *BSTIterator {
    it := &BSTIterator{}
    var inorder func(*TreeNode)
    inorder = func(n *TreeNode) {
        if n == nil { return }
        inorder(n.Left)
        it.nodes = append(it.nodes, n.Val)
        inorder(n.Right)
    }
    inorder(root)
    return it
}
func (it *BSTIterator) HasNext() bool { return it.idx < len(it.nodes) }
func (it *BSTIterator) Next() int    { v := it.nodes[it.idx]; it.idx++; return v }
func (it *BSTIterator) Reset()       { it.idx = 0 }

// Approach 2: Channel-based Iterator (lazy — Go idiom)
// Nhận từng phần tử qua channel — goroutine traverses tree
func BSTChannelIterator(root *TreeNode) <-chan int {
    ch := make(chan int)
    var traverse func(*TreeNode)
    traverse = func(n *TreeNode) {
        if n == nil { return }
        traverse(n.Left)
        ch <- n.Val
        traverse(n.Right)
    }
    go func() {
        defer close(ch)
        traverse(root)
    }()
    return ch
}
// for val := range BSTChannelIterator(root) { fmt.Println(val) }

// Approach 3: Lazy stack-based iterator (O(h) space, true O(1) per step)
type LazyBSTIter struct{ stack []*TreeNode }

func NewLazyBSTIter(root *TreeNode) *LazyBSTIter {
    it := &LazyBSTIter{}
    it.pushLeft(root)
    return it
}
func (it *LazyBSTIter) pushLeft(n *TreeNode) {
    for n != nil { it.stack = append(it.stack, n); n = n.Left }
}
func (it *LazyBSTIter) HasNext() bool { return len(it.stack) > 0 }
func (it *LazyBSTIter) Next() int {
    top := it.stack[len(it.stack)-1]
    it.stack = it.stack[:len(it.stack)-1]
    it.pushLeft(top.Right) // push right subtree
    return top.Val
}
// Best choice for large trees: O(h) space, true lazy evaluation

// Chain of Responsibility: HTTP Middleware
// Go-idiomatic: func(http.Handler) http.Handler pattern

type Middleware func(http.Handler) http.Handler

// Chain: compose middlewares left-to-right
func Chain(h http.Handler, mws ...Middleware) http.Handler {
    for i := len(mws) - 1; i >= 0; i-- {
        h = mws[i](h)
    }
    return h
}

// Logger middleware
func LoggerMiddleware(logger *slog.Logger) Middleware {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            start := time.Now()
            rw := &responseWriter{ResponseWriter: w}
            next.ServeHTTP(rw, r)
            logger.Info("request",
                "method", r.Method, "path", r.URL.Path,
                "status", rw.status, "dur", time.Since(start),
            )
        })
    }
}

// Auth middleware: validate Bearer token
func AuthMiddleware(validate func(string) (string, error)) Middleware {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            token := strings.TrimPrefix(r.Header.Get("Authorization"), "Bearer ")
            userID, err := validate(token)
            if err != nil {
                http.Error(w, "Unauthorized", http.StatusUnauthorized)
                return // short-circuit chain
            }
            ctx := context.WithValue(r.Context(), "userID", userID)
            next.ServeHTTP(w, r.WithContext(ctx))
        })
    }
}

// RateLimit middleware
func RateLimitMiddleware(rps int) Middleware {
    limiter := rate.NewLimiter(rate.Limit(rps), rps)
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            if !limiter.Allow() {
                http.Error(w, "Too Many Requests", http.StatusTooManyRequests)
                return
            }
            next.ServeHTTP(w, r)
        })
    }
}

// Mediator Pattern: Chat Room
// Users không communicate trực tiếp — tất cả qua ChatRoom (mediator)

type ChatMediator interface {
    SendMessage(from User, msg string, to ...string) // broadcast or DM
    Join(user User)
    Leave(userID string)
}

type User interface {
    ID() string
    Receive(from, msg string)
}

type ChatRoom struct {
    mu      sync.RWMutex
    users   map[string]User
    history []string
}

func NewChatRoom() *ChatRoom {
    return &ChatRoom{users: make(map[string]User)}
}

func (cr *ChatRoom) Join(u User) {
    cr.mu.Lock(); cr.users[u.ID()] = u; cr.mu.Unlock()
    cr.SendMessage(u, fmt.Sprintf("%s joined the room", u.ID()))
}

func (cr *ChatRoom) Leave(id string) {
    cr.mu.Lock(); delete(cr.users, id); cr.mu.Unlock()
    log.Printf("%s left", id)
}

func (cr *ChatRoom) SendMessage(from User, msg string, to ...string) {
    entry := fmt.Sprintf("[%s] %s: %s", time.Now().Format("15:04"), from.ID(), msg)
    cr.mu.Lock()
    cr.history = append(cr.history, entry)
    cr.mu.Unlock()

    cr.mu.RLock(); defer cr.mu.RUnlock()

    if len(to) == 0 { // broadcast
        for _, u := range cr.users {
            if u.ID() != from.ID() { u.Receive(from.ID(), msg) }
        }
        return
    }
    for _, id := range to { // direct message
        if u, ok := cr.users[id]; ok { u.Receive(from.ID(), msg) }
    }
}

// Concrete User: BotUser
type BotUser struct {
    id       string
    room     ChatMediator
    onMsg    func(from, msg string) string // reply generator
}
func (b *BotUser) ID() string { return b.id }
func (b *BotUser) Receive(from, msg string) {
    if reply := b.onMsg(from, msg); reply != "" {
        b.room.SendMessage(b, reply, from) // reply to sender
    }
}

// Concrete User: HumanUser (simulated)
type HumanUser struct{
    id   string
    room ChatMediator
    inbox []string
}
func (h *HumanUser) ID() string { return h.id }
func (h *HumanUser) Receive(from, msg string) {
    h.inbox = append(h.inbox, fmt.Sprintf("%s: %s", from, msg))

GOconfig_memento.goEXPERT
// Memento Pattern: Application Config với snapshot/rollback
// Originator: AppConfig  Memento: ConfigSnapshot  Caretaker: ConfigHistory

// Memento — immutable snapshot (không export fields)
type ConfigSnapshot struct {
    data      map[string]any
    createdAt time.Time
    label     string
}
func (s *ConfigSnapshot) Label() string    { return s.label }
func (s *ConfigSnapshot) CreatedAt() time.Time { return s.createdAt }

// Originator: AppConfig
type AppConfig struct {
    mu   sync.RWMutex
    data map[string]any
}

func NewAppConfig() *AppConfig {
    return &AppConfig{data: map[string]any{
        "debug": false, "port": 8080, "timeout": 30,
    }}
}

func (c *AppConfig) Set(key string, val any) {
    c.mu.Lock(); defer c.mu.Unlock()
    c.data[key] = val
}

func (c *AppConfig) Get(key string) (any, bool) {
    c.mu.RLock(); defer c.mu.RUnlock()
    v, ok := c.data[key]; return v, ok
}

// Save: tạo snapshot — deep copy để tránh aliasing
func (c *AppConfig) Save(label string) *ConfigSnapshot {
    c.mu.RLock(); defer c.mu.RUnlock()
    snap := make(map[string]any, len(c.data))
    for k, v := range c.data { snap[k] = v }
    return &ConfigSnapshot{data: snap, createdAt: time.Now(), label: label}
}

// Restore: apply snapshot state
func (c *AppConfig) Restore(s *ConfigSnapshot) {
    c.mu.Lock(); defer c.mu.Unlock()
    c.data = make(map[string]any, len(s.data))
    for k, v := range s.data { c.data[k] = v }
}

// Caretaker: manages snapshot history
type ConfigHistory struct {
    config   *AppConfig
    snapshots []*ConfigSnapshot
    cursor    int
}

func NewConfigHistory(c *AppConfig) *ConfigHistory {
    h := &ConfigHistory{config: c, cursor: -1}
    h.Checkpoint("initial")
    return h
}

func (h *ConfigHistory) Checkpoint(label string) {
    // Discard future snapshots if we branched
    h.snapshots = h.snapshots[:h.cursor+1]
    h.snapshots = append(h.snapshots, h.config.Save(label))
    h.cursor = len(h.snapshots) - 1
}

func (h *ConfigHistory) Undo() bool {
    if h.cursor <= 0 { return false }
    h.cursor--
    h.config.Restore(h.snapshots[h.cursor])
    return true
}

func (h *ConfigHistory) Redo() bool {
    if h.cursor >= len(h.snapshots)-1 { return false }
    h.cursor++
    h.config.Restore(h.snapshots[h.cursor])
    return true
}

func (h *ConfigHistory) Timeline() []string {
    out := make([]string, len(h.snapshots))
    for i, s := range h.snapshots {
        mark := " "
        if i == h.cursor { mark = "→" }
        out[i] = fmt.Sprintf("%s [%d] %s (%s)", mark, i, s.Label(), s.CreatedAt().Format("15:04:05"))
    }
    return out
}

GOast_visitor.goEXPERT
// Visitor Pattern: Math Expression AST với 3 visitors
// Evaluator, Printer, Optimizer — không đụng đến Expression structs

// Expression interface: Accept triggers double dispatch
type Expr interface {
    Accept(v ExprVisitor) float64
}

type ExprVisitor interface {
    VisitNumber(n *Number) float64
    VisitAdd(a *AddExpr) float64
    VisitMul(m *MulExpr) float64
    VisitVar(v *VarExpr) float64
}

// Concrete elements
type Number  struct{ Val float64 }
type AddExpr struct{ Left, Right Expr }
type MulExpr struct{ Left, Right Expr }
type VarExpr struct{ Name string }

func (n *Number)  Accept(v ExprVisitor) float64 { return v.VisitNumber(n) }
func (a *AddExpr) Accept(v ExprVisitor) float64 { return v.VisitAdd(a) }
func (m *MulExpr) Accept(v ExprVisitor) float64 { return v.VisitMul(m) }
func (vr *VarExpr) Accept(v ExprVisitor) float64 { return v.VisitVar(vr) }

// Visitor 1: Evaluator
type Evaluator struct{ Vars map[string]float64 }

func (e *Evaluator) VisitNumber(n *Number) float64 { return n.Val }
func (e *Evaluator) VisitAdd(a *AddExpr) float64 {
    return a.Left.Accept(e) + a.Right.Accept(e)
}
func (e *Evaluator) VisitMul(m *MulExpr) float64 {
    return m.Left.Accept(e) * m.Right.Accept(e)
}
func (e *Evaluator) VisitVar(v *VarExpr) float64 { return e.Vars[v.Name] }

// Visitor 2: Printer (returns string via side effect)
type Printer struct{ buf strings.Builder }

func (p *Printer) VisitNumber(n *Number) float64 { fmt.Fprintf(&p.buf, "%.g", n.Val); return 0 }
func (p *Printer) VisitAdd(a *AddExpr) float64 {
    p.buf.WriteString("(")
    a.Left.Accept(p); p.buf.WriteString(" + "); a.Right.Accept(p)
    p.buf.WriteString(")"); return 0
}
func (p *Printer) VisitMul(m *MulExpr) float64 {
    p.buf.WriteString("(")
    m.Left.Accept(p); p.buf.WriteString(" * "); m.Right.Accept(p)
    p.buf.WriteString(")"); return 0
}
func (p *Printer) VisitVar(v *VarExpr) float64 { p.buf.WriteString(v.Name); return 0 }
func (p *Printer) String() string        { return p.buf.String() }

// Example: (x + 2) * (x + 3) với x=5

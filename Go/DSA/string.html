01
Sliding Window Advanced → Expert

Kỹ thuật "cửa sổ trượt" duy trì một vùng con liên tiếp (window) trong array/string, thay vì tính lại từ đầu mỗi lần — từ đó giảm độ phức tạp từ O(n·k) xuống O(n). Hai biến thể chính: Fixed Window (kích thước cố định) và Dynamic Window (mở rộng/thu hẹp).

① DEFINE — Cơ chế hoạt động

Sliding Window — Fixed Size (k=3)
Dynamic Window — Two Pointer: left/right mở rộng & thu hẹp

② CODE — Fixed Window: Max Sum Substring

Intermediate
GO sliding_window_fixed.go 
// Package stringalgo — Fixed-size Sliding Window
// Tìm tổng lớn nhất của k phần tử liên tiếp trong mảng số nguyên
package stringalgo

import "fmt"

// MaxSumSubarray tìm tổng lớn nhất của k phần tử liên tiếp.
// Time: O(n) — chỉ duyệt một lần, không lồng vòng lặp
// Space: O(1) — không dùng extra memory
func MaxSumSubarray(nums []int, k int) (int, error) {
    if k > len(nums) || k <= 0 {
        return 0, fmt.Errorf("k=%d không hợp lệ với len=%d", k, len(nums))
    }

    // Bước 1: Tính tổng cửa sổ đầu tiên [0..k-1]
    windowSum := 0
    for i := 0; i < k; i++ {
        windowSum += nums[i]
    }
    maxSum := windowSum

    // Bước 2: Trượt cửa sổ — thêm phần tử mới, bỏ phần tử cũ (O(1) mỗi bước)
    for right := k; right < len(nums); right++ {
        left := right - k  // phần tử sắp bị loại khỏi window
        windowSum += nums[right] - nums[left]
        if windowSum > maxSum {
            maxSum = windowSum
        }
    }
    return maxSum, nil
}

// MaxSumSubarrayWithIndices trả về cả vị trí [start, end] của window tốt nhất
func MaxSumSubarrayWithIndices(nums []int, k int) (maxSum, start, end int) {
    for i := 0; i < k; i++ { maxSum += nums[i] }
    windowSum := maxSum
    start, end = 0, k-1

    for r := k; r < len(nums); r++ {
        windowSum += nums[r] - nums[r-k]
        if windowSum > maxSum {
            maxSum, start, end = windowSum, r-k+1, r
        }
    }
    return
}

// Benchmark reference: BenchmarkMaxSum-8  5000000  240 ns/op  0 allocs/op

Advanced
GO sliding_window_dynamic.go 
// LongestUniqueSubstring — Dynamic Sliding Window
// Tìm độ dài substring dài nhất không có ký tự lặp
// Strategy: dùng map[byte]int lưu vị trí CUỐI CÙNG của mỗi ký tự

func LongestUniqueSubstring(s string) (int, string) {
    lastSeen := make(map[byte]int)  // ký tự → index lần thấy gần nhất
    maxLen, best := 0, ""
    left := 0

    for right := 0; right < len(s); right++ {
        ch := s[right]

        // Nếu ký tự đã trong window, co left vượt qua vị trí cũ
        if idx, ok := lastSeen[ch]; ok && idx >= left {
            left = idx + 1  // co left sang phải, bỏ qua ký tự trùng
        }
        lastSeen[ch] = right  // cập nhật vị trí mới nhất

        if curLen := right - left + 1; curLen > maxLen {
            maxLen = curLen
            best = s[left:right+1]
        }
    }
    return maxLen, best
}

// AllLongestUnique — trả về tất cả substring dài nhất có unique chars (trường hợp có nhiều kết quả)
// Ví dụ: "abab" → ["ab", "ba"] (cùng độ dài 2)
func AllLongestUnique(s string) []string {
    lastSeen := make(map[byte]int)
    maxLen, left := 0, 0
    var result []string

    for right := 0; right < len(s); right++ {
        if idx, ok := lastSeen[s[right]]; ok && idx >= left {
            left = idx + 1
        }
        lastSeen[s[right]] = right
        cur := s[left : right+1]

        switch curLen := len(cur); {
        case curLen > maxLen:
            maxLen, result = curLen, []string{cur}
        case curLen == maxLen:
            result = append(result, cur)
        }
    }
    return result
}

// Test cases:
// LongestUniqueSubstring("abcabcbb") → 3, "abc"
// LongestUniqueSubstring("pwwkew")   → 3, "wke"
// LongestUniqueSubstring("bbbbb")    → 1, "b"

④ CODE — Advanced: Minimum Window Substring

Advanced
GO min_window_substring.go 
// MinWindowSubstring — LeetCode #76 variant
// Tìm substring ngắn nhất của s chứa TẤT CẢ ký tự trong t
// Kỹ thuật: dùng "formed" counter — track khi nào đủ điều kiện

func MinWindowSubstring(s, t string) string {
    if len(s) == 0 || len(t) == 0 { return "" }

    // need: số lần mỗi ký tự trong t cần xuất hiện
    need := make(map[byte]int)
    for i := 0; i < len(t); i++ { need[t[i]]++ }

    window := make(map[byte]int)
    required := len(need)  // số ký tự unique trong t cần đạt
    formed := 0          // số ký tự đã thỏa đủ tần suất

    left := 0
    minLen := len(s) + 1
    minLeft := 0

    for right := 0; right < len(s); right++ {
        ch := s[right]
        window[ch]++

        // Kiểm tra: ký tự ch vừa thêm vào có đạt đủ tần suất cần không?
        if cnt, ok := need[ch]; ok && window[ch] == cnt {
            formed++
        }

        // Khi window đã thỏa — co left để tìm window nhỏ hơn
        for formed == required && left <= right {
            if right-left+1 < minLen {
                minLen, minLeft = right-left+1, left
            }
            lch := s[left]
            window[lch]--
            // Nếu bỏ ký tự left làm thiếu — giảm formed
            if cnt, ok := need[lch]; ok && window[lch] < cnt {
                formed--
            }
            left++
        }
    }
    if minLen > len(s) { return "" }
    return s[minLeft : minLeft+minLen]
}

// MinWindowSubstring("ADOBECODEBANC", "ABC") → "BANC"
// Time: O(|s| + |t|)  Space: O(|s| + |t|)

⑤ CODE — Expert: Rabin-Karp Rolling Hash

★ Expert
💡

Rabin-Karp dùng rolling hash để so sánh substring trong O(1) thay vì O(m) — tìm tất cả anagram/pattern trong văn bản lớn, search plagiarism detection, DNA sequence matching.

GO rabin_karp.go 
package stringalgo

import (
    "math/big"
)

const (
    base  = 31              // base cho polynomial hash (prime gần alphabet size)
    modp  = 1_000_000_007  // large prime tránh hash collision
)

// RabinKarp tìm tất cả vị trí pattern xuất hiện trong text.
// Time: O(n+m) trung bình, O(nm) worst-case (khi nhiều collision)
// Trick: dùng double hashing để giảm collision rate xuống cực thấp
func RabinKarp(text, pattern string) []int {
    n, m := len(text), len(pattern)
    if m == 0 || m > n { return nil }

    // Tính base^(m-1) % modp để loại hash của ký tự trái nhất
    power := 1
    for i := 0; i < m-1; i++ {
        power = (power * base) % modp
    }

    // Hàm hash nội tuyến: h = (h * base + char) % modp
    hashOf := func(s string) int {
        h := 0
        for i := 0; i < len(s); i++ {
            h = (h*base + int(s[i])) % modp
        }
        return h
    }

    patHash := hashOf(pattern)
    winHash := hashOf(text[:m])
    var result []int

    for i := 0; i <= n-m; i++ {
        // So sánh hash trước — chỉ verify string nếu hash khớp (giảm O(m) xuống O(1))
        if winHash == patHash && text[i:i+m] == pattern {
            result = append(result, i)
        }
        if i < n-m {
            // Rolling hash: loại ký tự text[i], thêm ký tự text[i+m]
            winHash = (base*(winHash-int(text[i])*power%modp+modp) + int(text[i+m])) % modp
        }
    }
    return result
}

// FindAllAnagrams — tìm tất cả vị trí bắt đầu của anagram của p trong s
// Biến thể Sliding Window: dùng frequency array thay vì sort, O(n) guaranteed
func FindAllAnagrams(s, p string) []int {
    if len(p) > len(s) { return nil }

    var need, have [26]int  // fixed-size array — zero allocation
    for _, c := range p { need[c-'a']++ }

    m := len(p)
    var res []int

    for i, c := range s {
        have[c-'a']++
        if i >= m {
            have[s[i-m]-'a']--  // loại ký tự rời khỏi window
        }
        if have == need {       // so sánh array O(26) = O(1)
            res = append(res, i-m+1)
        }
    }
    return res
}

// Ví dụ: FindAllAnagrams("cbaebabacd", "abc") → [0, 6]
// Dùng [26]int array thay map: zero allocation, cache-friendly

// VerifyPrimality dùng big.Int để kiểm tra modp là prime (dùng cho production)
var _ = big.NewInt(modp).ProbablyPrime(20) // true

② CODE — Trie cơ bản với Delete

Intermediate
GO trie.go 
package stringalgo

// TrieNode — mỗi node chứa tối đa 26 con (lowercase letters)
// Dùng array thay map: O(1) access, cache-friendly hơn
type TrieNode struct {
    children  [26]*TrieNode
    isEnd     bool
    wordCount int  // đếm bao nhiêu từ pass qua node này
}

type Trie struct {
    root *TrieNode
    size int  // số từ trong trie
}

func NewTrie() *Trie {
    return &Trie{root: &TrieNode{}}
}

// Insert thêm word vào Trie — O(m), m = len(word)
func (t *Trie) Insert(word string) {
    node := t.root
    for _, ch := range word {
        idx := ch - 'a'
        if node.children[idx] == nil {
            node.children[idx] = &TrieNode{}
        }
        node = node.children[idx]
        node.wordCount++  // dùng cho prefix count
    }
    if !node.isEnd {
        node.isEnd = true
        t.size++
    }
}

// Search kiểm tra word có trong Trie không (phân biệt hoa/thường)
func (t *Trie) Search(word string) bool {
    node := t.find(word)
    return node != nil && node.isEnd
}

// StartsWith kiểm tra có từ nào bắt đầu bằng prefix không
func (t *Trie) StartsWith(prefix string) bool {
    return t.find(prefix) != nil
}

// CountWordsWithPrefix đếm số từ bắt đầu bằng prefix — O(m)
func (t *Trie) CountWordsWithPrefix(prefix string) int {
    node := t.find(prefix)
    if node == nil { return 0 }
    return node.wordCount
}

func (t *Trie) find(s string) *TrieNode {
    node := t.root
    for _, ch := range s {
        idx := ch - 'a'
        if idx < 0 || idx > 25 || node.children[idx] == nil {
            return nil
        }
        node = node.children[idx]
    }
    return node
}

// Delete xóa word khỏi Trie — trả về true nếu xóa được
// Recursive approach: chỉ xóa node nếu không còn con nào
func (t *Trie) Delete(word string) bool {
    deleted := t.deleteHelper(t.root, word, 0)
    if deleted { t.size-- }
    return deleted
}

func (t *Trie) deleteHelper(node *TrieNode, word string, depth int) bool {
    if depth == len(word) {
        if !node.isEnd { return false } // không có từ này
        node.isEnd = false
        return true
    }
    idx := word[depth] - 'a'
    child := node.children[idx]
    if child == nil { return false }

    if !t.deleteHelper(child, word, depth+1) { return false }

    child.wordCount--
    // Xóa node nếu không còn từ nào pass qua và không phải end
    if child.wordCount == 0 && !child.isEnd {
        node.children[idx] = nil  // garbage collector sẽ thu hồi
    }
    return true
}

③ CODE — Generic Trie (Go 1.18+)

Advanced
GO trie_generic.go 
// GenericTrie — Trie với value type T, dùng generics Go 1.18+
// Có thể lưu bất kỳ metadata nào tại EndOfWord node
// Use case: autocomplete + metadata (frequency, score, document_id...)

package stringalgo

type TrieValue[T any] struct {
    children map[rune]*TrieValue[T]  // map thay array: hỗ trợ Unicode đầy đủ
    isEnd    bool
    Value    T                        // metadata tùy chỉnh tại leaf/end node
}

type GenericTrie[T any] struct {
    root *TrieValue[T]
}

func NewGenericTrie[T any]() *GenericTrie[T] {
    return &GenericTrie[T]{root: &TrieValue[T]{
        children: make(map[rune]*TrieValue[T]),
    }}
}

// Upsert chèn/cập nhật word với value — idempotent
func (t *GenericTrie[T]) Upsert(word string, value T) {
    node := t.root
    for _, ch := range word {  // range string iter over runes — Unicode safe
        if _, ok := node.children[ch]; !ok {
            node.children[ch] = &TrieValue[T]{
                children: make(map[rune]*TrieValue[T]),
            }
        }
        node = node.children[ch]
    }
    node.isEnd = true
    node.Value = value
}

// Get trả về value và found flag — API giống map
func (t *GenericTrie[T]) Get(word string) (T, bool) {
    node := t.root
    for _, ch := range word {
        n, ok := node.children[ch]
        if !ok { var zero T; return zero, false }
        node = n
    }
    if !node.isEnd { var zero T; return zero, false }
    return node.Value, true
}

// AllWithPrefix trả về tất cả (word, value) có prefix nhất định — DFS
func (t *GenericTrie[T]) AllWithPrefix(prefix string) []struct{ Word string; Val T } {
    node := t.root
    for _, ch := range prefix {
        n, ok := node.children[ch]
        if !ok { return nil }
        node = n
    }
    var results []struct{ Word string; Val T }
    t.dfs(node, []rune(prefix), &results)
    return results
}

func (t *GenericTrie[T]) dfs(node *TrieValue[T], path []rune, out *[]struct{ Word string; Val T }) {
    if node.isEnd {
        *out = append(*out, struct{ Word string; Val T }{
            Word: string(path),
            Val:  node.Value,
        })
    }
    for ch, child := range node.children {
        t.dfs(child, append(path, ch), out)
    }
}

// Ví dụ sử dụng:
// trie := NewGenericTrie[int]()  // int = search frequency
// trie.Upsert("golang", 1500)
// trie.Upsert("go", 5000)
// results := trie.AllWithPrefix("go")
// → [{Word:"go", Val:5000}, {Word:"golang", Val:1500}]

④ CODE — Autocomplete Engine với Top-K

Advanced
GO autocomplete.go 
package stringalgo

import (
    "container/heap"
    "sort"
)

// Suggestion — kết quả autocomplete có score
type Suggestion struct {
    Word  string
    Score int  // tần suất search, hoặc relevance score
}

// acNode — node với max-heap để track top suggestions trong subtree
type acNode struct {
    children [26]*acNode
    topK     []Suggestion  // pre-computed top-k tại mỗi node
    isEnd    bool
    score    int
}

type AutocompleteEngine struct {
    root *acNode
    k    int   // số suggestion tối đa trả về
}

func NewAutocompleteEngine(k int) *AutocompleteEngine {
    return &AutocompleteEngine{root: &acNode{}, k: k}
}

func (a *AutocompleteEngine) AddWord(word string, score int) {
    node := a.root
    sug := Suggestion{word, score}

    for _, ch := range word {
        idx := ch - 'a'
        if idx < 0 || idx >= 26 { continue }
        if node.children[idx] == nil {
            node.children[idx] = &acNode{}
        }
        node = node.children[idx]
        // Cập nhật top-K tại mỗi node trên đường đi — O(k)
        node.topK = a.insertTopK(node.topK, sug)
    }
    node.isEnd = true
    node.score = score
}

// insertTopK duy trì slice sorted by score, chỉ giữ k phần tử lớn nhất
func (a *AutocompleteEngine) insertTopK(topK []Suggestion, s Suggestion) []Suggestion {
    topK = append(topK, s)
    sort.Slice(topK, func(i, j int) bool {
        return topK[i].Score > topK[j].Score
    })
    if len(topK) > a.k { topK = topK[:a.k] }
    return topK
}

// Search trả về top-K suggestions cho prefix — O(m) sau khi xây dựng
// Trade-off: tốn O(k) memory mỗi node, nhưng query cực nhanh
func (a *AutocompleteEngine) Search(prefix string) []Suggestion {
    node := a.root
    for _, ch := range prefix {
        idx := ch - 'a'
        if idx < 0 || idx >= 26 || node.children[idx] == nil {
            return nil
        }
        node = node.children[idx]
    }
    return node.topK  // đã pre-sorted — trả về ngay, O(1)
}

// MinHeap để dùng với container/heap khi k rất lớn
type SugHeap []Suggestion
func (h SugHeap) Len() int            { return len(h) }
func (h SugHeap) Less(i, j int) bool { return h[i].Score < h[j].Score }
func (h SugHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }
func (h *SugHeap) Push(x any)        { *h = append(*h, x.(Suggestion)) }
func (h *SugHeap) Pop() any          { old := *h; n := len(old); x := old[n-1]; *h = old[:n-1]; return x }

var _ heap.Interface = (*SugHeap)(nil) // compile-time interface check

⑤ CODE — Expert: Concurrent Trie với RWMutex

★ Expert
⚡

Production Pattern: Trie đơn luồng không an toàn khi nhiều goroutine ghi đồng thời. Dùng sync.RWMutex — nhiều reader song song, writer độc quyền. Nếu read-heavy (autocomplete), đây là lựa chọn tối ưu.

GO concurrent_trie.go 
package stringalgo

import (
    "context"
    "sync"
    "sync/atomic"
)

// ConcurrentTrie — thread-safe Trie, safe cho multiple readers + concurrent writers.
// Design: một RWMutex toàn cục → đơn giản, phù hợp read-heavy workload.
// Alternative: per-node locking → phức tạp hơn, tốt hơn khi write-heavy.
type ConcurrentTrie struct {
    mu   sync.RWMutex
    root *TrieNode
    ops  atomic.Int64  // metric: tổng số operations
}

func NewConcurrentTrie() *ConcurrentTrie {
    return &ConcurrentTrie{root: &TrieNode{}}
}

// Insert — write lock, O(m)
func (ct *ConcurrentTrie) Insert(word string) {
    ct.mu.Lock()
    defer ct.mu.Unlock()
    ct.ops.Add(1)

    node := ct.root
    for _, ch := range word {
        idx := ch - 'a'
        if node.children[idx] == nil {
            node.children[idx] = &TrieNode{}
        }
        node = node.children[idx]
    }
    node.isEnd = true
}

// Search — read lock, nhiều goroutine có thể search cùng lúc
func (ct *ConcurrentTrie) Search(word string) bool {
    ct.mu.RLock()
    defer ct.mu.RUnlock()
    ct.ops.Add(1)

    node := ct.root
    for _, ch := range word {
        idx := ch - 'a'
        if idx < 0 || idx > 25 || node.children[idx] == nil { return false }
        node = node.children[idx]
    }
    return node.isEnd
}

// BatchInsert — bulk insert với single lock acquisition, hiệu quả hơn insert riêng lẻ
func (ct *ConcurrentTrie) BatchInsert(ctx context.Context, words []string) error {
    ct.mu.Lock()
    defer ct.mu.Unlock()

    for _, w := range words {
        select {
        case <-ctx.Done():
            return ctx.Err()  // respect cancellation ngay cả trong lock
        default:
        }
        node := ct.root
        for _, ch := range w {
            idx := ch - 'a'
            if idx < 0 || idx > 25 { continue }
            if node.children[idx] == nil {
                node.children[idx] = &TrieNode{}
            }
            node = node.children[idx]
        }
        node.isEnd = true
    }
    ct.ops.Add(int64(len(words)))
    return nil
}

// Metrics — đọc atomic counter không cần lock
func (ct *ConcurrentTrie) TotalOps() int64 { return ct.ops.Load() }

package stringalgo

// LongestPalindrome — Expand Around Center, O(n²) time O(1) space
// Xét từng center (gồm cả giữa 2 ký tự), expand ra 2 phía
func LongestPalindrome(s string) string {
    if len(s) == 0 { return "" }

    start, maxLen := 0, 1

    // expand trả về độ dài palindrome với center tại (l, r)
    expand := func(l, r int) {
        for l >= 0 && r < len(s) && s[l] == s[r] {
            if r-l+1 > maxLen {
                maxLen, start = r-l+1, l
            }
            l--
            r++
        }
    }

    for i := 0; i < len(s); i++ {
        expand(i, i)    // odd-length palindrome (center tại i)
        expand(i, i+1) // even-length palindrome (center giữa i và i+1)
    }
    return s[start : start+maxLen]
}

// IsPalindrome kiểm tra toàn bộ string — O(n), bỏ qua non-alphanumeric
func IsPalindromeClean(s string) bool {
    runes := []rune{}
    for _, r := range s {
        if (r >= 'a' && r <= 'z') || (r >= 'A' && r <= 'Z') || (r >= '0' && r <= '9') {
            // normalize: lowercase
            if r >= 'A' && r <= 'Z' { r += 32 }
            runes = append(runes, r)
        }
    }
    for l, r := 0, len(runes)-1; l < r; l, r = l+1, r-1 {
        if runes[l] != runes[r] { return false }
    }
    return true
}

// CountPalindromicSubstrings đếm tất cả substring là palindrome
// "abc" → 3 (a, b, c)
// "aaa" → 6 (a, a, a, aa, aa, aaa)
func CountPalindromicSubstrings(s string) int {
    count := 0
    countExpand := func(l, r int) {
        for l >= 0 && r < len(s) && s[l] == s[r] {
            count++
            l--
            r++
        }
    }
    for i := 0; i < len(s); i++ {
        countExpand(i, i)
        countExpand(i, i+1)
    }
    return count
}

package stringalgo

// LongestPalindromicSubsequence — DP O(n²) time O(n²) space
// dp[i][j] = độ dài LPS của s[i..j]
// Recurrence:
//   s[i]==s[j]: dp[i][j] = dp[i+1][j-1] + 2
//   s[i]!=s[j]: dp[i][j] = max(dp[i+1][j], dp[i][j-1])
func LongestPalindromicSubsequence(s string) int {
    n := len(s)
    // Khởi tạo dp[n][n] — 2D slice
    dp := make([][]int, n)
    for i := range dp {
        dp[i] = make([]int, n)
        dp[i][i] = 1  // single character là palindrome độ dài 1
    }

    for length := 2; length <= n; length++ {
        for i := 0; i <= n-length; i++ {
            j := i + length - 1
            if s[i] == s[j] {
                if length == 2 {
                    dp[i][j] = 2
                } else {
                    dp[i][j] = dp[i+1][j-1] + 2
                }
            } else {
                dp[i][j] = max(dp[i+1][j], dp[i][j-1])
            }
        }
    }
    return dp[0][n-1]
}

// LongestPalindromicSubsequenceOptimized — O(n²) time, O(n) space
// Space optimization: chỉ cần 2 hàng dp thay vì n×n matrix
func LongestPalindromicSubsequenceOptimized(s string) int {
    n := len(s)
    prev := make([]int, n)
    curr := make([]int, n)
    for i := range prev { prev[i] = 1 }

    for i := n - 2; i >= 0; i-- {
        curr[i] = 1
        prevDiag := 0  // dp[i+1][j-1] từ vòng lặp trước
        for j := i + 1; j < n; j++ {
            temp := curr[j]
            if s[i] == s[j] {
                curr[j] = prevDiag + 2
            } else {
                curr[j] = max(prev[j], curr[j-1])
            }
            prevDiag = prev[j]
            prev[j] = temp
        }
        prev, curr = curr, prev
    }
    return prev[n-1]
}

package stringalgo

// Manacher — O(n) time O(n) space, tìm palindrome dài nhất
// Ý tưởng: tận dụng tính đối xứng — nếu đã biết P[mirror], P[i] có bound
//
// Biến quan trọng:
//   c = center của palindrome đang xét đang "bao phủ" xa nhất
//   r = right boundary của palindrome đó (c + P[c])
//   mirror = phần đối xứng của i qua c
func Manacher(s string) string {
    if len(s) == 0 { return "" }

    // Bước 1: Transform "abc" → "#a#b#c#"
    t := transform(s)
    n := len(t)
    p := make([]int, n)  // p[i] = bán kính palindrome tại center t[i]

    c, r := 0, 0  // center và right boundary của palindrome xa nhất

    for i := 0; i < n; i++ {
        if i < r {
            // i nằm trong palindrome [c-P[c], r] → khởi tạo từ mirror
            mirror := 2*c - i
            p[i] = min(r-i, p[mirror])  // không vượt qua r
        }

        // Expand quanh center i (bắt đầu từ p[i] đã biết)
        lo, hi := i-(p[i]+1), i+(p[i]+1)
        for lo >= 0 && hi < n && t[lo] == t[hi] {
            p[i]++
            lo--
            hi++
        }

        // Cập nhật (c, r) nếu palindrome của i mở rộng ra xa hơn
        if i+p[i] > r {
            c, r = i, i+p[i]
        }
    }

    // Bước 2: Tìm center có p[i] lớn nhất
    bestCenter, bestLen := 0, 0
    for i, v := range p {
        if v > bestLen {
            bestLen, bestCenter = v, i
        }
    }

    // Bước 3: Map ngược về string gốc
    // start trong s = (bestCenter - bestLen) / 2
    start := (bestCenter - bestLen) / 2
    return s[start : start+bestLen]
}

// transform chèn '#' để xử lý odd/even uniform
func transform(s string) string {
    b := make([]byte, 2*len(s)+1)
    for i := range b {
        if i%2 == 0 {
            b[i] = '#'
        } else {
            b[i] = s[i/2]
        }
    }
    return string(b)
}

func min(a, b int) int { if a < b { return a }; return b }

// ManacherAllPalindromes trả về tất cả palindromic substrings với range [start, end]
func ManacherAllPalindromes(s string) [][2]int {
    t := transform(s)
    n := len(t)
    p := make([]int, n)
    c, r := 0, 0

    for i := 0; i < n; i++ {
        if i < r { p[i] = min(r-i, p[2*c-i]) }
        for lo, hi := i-p[i]-1, i+p[i]+1; lo >= 0 && hi < n && t[lo] == t[hi]; lo, hi = lo-1, hi+1 {
            p[i]++
        }
        if i+p[i] > r { c, r = i, i+p[i] }
    }

    var result [][2]int
    for i, v := range p {
        if v > 0 {
            start := (i - v) / 2
            result = append(result, [2]int{start, start + v})
        }
    }
    return result
}

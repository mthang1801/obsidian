01
Linear Search — Tìm kiếm tuyến tính
Thuật toán đơn giản nhất: duyệt tuần tự từng phần tử
Basic
📖 Định nghĩa

Linear Search (hay Sequential Search) là thuật toán tìm kiếm đơn giản nhất. Nó duyệt qua từng phần tử trong collection theo thứ tự, so sánh mỗi phần tử với giá trị cần tìm cho đến khi tìm thấy hoặc đã duyệt hết mảng. Không cần data được sorted. Time complexity là O(n) trong worst case.

Best Case
O(1)
Phần tử ở đầu mảng
Average Case
O(n/2)
Phần tử ở giữa mảng
Worst Case
O(n)
Phần tử cuối hoặc không tồn tại
Space
O(1)
Không cần extra memory

// Package search cung cấp các thuật toán tìm kiếm
package search

// LinearSearch tìm kiếm target trong slice, trả về index đầu tiên tìm thấy.
// Trả về -1 nếu không tìm thấy. Time: O(n), Space: O(1)
func LinearSearch(arr []int, target int) int {
    for i, v := range arr {
        if v == target {
            return i
        }
    }
    return -1
}

// LinearSearchAll tìm TẤT CẢ indices chứa target (multi-match)
func LinearSearchAll(arr []int, target int) []int {
    result := make([]int, 0)
    for i, v := range arr {
        if v == target {
            result = append(result, i)
        }
    }
    return result
}

// Example usage
func main() {
    data := []int{3, 1, 9, 7, 5, 2, 7, 8}
    idx := LinearSearch(data, 7)       // → 3
    all := LinearSearchAll(data, 7)    // → [3, 6]
    _ = idx; _ = all
}

package search

import "cmp"

// LinearSearchFunc tìm kiếm theo custom predicate — linh hoạt nhất
// Time: O(n), dùng được với bất kỳ struct nào
func LinearSearchFunc[T any](arr []T, predicate func(T) bool) (T, int, bool) {
    for i, v := range arr {
        if predicate(v) {
            return v, i, true
        }
    }
    var zero T
    return zero, -1, false
}

// LinearSearchOrdered dùng cmp.Ordered constraint — support int, float, string
func LinearSearchOrdered[T cmp.Ordered](arr []T, target T) int {
    for i, v := range arr {
        if v == target {
            return i
        }
    }
    return -1
}

package search

import (
    "context"
    "runtime"
    "sync"
)

// ConcurrentLinearSearch chia array thành N chunks, tìm kiếm song song.
// Sử dụng context để cancel các goroutines khi đã tìm thấy kết quả.
// Hiệu quả với dataset rất lớn (>1M phần tử).
func ConcurrentLinearSearch[T comparable](ctx context.Context, arr []T, target T) (int, bool) {
    if len(arr) == 0 {
        return -1, false
    }

    numWorkers := runtime.NumCPU()
    chunkSize := (len(arr) + numWorkers - 1) / numWorkers

    type result struct {
        index int
        found bool
    }

    ctx, cancel := context.WithCancel(ctx)
    defer cancel() // đảm bảo cleanup

    resultCh := make(chan result, numWorkers)
    var wg sync.WaitGroup

    for w := 0; w < numWorkers; w++ {
        start := w * chunkSize
        end := min(start+chunkSize, len(arr))
        if start >= len(arr) {
            break
        }
        wg.Add(1)
        go func(chunk []T, offset int) {
            defer wg.Done()
            for i, v := range chunk {
                select {
                case <-ctx.Done():
                    return // Đã có worker khác tìm thấy, dừng lại
                default:
                    if v == target {
                        cancel() // Thông báo cho tất cả workers dừng
                        resultCh <- result{offset + i, true}
                        return
                    }
                }
            }
        }(arr[start:end], start)
    }

    go func() {
        wg.Wait()
        close(resultCh)
    }()

    for r := range resultCh {
        if r.found {
            return r.index, true
        }
    }
    return -1, false
}

02
Binary Search — Tìm kiếm nhị phân
Divide & Conquer — chia đôi search space mỗi bước
Advanced
📖 Định nghĩa

Binary Search là thuật toán tìm kiếm hiệu quả trên sorted array. Mỗi iteration, nó so sánh target với phần tử ở giữa mảng, loại bỏ nửa không chứa target. Kỹ thuật Divide and Conquer này giảm search space từ n → n/2 → n/4 → ... → 1, đạt time complexity O(log n). Đây là một trong những thuật toán quan trọng nhất trong Computer Science.

Best Case
O(1)
Target ở mid ngay lần đầu
Average / Worst
O(log n)
Mỗi bước giảm ½ search space
Space (Iterative)
O(1)
Không dùng extra memory
Space (Recursive)
O(log n)
Call stack depth = log n

package search

// BinarySearch tìm kiếm target trong sorted slice (ascending).
// Trả về index và bool. Time: O(log n), Space: O(1)
// QUAN TRỌNG: arr phải được sorted trước khi gọi hàm này.
func BinarySearch(arr []int, target int) (int, bool) {
    low, high := 0, len(arr)-1

    for low <= high {
        // Tránh integer overflow: dùng low + (high-low)/2 thay vì (low+high)/2
        mid := low + (high-low)/2

        switch {
        case arr[mid] == target:
            return mid, true
        case arr[mid] < target:
            low = mid + 1 // Target ở nửa phải
        default:
            high = mid - 1 // Target ở nửa trái
        }
    }
    return -1, false
}

// BinarySearchFirst tìm FIRST occurrence (quan trọng với duplicate values)
func BinarySearchFirst(arr []int, target int) int {
    low, high, result := 0, len(arr)-1, -1
    for low <= high {
        mid := low + (high-low)/2
        if arr[mid] == target {
            result = mid
            high = mid - 1 // Tiếp tục tìm bên trái
        } else if arr[mid] < target {
            low = mid + 1
        } else {
            high = mid - 1
        }
    }
    return result
}

// BinarySearchLast tìm LAST occurrence
func BinarySearchLast(arr []int, target int) int {
    low, high, result := 0, len(arr)-1, -1
    for low <= high {
        mid := low + (high-low)/2
        if arr[mid] == target {
            result = mid
            low = mid + 1 // Tiếp tục tìm bên phải
        } else if arr[mid] < target {
            low = mid + 1
        } else {
            high = mid - 1
        }
    }
    return result
}

package search

import (
    "cmp"
    "sort"
)

// BinarySearchGeneric — generic binary search với cmp.Ordered constraint
func BinarySearchGeneric[T cmp.Ordered](arr []T, target T) (int, bool) {
    low, high := 0, len(arr)-1
    for low <= high {
        mid := low + (high-low)/2
        switch cmp.Compare(arr[mid], target) {
        case 0:  return mid, true
        case -1: low = mid + 1
        case 1:  high = mid - 1
        }
    }
    return -1, false
}

// LowerBound: tìm index nhỏ nhất i sao cho arr[i] >= target (C++ lower_bound)
func LowerBound[T cmp.Ordered](arr []T, target T) int {
    return sort.Search(len(arr), func(i int) bool {
        return cmp.Compare(arr[i], target) >= 0
    })
}

// UpperBound: tìm index nhỏ nhất i sao cho arr[i] > target (C++ upper_bound)
func UpperBound[T cmp.Ordered](arr []T, target T) int {
    return sort.Search(len(arr), func(i int) bool {
        return cmp.Compare(arr[i], target) > 0
    })
}

// EqualRange: trả về [lo, hi) range chứa tất cả elements == target
func EqualRange[T cmp.Ordered](arr []T, target T) (int, int) {
    return LowerBound(arr, target), UpperBound(arr, target)
}

// ─── Ví dụ thực tế: Binary Search trên sorted struct slice ───

type Product struct {
    ID    int
    Price float64
    Name  string
}

// FindProductByID tìm product trong sorted-by-ID slice bằng binary search
func FindProductByID(products []Product, id int) (Product, bool) {
    i := sort.Search(len(products), func(j int) bool {
        return products[j].ID >= id
    })
    if i < len(products) && products[i].ID == id {
        return products[i], true
    }
    return Product{}, false
}

// Usage:
// lo, hi := EqualRange([]int{1,2,2,2,3,4}, 2) → lo=1, hi=4
// Tức là arr[1:4] = [2,2,2] chứa tất cả giá trị 2
⚠️
Integer Overflow Bug — Pitfall phổ biến

Công thức mid = (low + high) / 2 có thể gây integer overflow khi low + high vượt quá math.MaxInt. Luôn dùng mid = low + (high - low) / 2. Đây là bug kinh điển từng xuất hiện trong JDK binary search suốt 9 năm (Joshua Bloch, 2006).

03
Jump Search — Tìm kiếm nhảy bước
Nhảy theo block √n rồi linear search trong block tìm được
Advanced
📖 Định nghĩa

Jump Search hoạt động trên sorted array bằng cách nhảy theo từng block có kích thước √n. Khi tìm được block có thể chứa target (arr[jump] >= target), thực hiện linear search ngược lại trong block đó. Ý tưởng: tối ưu kích thước block để minimize tổng số so sánh — block_size = √n cho tổng số bước ít nhất.

Best Case
O(1)
Target ở block đầu tiên
Average / Worst
O(√n)
Block jumps + linear scan
Space
O(1)
Constant extra space
Optimal block
√n
math.Sqrt(float64(n))

package search

import "math"

// JumpSearch tìm kiếm trong sorted slice bằng cách nhảy theo block √n.
// Time: O(√n), Space: O(1). Phù hợp khi backward jumps có chi phí cao.
func JumpSearch(arr []int, target int) int {
    n := len(arr)
    if n == 0 {
        return -1
    }

    step := int(math.Sqrt(float64(n)))
    prev := 0

    // Phase 1: Jump forward theo block size √n
    for prev < n && arr[min(step, n)-1] < target {
        prev = step
        step += int(math.Sqrt(float64(n)))
        if prev >= n {
            return -1
        }
    }

    // Phase 2: Linear search trong block [prev, min(step, n))
    limit := min(step, n)
    for i := prev; i < limit; i++ {
        if arr[i] == target {
            return i
        }
        if arr[i] > target {
            return -1 // Sorted array → target không tồn tại
        }
    }
    return -1
}

// JumpSearchCustomStep cho phép chỉ định custom block size
func JumpSearchCustomStep(arr []int, target, blockSize int) int {
    n := len(arr)
    if n == 0 || blockSize <= 0 {
        return -1
    }
    prev, step := 0, blockSize
    for prev < n && arr[min(step, n)-1] < target {
        prev = step
        step += blockSize
        if prev >= n { return -1 }
    }
    for i := prev; i < min(step, n); i++ {
        if arr[i] == target { return i }
        if arr[i] > target  { return -1 }
    }
    return -1
}
04
Interpolation Search — Tìm kiếm nội suy
Ước tính vị trí target dựa trên phân phối giá trị
Expert
📖 Định nghĩa

Interpolation Search là phiên bản cải tiến của Binary Search dành cho data phân bố đều (uniformly distributed). Thay vì luôn chọn midpoint, nó ước tính vị trí của target theo công thức nội suy tuyến tính: pos = low + (target - arr[low]) * (high - low) / (arr[high] - arr[low]). Tương tự như cách ta tìm số trang trong sách — ta ước lượng trang cần mở dựa trên giá trị bảng chữ cái.

Best Case
O(1)
Target ước lượng đúng ngay lần đầu
Average (uniform)
O(log log n)
Tốt hơn Binary Search!
Worst Case
O(n)
Data không đều (skewed)
Space
O(1)
Constant space

package search

// InterpolationSearch tìm kiếm trong sorted slice với data phân bố đều.
// Time: O(log log n) average cho uniform data, O(n) worst case.
// QUAN TRỌNG: Chỉ hiệu quả khi data phân bố đều (uniform distribution).
func InterpolationSearch(arr []int, target int) int {
    low, high := 0, len(arr)-1

    for low <= high && target >= arr[low] && target <= arr[high] {
        if low == high {
            if arr[low] == target { return low }
            return -1
        }

        // Công thức nội suy: ước tính vị trí dựa trên tỉ lệ giá trị
        // Tránh division by zero khi arr[high] == arr[low]
        rangeDiff := arr[high] - arr[low]
        if rangeDiff == 0 {
            if arr[low] == target { return low }
            return -1
        }

        pos := low + (target-arr[low])*(high-low) / rangeDiff

        // Clamp pos trong [low, high] để tránh out-of-bounds
        if pos < low { pos = low }
        if pos > high { pos = high }

        switch {
        case arr[pos] == target:
            return pos
        case arr[pos] < target:
            low = pos + 1
        default:
            high = pos - 1
        }
    }
    return -1
}

// InterpolationSearchFloat64 cho float64 data (cần xử lý floating point)
func InterpolationSearchFloat64(arr []float64, target float64) int {
    low, high := 0, len(arr)-1

    for low <= high && target >= arr[low] && target <= arr[high] {
        if low == high {
            if arr[low] == target { return low }
            return -1
        }
        rangeDiff := arr[high] - arr[low]
        if rangeDiff == 0 { break }

        ratio := (target - arr[low]) / rangeDiff
        pos := low + int(ratio*float64(high-low))

        if pos < low { pos = low }
        if pos > high { pos = high }

        switch {
        case arr[pos] == target: return pos
        case arr[pos] < target:  low = pos + 1
        default:                 high = pos - 1
        }
    }
    return -1
}
🚫
Khi nào KHÔNG dùng Interpolation Search

Nếu data không phân bố đều (ví dụ: exponential growth, Zipf distribution, highly skewed), Interpolation Search có thể hoạt động tệ hơn Binary Search với O(n) trong worst case. Hãy luôn profile data trước khi chọn thuật toán này.

05
Exponential Search — Tìm kiếm hàm mũ
Tìm bound theo lũy thừa 2, rồi Binary Search trong range đó
Expert
📖 Định nghĩa

Exponential Search gồm 2 phase: Phase 1 — tìm range [2^(k-1), 2^k] chứa target bằng cách nhân đôi index (1 → 2 → 4 → 8 → 16 → ...) cho đến khi arr[i] >= target. Phase 2 — thực hiện Binary Search trong range tìm được. Đặc biệt hữu ích với unbounded arrays hoặc khi target nằm gần đầu mảng. Cũng là nền tảng của nhiều search engine và index structure.

Best Case
O(1)
Target ở index đầu tiên
Average / Worst
O(log n)
Phase1: O(log n) + Phase2: O(log n)
Space
O(1)
Iterative binary search
Bounded range
2·log(pos)
Chỉ search đến vị trí target

package search

// ExponentialSearch tìm kiếm trong sorted slice.
// Time: O(log n), Space: O(1).
// Đặc biệt hiệu quả khi target ở gần đầu hoặc với unbounded arrays.
func ExponentialSearch(arr []int, target int) int {
    n := len(arr)
    if n == 0 { return -1 }
    if arr[0] == target { return 0 }

    // Phase 1: Tìm range [bound/2, bound] chứa target
    bound := 1
    for bound < n && arr[bound] < target {
        bound *= 2 // Mũ hóa: 1, 2, 4, 8, 16, ...
    }

    // Phase 2: Binary search trong [bound/2, min(bound, n-1)]
    low  := bound / 2
    high := min(bound, n-1)
    idx, _ := binarySearchRange(arr, low, high, target)
    return idx
}

// binarySearchRange là binary search trong range [low, high]
func binarySearchRange(arr []int, low, high, target int) (int, bool) {
    for low <= high {
        mid := low + (high-low)/2
        switch {
        case arr[mid] == target: return mid, true
        case arr[mid] < target:  low = mid + 1
        default:                 high = mid - 1
        }
    }
    return -1, false
}

// ─── Unbounded Array: dành cho streaming / infinite data sources ───

// ArrayAccessor là abstraction cho unbounded data source (database cursor, stream...)
type ArrayAccessor[T any] interface {
    Get(index int) (T, error)  // Returns error nếu index out of range
    Size() int               // -1 nếu unknown (infinite)
}

// ExponentialSearchUnbounded hoạt động với data source không biết kích thước
func ExponentialSearchUnbounded(src ArrayAccessor[int], target int) (int, error) {
    // Kiểm tra phần tử đầu tiên
    first, err := src.Get(0)
    if err != nil { return -1, err }
    if first == target { return 0, nil }

    bound := 1
    for {
        val, err := src.Get(bound)
        if err != nil {
            // Đã vượt qua kích thước mảng → binary search trong [bound/2, bound-1]
            break
        }
        if val >= target { break }
        bound *= 2
    }

    // Binary search trong [bound/2, bound) với safe access
    low, high := bound/2, bound-1
    for low <= high {
        mid := low + (high-low)/2
        val, err := src.Get(mid)
        if err != nil { high = mid - 1; continue }
        switch {
        case val == target: return mid, nil
        case val < target:  low = mid + 1
        default:            high = mid - 1
        }
    }
    return -1, nil
}

package search

import (
    "cmp"
    "fmt"
    "math"
    "time"
)

// Strategy enum cho các thuật toán tìm kiếm
type Algorithm string

const (
    AlgoLinear        Algorithm = "linear"
    AlgoBinary        Algorithm = "binary"
    AlgoJump          Algorithm = "jump"
    AlgoInterpolation Algorithm = "interpolation"
    AlgoExponential   Algorithm = "exponential"
)

// SearchResult chứa kết quả và metadata
type SearchResult[T any] struct {
    Index       int
    Value       T
    Found       bool
    Algorithm   Algorithm
    Comparisons int
    Duration    time.Duration
}

func (r SearchResult[T]) String() string {
    if !r.Found {
        return fmt.Sprintf("[%s] Not found | comparisons=%d | time=%v",
            r.Algorithm, r.Comparisons, r.Duration)
    }
    return fmt.Sprintf("[%s] Found at index=%d | comparisons=%d | time=%v",
        r.Algorithm, r.Index, r.Comparisons, r.Duration)
}

// SearchEngine — unified interface cho tất cả algorithms
type SearchEngine[T cmp.Ordered] struct {
    algo Algorithm
}

func NewSearchEngine[T cmp.Ordered](algo Algorithm) *SearchEngine[T] {
    return &SearchEngine[T]{algo: algo}
}

// Search thực hiện tìm kiếm và trả về SearchResult với metadata
func (e *SearchEngine[T]) Search(arr []T, target T) SearchResult[T] {
    start := time.Now()
    comparisons := 0

    var idx int
    switch e.algo {
    case AlgoLinear:
        idx = linearWithCount(arr, target, &comparisons)
    case AlgoBinary:
        idx = binaryWithCount(arr, target, &comparisons)
    case AlgoExponential:
        idx = exponentialWithCount(arr, target, &comparisons)
    default:
        idx = binaryWithCount(arr, target, &comparisons)
    }

    result := SearchResult[T]{
        Index:       idx,
        Found:       idx != -1,
        Algorithm:   e.algo,
        Comparisons: comparisons,
        Duration:    time.Since(start),
    }
    if result.Found {
        result.Value = arr[idx]
    }
    return result
}

func linearWithCount[T cmp.Ordered](arr []T, target T, c *int) int {
    for i, v := range arr {
        *c++
        if v == target { return i }
    }
    return -1
}

func binaryWithCount[T cmp.Ordered](arr []T, target T, c *int) int {
    lo, hi := 0, len(arr)-1
    for lo <= hi {
        *c++
        mid := lo + (hi-lo)/2
        switch cmp.Compare(arr[mid], target) {
        case 0:  return mid
        case -1: lo = mid+1
        case 1:  hi = mid-1
        }
    }
    return -1
}

func exponentialWithCount[T cmp.Ordered](arr []T, target T, c *int) int {
    n := len(arr)
    if n == 0 { return -1 }
    *c++
    if arr[0] == target { return 0 }
    bound := 1
    for bound < n {
        *c++
        if cmp.Compare(arr[bound], target) >= 0 { break }
        bound *= 2
    }
    return binaryWithCount(arr[bound/2:min(bound+1,n)], target, c)
}

// OptimalStep tính block size tối ưu cho Jump Search
func OptimalStep(n int) int {
    return int(math.Sqrt(float64(n)))
}

package search_test

// ─── Table-driven Tests ───

func TestAllSearchAlgorithms(t *testing.T) {
    type testCase struct {
        name     string
        arr      []int
        target   int
        wantIdx  int
    }

    tests := []testCase{
        {"empty array",        []int{},          1,  -1},
        {"single element hit", []int{42},        42, 0},
        {"single element miss",[]int{42},        1,  -1},
        {"first element",      []int{1,3,5,7},  1,  0},
        {"last element",       []int{1,3,5,7},  7,  3},
        {"middle element",     []int{1,3,5,7},  3,  1},
        {"not found",          []int{1,3,5,7},  4,  -1},
        {"large sorted", generateSorted(10000), 7777, 3888},
    }

    algorithms := []struct{
        name string
        fn   func([]int, int) int
    }{
        {"Linear",        search.LinearSearch},
        {"Binary",        func(a []int, t int) int { i, _ := search.BinarySearch(a, t); return i }},
        {"Jump",          search.JumpSearch},
        {"Interpolation", search.InterpolationSearch},
        {"Exponential",   search.ExponentialSearch},
    }

    for _, algo := range algorithms {
        for _, tc := range tests {
            t.Run(algo.name+"/"+tc.name, func(t *testing.T) {
                got := algo.fn(tc.arr, tc.target)
                if got != tc.wantIdx {
                    t.Errorf("got=%d, want=%d", got, tc.wantIdx)
                }
            })
        }
    }
}

func generateSorted(n int) []int {
    data := make([]int, n)
    for i := range data { data[i] = rand.Intn(n * 2) }
    sort.Ints(data)

Tình huống Thuật toán được đề xuất Lý do

Sorted array, general case Binary Search O(log n) đáng tin cậy, không có worst case bad
Data phân bố đều, sorted Interpolation Search O(log log n) average — nhanh nhất trong điều kiện lý tưởng
Target thường ở gần đầu Exponential Search Phase 1 nhanh khi bound nhỏ, ideal cho unbounded
Backward jump tốn kém (tape) Jump Search Minimize backward operations với O(√n)
Unsorted, single search Linear Search Sort + Binary Search = O(n log n) > O(n)
Unsorted, nhiều queries Sort + Binary Search Sort một lần O(n log n), sau đó O(log n)/query

Thực tế trong Go — Dùng stdlib khi có thể

Go stdlib đã có sort.Search() (binary search), slices.BinarySearch() (Go 1.21+), và slices.Contains(). Chỉ tự implement khi cần tối ưu đặc biệt (custom comparator, multi-search, concurrent, hoặc non-standard data structures). Nguyên tắc: không reinvent the wheel mà không có lý do.

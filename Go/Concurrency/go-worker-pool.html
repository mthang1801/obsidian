Giới thiệu Worker Pool Pattern

Cơ bản
Hình 1.1: Sơ đồ hoạt động của Bounded Worker Pool Pattern — Các Producer ghi công việc vào hàng đợi (Buffered Job Channel), điều phối song song cho các Worker Goroutine tái sử dụng và kiểm soát bằng WaitGroup để Janitor Goroutine đóng kết quả an toàn.

Worker Pool (hay Thread Pool) là một pattern concurrency kinh điển: thay vì spawn một goroutine mới cho mỗi task (tốn memory, không kiểm soát được), ta tạo sẵn một tập cố định các worker goroutine và phân phối công việc qua job channel. Pattern này là nền tảng của nhiều hệ thống xử lý tác vụ song song — tương đương với BullMQ (Node.js) nhưng được triển khai thuần Go với goroutine và channel.

💡
Go Concurrency Mantra: "Don't communicate by sharing memory; share memory by communicating." Worker Pool hiện thực hóa triết lý này: các worker không share state — chúng nhận job qua channel và trả result qua channel.

Khái niệm cốt lõi

Trước khi viết code, cần nắm rõ các thành phần tạo nên một Worker Pool trong Go:

Thành phần Kiểu dữ liệu Go Vai trò
Job chan Job Đơn vị công việc được phân phối cho worker. Producer ghi vào channel này.
Worker goroutine Vòng lặp vô hạn đọc job từ channel, xử lý, ghi kết quả ra result channel.
Result chan Result Kết quả xử lý (bao gồm lỗi nếu có). Consumer đọc từ channel này.
WaitGroup sync.WaitGroup Đồng bộ: chờ tất cả worker hoàn thành trước khi đóng result channel.
Context context.Context Cancellation signal: worker kiểm tra ctx.Done() để dừng gracefully.
Pool Size int (numWorkers) Số lượng worker cố định. Thường = runtime.NumCPU() hoặc bội số.
⚠️
Vì sao KHÔNG spawn goroutine cho mỗi task? Mỗi goroutine tốn ~2KB stack ban đầu. 100,000 task = ~200MB chỉ riêng stack. Ngoài ra scheduler overhead tăng đáng kể. Worker Pool giải quyết bằng cách tái sử dụng goroutine đã có.

Cấu trúc một Worker

Cơ bản

Một worker trong Go có cấu trúc vòng lặp for-select chuẩn, xử lý 3 sự kiện chính:

Vòng Lặp Xử Lý Của Một Worker Goroutine
GO worker_anatomy.go — Cấu trúc chuẩn của một worker 
// worker là goroutine sống suốt vòng đời của pool.
// Nó chỉ kết thúc khi: (1) ctx bị cancel, (2) jobs channel bị đóng.
func worker(
    ctx    context.Context,
    id     int,
    jobs   <-chan Job,
    results chan<- Result,
    wg     *sync.WaitGroup,
) {
    defer wg.Done() // báo hiệu khi worker kết thúc

    for {
        select {
        // Sự kiện 1: context bị cancel (timeout / shutdown)
        case <-ctx.Done():
            return

        // Sự kiện 2: nhận job từ channel
        case job, ok := <-jobs:
            if !ok {
                return // jobs channel đã đóng → thoát gracefully
            }
            result := processJob(ctx, job)
            select {
            case results <- result:
            case <-ctx.Done():
                return // tránh block khi ghi result
            }
        }
    }
}

Worker Pool cơ bản

Cơ bản

Implementation đơn giản nhất: N workers cố định, một job channel, một result channel. Đây là nền tảng của mọi biến thể phức tạp hơn.

GO worker_pool_basic.go 
package workerpool

import (
    "context"
    "fmt"
    "sync"
    "time"
)

// Job đại diện cho một đơn vị công việc.
type Job struct {
    ID      int
    Payload string
}

// Result chứa kết quả xử lý của một Job.
type Result struct {
    JobID    int
    Output   string
    Duration time.Duration
    Err      error
}

// WorkerPool khởi động numWorkers goroutine và trả về result channel.
func WorkerPool(ctx context.Context, numWorkers int, jobs <-chan Job) <-chan Result {
    results := make(chan Result, numWorkers) // buffered: không block worker khi ghi

    var wg sync.WaitGroup
    for i := 0; i < numWorkers; i++ {
        wg.Add(1)
        go func(workerID int) {
            defer wg.Done()
            for {
                select {
                case <-ctx.Done():
                    return
                case job, ok := <-jobs:
                    if !ok {
                        return
                    }
                    start := time.Now()
                    output, err := process(ctx, job)
                    results <- Result{
                        JobID:    job.ID,
                        Output:   output,
                        Duration: time.Since(start),
                        Err:      err,
                    }
                }
            }
        }(i)
    }

    // Goroutine "janitor": đóng result channel sau khi tất cả worker xong
    go func() {
        wg.Wait()
        close(results)
    }()

    return results
}

func process(ctx context.Context, job Job) (string, error) {
    // Mô phỏng công việc tốn thời gian
    select {
    case <-time.After(100 * time.Millisecond):
        return fmt.Sprintf("processed:%s", job.Payload), nil
    case <-ctx.Done():
        return "", ctx.Err()
    }
}

// ── USAGE ───────────────────────────────────────────────────────────
func main() {
    ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    // Tạo job channel và gửi 20 jobs
    jobs := make(chan Job, 20)
    go func() {
        defer close(jobs) // QUAN TRỌNG: chỉ đóng từ phía sender
        for i := 1; i <= 20; i++ {
            jobs <- Job{ID: i, Payload: fmt.Sprintf("task-%d", i)}
        }
    }()

    // Khởi động pool với 5 workers
    results := WorkerPool(ctx, 5, jobs)

    // Collect results
    for r := range results {
        if r.Err != nil {
            fmt.Printf("Job %d FAILED: %v\n", r.JobID, r.Err)
            continue
        }
        fmt.Printf("Job %d OK: %s (%v)\n", r.JobID, r.Output, r.Duration)
    }
}

Multi-Stage Worker Pool

Nâng cao

Kết hợp nhiều Worker Pool thành pipeline: output của stage trước là input của stage sau. Pattern này phù hợp với các luồng xử lý ETL, image processing, event streaming.

Multi-Stage Pipeline Architecture
GO multi_stage_pool.go 
package pipeline

import (
    "context"
    "sync"
)

type RawData   struct { ID int; Source string }
type Processed struct { ID int; Value  float64 }
type Stored    struct { ID int; Saved  bool }

// stage tạo một pool worker chạy hàm fn trên mỗi input item.
// Generic helper để tái sử dụng pattern.
func stage[In, Out any](
    ctx     context.Context,
    workers int,
    in      <-chan In,
    fn      func(context.Context, In) (Out, error),
) <-chan Out {
    out := make(chan Out, workers)
    var wg sync.WaitGroup

    for i := 0; i < workers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for item := range in {
                select {
                case <-ctx.Done():
                    return
                default:
                }
                result, err := fn(ctx, item)
                if err != nil {
                    continue // hoặc gửi sang error channel
                }
                select {
                case out <- result:
                case <-ctx.Done():
                    return
                }
            }
        }()
    }

    go func() { wg.Wait(); close(out) }()
    return out
}

// RunPipeline kết hợp 3 stage thành một pipeline hoàn chỉnh.
func RunPipeline(ctx context.Context, raw <-chan RawData) <-chan Stored {
    // Stage 1: Extract (2 workers)
    extracted := stage(ctx, 2, raw, extractFn)

    // Stage 2: Transform (4 workers — bottleneck thường ở đây)
    transformed := stage(ctx, 4, extracted, transformFn)

    // Stage 3: Load (2 workers)
    stored := stage(ctx, 2, transformed, loadFn)

    return stored
}

Priority Worker Pool

Nâng cao

Tương tự priority queue trong BullMQ, các job có độ ưu tiên cao được xử lý trước. Trong Go, ta dùng heap (container/heap) hoặc multiple channel với select để implement priority.

GO priority_pool.go 
package priority

import (
    "container/heap"
    "context"
    "sync"
)

type Priority int

const (
    Low      Priority = 1
    Normal   Priority = 5
    High     Priority = 10
    Critical Priority = 100
)

type PriorityJob struct {
    ID       int
    Priority Priority
    Payload  any
    index    int // heap internal
}

// Implement heap.Interface
type jobHeap [*PriorityJob]

func (h jobHeap) Len() int           { return len(h) }
func (h jobHeap) Less(i, j int) bool { return h[i].Priority > h[j].Priority } // max-heap
func (h jobHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i]; h[i].index, h[j].index = i, j }
func (h *jobHeap) Push(x any) {
    n := len(*h)
    item := x.(*PriorityJob)
    item.index = n
    *h = append(*h, item)
}
func (h *jobHeap) Pop() any {
    old := *h
    n := len(old)
    item := old[n-1]
    old[n-1] = nil
    item.index = -1
    *h = old[0 : n-1]
    return item
}

// PriorityPool phân phối job theo priority, worker được ready thì nhận job cao nhất.
type PriorityPool struct {
    mu      sync.Mutex
    cond    *sync.Cond
    heap    jobHeap
    closed  bool
    workers int
}

func NewPriorityPool(workers int) *PriorityPool {
    p := &PriorityPool{workers: workers}
    p.cond = sync.NewCond(&p.mu)
    return p
}

func (p *PriorityPool) Submit(job *PriorityJob) {
    p.mu.Lock()
    heap.Push(&p.heap, job)
    p.mu.Unlock()
    p.cond.Signal() // đánh thức một worker đang chờ
}

func (p *PriorityPool) next() (*PriorityJob, bool) {
    p.mu.Lock()
    defer p.mu.Unlock()
    for p.heap.Len() == 0 && !p.closed {
        p.cond.Wait() // ngủ cho đến khi có job mới
    }
    if p.heap.Len() == 0 {
        return nil, false
    }
    return heap.Pop(&p.heap).(*PriorityJob), true
}

func (p *PriorityPool) Run(ctx context.Context, handler func(context.Context, *PriorityJob)) {
    var wg sync.WaitGroup
    for i := 0; i < p.workers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for {
                job, ok := p.next()
                if !ok {
                    return
                }
                handler(ctx, job)
            }
        }()
    }
    wg.Wait()
}

func (p *PriorityPool) Close() {
    p.mu.Lock()
    p.closed = true
    p.mu.Unlock()
    p.cond.Broadcast() // đánh thức tất cả workers để thoát
}

Retry + Dead Letter Queue

Nâng cao

Khi job thất bại, ta thử lại với exponential backoff. Sau số lần retry tối đa, job được chuyển sang Dead Letter Queue (DLQ) để phân tích sau — pattern tương đương BullMQ failedJobsHistory.

GO retry_pool.go — Exponential Backoff + Dead Letter 
package retry

import (
    "context"
    "fmt"
    "math"
    "sync"
    "time"
)

type RetryJob struct {
    ID         int
    Payload    any
    Attempts   int       // số lần đã thử
    MaxRetries int       // tối đa
    LastError  error     // lỗi gần nhất
    NextRetry  time.Time // thời điểm retry tiếp theo
}

// backoff tính thời gian chờ theo exponential: 2^attempt * base
func backoff(attempt int, base time.Duration) time.Duration {
    exp := time.Duration(math.Pow(2, float64(attempt)))
    d := base * exp
    if d > 30*time.Second {
        return 30 * time.Second // cap ở 30s
    }
    return d
}

type RetryPool struct {
    jobs    chan *RetryJob
    dlq     chan *RetryJob // Dead Letter Queue
    workers int
    base    time.Duration
}

func NewRetryPool(workers, queueSize int, base time.Duration) *RetryPool {
    return &RetryPool{
        jobs:    make(chan *RetryJob, queueSize),
        dlq:     make(chan *RetryJob, queueSize),
        workers: workers,
        base:    base,
    }
}

func (p *RetryPool) Submit(job *RetryJob) { p.jobs <- job }
func (p *RetryPool) DLQ() <-chan *RetryJob   { return p.dlq }

func (p *RetryPool) Run(ctx context.Context, handler func(context.Context, *RetryJob) error) {
    var wg sync.WaitGroup
    for i := 0; i < p.workers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for {
                select {
                case <-ctx.Done():
                    return
                case job, ok := <-p.jobs:
                    if !ok {
                        return
                    }
                    // Chờ đến thời điểm retry
                    if wait := time.Until(job.NextRetry); wait > 0 {
                        select {
                        case <-time.After(wait):
                        case <-ctx.Done():
                            return
                        }
                    }
                    err := handler(ctx, job)
                    if err == nil {
                        continue // thành công
                    }
                    job.Attempts++
                    job.LastError = err
                    if job.Attempts >= job.MaxRetries {
                        // Hết retry → gửi vào DLQ
                        select {
                        case p.dlq <- job:
                        default:
                            fmt.Printf("DLQ full, dropping job %d\n", job.ID)
                        }
                        continue
                    }
                    // Lên lịch retry với backoff
                    job.NextRetry = time.Now().Add(backoff(job.Attempts, p.base))
                    go func(j *RetryJob) { p.jobs <- j }(job)
                }
            }
        }()
    }
    wg.Wait()
    close(p.dlq)
}

Rate Limiting Worker Pool

Nâng cao

Kiểm soát throughput bằng Token Bucket hoặc golang.org/x/time/rate. Hữu ích khi gọi external API có rate limit, tránh overwhelming downstream services.

GO rate_limited_pool.go — Token Bucket Rate Limiting 
package ratelimit

import (
    "context"
    "sync"
    "golang.org/x/time/rate"
)

type RateLimitedPool struct {
    limiter *rate.Limiter
    workers int
    jobs    chan Job
    results chan Result
}

// NewRateLimitedPool tạo pool với rate limit rps requests/giây,
// burst = số request tối đa được phép burst.
func NewRateLimitedPool(workers, rps, burst int) *RateLimitedPool {
    return &RateLimitedPool{
        limiter: rate.NewLimiter(rate.Limit(rps), burst),
        workers: workers,
        jobs:    make(chan Job, workers*2),
        results: make(chan Result, workers*2),
    }
}

func (p *RateLimitedPool) Run(ctx context.Context) <-chan Result {
    var wg sync.WaitGroup
    for i := 0; i < p.workers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for job := range p.jobs {
                // Chờ token từ rate limiter
                if err := p.limiter.Wait(ctx); err != nil {
                    return // ctx cancelled
                }
                result := processJob(ctx, job)
                select {
                case p.results <- result:
                case <-ctx.Done():
                    return
                }
            }
        }()
    }
    go func() { wg.Wait(); close(p.results) }()
    return p.results
}

// Cách dùng: pool tối đa 10 req/s, burst 20, với 5 workers

Dynamic Scaling Worker Pool

Nâng cao

Tự động điều chỉnh số lượng worker dựa theo queue depth. Khi load tăng, spawn thêm worker. Khi nhàn, thu hồi worker để tiết kiệm tài nguyên.

Dynamic Scaling — Vòng Điều Chỉnh Tự Động
GO dynamic_pool.go — Auto-scaling Worker Pool 
package dynamic

import (
    "context"
    "runtime"
    "sync"
    "sync/atomic"
    "time"
)

type DynamicPool struct {
    jobs        chan Job
    results     chan Result
    minWorkers  int
    maxWorkers  int
    activeCount atomic.Int64
    mu          sync.Mutex
    cancel      map[int64]context.CancelFunc
    nextID      atomic.Int64
}

func NewDynamicPool(min, max, queueSize int) *DynamicPool {
    p := &DynamicPool{
        jobs:       make(chan Job, queueSize),
        results:    make(chan Result, queueSize),
        minWorkers: min,
        maxWorkers: max,
        cancel:     make(map[int64]context.CancelFunc),
    }
    return p
}

func (p *DynamicPool) spawnWorker(ctx context.Context) {
    id := p.nextID.Add(1)
    wCtx, wCancel := context.WithCancel(ctx)

    p.mu.Lock()
    p.cancel[id] = wCancel
    p.mu.Unlock()

    p.activeCount.Add(1)
    go func() {
        defer func() {
            p.activeCount.Add(-1)
            p.mu.Lock()
            delete(p.cancel, id)
            p.mu.Unlock()
        }()
        for {
            select {
            case <-wCtx.Done():
                return
            case job, ok := <-p.jobs:
                if !ok {
                    return
                }
                p.results <- processJob(wCtx, job)
            }
        }
    }()
}

// scaleDown dừng một worker tùy ý (không phá hủy job đang chạy).
func (p *DynamicPool) scaleDown() {
    p.mu.Lock()
    defer p.mu.Unlock()
    for id, cancel := range p.cancel {
        cancel()
        delete(p.cancel, id)
        return // chỉ xóa 1 worker mỗi lần
    }
}

// autoScale chạy vòng lặp điều chỉnh theo queue depth.
func (p *DynamicPool) autoScale(ctx context.Context) {
    ticker := time.NewTicker(500 * time.Millisecond)
    defer ticker.Stop()
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            queueLen := len(p.jobs)
            queueCap := cap(p.jobs)
            load := float64(queueLen) / float64(queueCap)
            active := p.activeCount.Load()

            switch {
            case load > 0.8 && active < int64(p.maxWorkers):
                // Scale up: spawn thêm worker
                toAdd := int64(runtime.NumCPU())
                if active+toAdd > int64(p.maxWorkers) {
                    toAdd = int64(p.maxWorkers) - active
                }
                for i := int64(0); i < toAdd; i++ {
                    p.spawnWorker(ctx)
                }

            case load < 0.2 && active > int64(p.minWorkers):
                // Scale down: dừng một worker
                p.scaleDown()
            }
        }
    }
}

Graceful Shutdown

Nâng cao

Khi nhận SIGTERM/SIGINT, pool cần: (1) không nhận job mới, (2) chờ job đang xử lý hoàn thành, (3) thoát sạch. Đây là yêu cầu bắt buộc trong production.

GO graceful_shutdown.go — Production-grade Shutdown 
package pool

import (
    "context"
    "fmt"
    "os"
    "os/signal"
    "sync"
    "syscall"
    "time"
)

type GracefulPool struct {
    jobs        chan Job
    results     chan Result
    wg          sync.WaitGroup
    once        sync.Once // đảm bảo close(jobs) chỉ gọi 1 lần
    shutdownCh  chan struct{}
}

func NewGracefulPool(workers, queueSize int) *GracefulPool {
    p := &GracefulPool{
        jobs:       make(chan Job, queueSize),
        results:    make(chan Result, queueSize),
        shutdownCh: make(chan struct{}),
    }
    for i := 0; i < workers; i++ {
        p.wg.Add(1)
        go p.worker()
    }
    go func() { p.wg.Wait(); close(p.results) }()
    return p
}

func (p *GracefulPool) worker() {
    defer p.wg.Done()
    for job := range p.jobs { // range tự thoát khi channel đóng
        p.results <- processJob(context.Background(), job)
    }
}

// Submit gửi job vào pool, trả về false nếu pool đã shutdown.
func (p *GracefulPool) Submit(job Job) bool {
    select {
    case <-p.shutdownCh:
        return false // pool đã shutdown, reject job
    case p.jobs <- job:
        return true
    }
}

// Shutdown dừng pool, chờ tối đa timeout để jobs hoàn thành.
func (p *GracefulPool) Shutdown(timeout time.Duration) error {
    p.once.Do(func() {
        close(p.shutdownCh) // chặn Submit mới
        close(p.jobs)      // báo workers không còn job mới
    })

    done := make(chan struct{})
    go func() { p.wg.Wait(); close(done) }()

    select {
    case <-done:
        return nil
    case <-time.After(timeout):
        return fmt.Errorf("pool shutdown timed out after %v", timeout)
    }
}

// ── MAIN: Signal-aware graceful shutdown ─────────────────────────
func main() {
    pool := NewGracefulPool(5, 100)

    // Lắng nghe OS signal
    sigCh := make(chan os.Signal, 1)
    signal.Notify(sigCh, syscall.SIGINT, syscall.SIGTERM)

    // Producer goroutine
    go func() {
        for i := 0; ; i++ {
            if !pool.Submit(Job{ID: i}) {
                return
            }
            time.Sleep(10 * time.Millisecond)
        }
    }()

    // Chờ signal
    sig := <-sigCh
    fmt.Printf("received %v, shutting down...\n", sig)

    if err := pool.Shutdown(30 * time.Second); err != nil {
        fmt.Printf("shutdown error: %v\n", err)
    }
    fmt.Println("shutdown complete")
}

Generic Worker Pool (Go 1.18+)

Expert

Sử dụng type parameters để tạo Worker Pool hoàn toàn generic, type-safe, tái sử dụng cho mọi kiểu dữ liệu mà không cần interface{} hay reflection.

GO generic_pool.go — Type-safe Generic Worker Pool (Go 1.21+) 
package pool

import (
    "context"
    "sync"
)

// Pool là generic worker pool.
// I = Input type, O = Output type.
type Pool[I, O any] struct {
    numWorkers int
    fn         func(context.Context, I) (O, error)
}

type WorkResult[I, O any] struct {
    Input  I
    Output O
    Err    error
}

// New tạo một Pool với fn là hàm xử lý.
func New[I, O any](workers int, fn func(context.Context, I) (O, error)) *Pool[I, O] {
    return &Pool[I, O]{numWorkers: workers, fn: fn}
}

// Process nhận vào slice input, trả về channel kết quả (type-safe).
func (p *Pool[I, O]) Process(ctx context.Context, inputs []I) <-chan WorkResult[I, O] {
    jobs := make(chan I, len(inputs))
    for _, in := range inputs {
        jobs <- in
    }
    close(jobs)

    results := make(chan WorkResult[I, O], len(inputs))
    var wg sync.WaitGroup

    for i := 0; i < p.numWorkers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for input := range jobs {
                select {
                case <-ctx.Done():
                    return
                default:
                }
                output, err := p.fn(ctx, input)
                results <- WorkResult[I, O]{
                    Input: input, Output: output, Err: err,
                }
            }
        }()
    }

    go func() { wg.Wait(); close(results) }()
    return results
}

// Collect thu thập tất cả results thành slice (blocking).
func (p *Pool[I, O]) Collect(ctx context.Context, inputs []I) ([]WorkResult[I, O], error) {
    results := make([]WorkResult[I, O], 0, len(inputs))
    for r := range p.Process(ctx, inputs) {
        if r.Err != nil {
            return results, r.Err
        }
        results = append(results, r)
    }
    return results, nil
}

    defer p.wg.Done()
    for {
        select {
        case <-ctx.Done():
            return
        case job, ok := <-p.jobs:
            if !ok {
                return
            }
            // Track metrics
            p.metrics.queueDepth.Dec()
            p.metrics.activeWorkers.Inc()

                status = "error"
            }
            p.metrics.jobsTotal.WithLabelValues(status).Inc()

            select {
            case p.results <- result:
            case <-ctx.Done():
                return
            }
        }
    }
}

    p.jobs <- job
    p.metrics.queueDepth.Inc()
}

// ── CIRCUIT BREAKER ──────────────────────────────────────────────────

type CBState int32

const (
    CBClosed   CBState = 0 // bình thường
    CBOpen     CBState = 1 // từ chối mọi request
    CBHalfOpen CBState = 2 // thử recovery
)

type CircuitBreaker struct {
    state       atomic.Int32
    failures    atomic.Int32
    threshold   int32
    timeout     time.Duration
    lastFailure atomic.Int64 // unix nano
}

func NewCircuitBreaker(threshold int, timeout time.Duration) *CircuitBreaker {
    return &CircuitBreaker{threshold: int32(threshold), timeout: timeout}
}

func (cb *CircuitBreaker) Allow() bool {
    state := CBState(cb.state.Load())
    switch state {
    case CBClosed:
        return true
    case CBOpen:
        // Kiểm tra nếu timeout đã qua → chuyển half-open
        lastFail := time.Unix(0, cb.lastFailure.Load())
        if time.Since(lastFail) > cb.timeout {
            cb.state.CompareAndSwap(int32(CBOpen), int32(CBHalfOpen))
            return true
        }
        return false
    default:
        return true
    }
}

func (cb *CircuitBreaker) RecordSuccess() {
    cb.failures.Store(0)
    cb.state.Store(int32(CBClosed))
}

func (cb *CircuitBreaker) RecordFailure() {
    cb.lastFailure.Store(time.Now().UnixNano())
    if cb.failures.Add(1) >= cb.threshold {
        cb.state.Store(int32(CBOpen))
    }
}

// ── PRODUCTION POOL ──────────────────────────────────────────────────

type Config struct {
    Workers     int
    QueueSize   int
    MaxRetries  int
    RetryBase   time.Duration
    RateRPS     int
    RateBurst   int
    CBThreshold int
    CBTimeout   time.Duration
    ShutTimeout time.Duration
}

var DefaultConfig = Config{
    Workers: 10, QueueSize: 1000, MaxRetries: 3,
    RetryBase: 500 * time.Millisecond,
    RateRPS: 100, RateBurst: 20,
    CBThreshold: 5, CBTimeout: 30 * time.Second,
    ShutTimeout: 30 * time.Second,
}

type ProductionPool[I, O any] struct {
    cfg      Config
    jobs     chan *retryItem[I]
    results  chan WorkResult[I, O]
    dlq      chan *retryItem[I]
    handler  func(context.Context, I) (O, error)
    limiter  *rate.Limiter
    cb       *CircuitBreaker
    logger   *slog.Logger
    wg       sync.WaitGroup
    once     sync.Once
    shutdown chan struct{}
}

type retryItem[I any] struct {
    input    I
    attempts int
    nextAt   time.Time
    lastErr  error
}

func NewProductionPool[I, O any](
    cfg     Config,
    handler func(context.Context, I) (O, error),
    logger  *slog.Logger,
) *ProductionPool[I, O] {
    return &ProductionPool[I, O]{
        cfg:      cfg,
        jobs:     make(chan *retryItem[I], cfg.QueueSize),
        results:  make(chan WorkResult[I, O], cfg.QueueSize),
        dlq:      make(chan *retryItem[I], cfg.QueueSize),
        handler:  handler,
        limiter:  rate.NewLimiter(rate.Limit(cfg.RateRPS), cfg.RateBurst),
        cb:       NewCircuitBreaker(cfg.CBThreshold, cfg.CBTimeout),
        logger:   logger,
        shutdown: make(chan struct{}),
    }
}

func (p *ProductionPool[I, O]) Start(ctx context.Context) {
    for i := 0; i < p.cfg.Workers; i++ {
        p.wg.Add(1)
        go p.worker(ctx)
    }
    go func() { p.wg.Wait(); close(p.results); close(p.dlq) }()
}

func (p *ProductionPool[I, O]) worker(ctx context.Context) {
    defer p.wg.Done()
    for {
        select {
        case <-ctx.Done():
            return
        case item, ok := <-p.jobs:
            if !ok {
                return
            }
            p.processItem(ctx, item)
        }
    }
}

func (p *ProductionPool[I, O]) processItem(ctx context.Context, item *retryItem[I]) {
    // 1. Rate limit
    if err := p.limiter.Wait(ctx); err != nil {
        return
    }
    // 2. Circuit breaker
    if !p.cb.Allow() {
        p.logger.Warn("circuit breaker open, skipping")
        go func() { p.jobs <- item }() // requeue
        time.Sleep(1 * time.Second)
        return
    }
    // 3. Process
    output, err := p.handler(ctx, item.input)
    if err == nil {
        p.cb.RecordSuccess()
        p.results <- WorkResult[I, O]{Input: item.input, Output: output}
        return
    }
    // 4. Handle error
    p.cb.RecordFailure()
    item.attempts++
    item.lastErr = err

    if item.attempts >= p.cfg.MaxRetries {
        p.logger.Error("job moved to DLQ",
            "attempts", item.attempts,
            "err", err)
        select {
        case p.dlq <- item:
        default:
            p.logger.Error("DLQ full")
        }
        return
    }
    // 5. Schedule retry
    delay := backoff(item.attempts, p.cfg.RetryBase)
    p.logger.Info("retrying job", "attempt", item.attempts, "delay", delay)
    go func() {
        time.Sleep(delay)
        select {
        case p.jobs <- item:
        case <-ctx.Done():
        }
    }()
}

func (p *ProductionPool[I, O]) Submit(input I) error {
    select {
    case <-p.shutdown:
        return errors.New("pool is shut down")
    case p.jobs <- &retryItem[I]{input: input}:
        return nil
    default:
        return fmt.Errorf("queue full (len=%d)", len(p.jobs))
    }
}

func (p *ProductionPool[I, O]) Results() <-chan WorkResult[I, O] { return p.results }
func (p *ProductionPool[I, O]) DLQ()     <-chan *retryItem[I]             { return p.dlq }

func (p *ProductionPool[I, O]) Shutdown() error {
    p.once.Do(func() {
        close(p.shutdown)
        close(p.jobs)
    })
    done := make(chan struct{})
    go func() { p.wg.Wait(); close(done) }()
    select {
    case <-done:
        return nil
    case <-time.After(p.cfg.ShutTimeout):
        return fmt.Errorf("shutdown timeout after %v", p.cfg.ShutTimeout)
    }
}

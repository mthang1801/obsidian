01

Tổng quan Rate Limiting

★ Cơ bản

Rate Limiting là kỹ thuật kiểm soát tốc độ xử lý request nhằm bảo vệ hệ thống khỏi bị quá tải, đảm bảo chất lượng dịch vụ (QoS), và ngăn chặn lạm dụng API. Trong Go, rate limiting được triển khai hiệu quả thông qua hai thuật toán chính: Token Bucket và Leaky Bucket.

🛡️
Bảo vệ hệ thống
Ngăn chặn DDoS, traffic spike làm sập service downstream
⚡
Kiểm soát throughput
Đảm bảo SLA, giới hạn bandwidth cho từng client/tenant
🔄
Backpressure
Truyền tín hiệu ngược cho producer khi consumer quá tải
💰
Cost control
Giới hạn calls tới external API tính phí theo usage

Khi nào cần Rate Limiting?

ℹ️
Quy tắc thực tế: Bất kỳ endpoint nào tiêu thụ tài nguyên có giới hạn (CPU, memory, DB connection, external API quota) đều nên có rate limiting. Đây là tuyến phòng thủ đầu tiên trong kiến trúc microservices.

Khái niệm cốt lõi

★ Cơ bản

Token Bucket — Cơ chế hoạt động

Token Bucket giữ một "xô" chứa token. Token được nạp liên tục theo tốc độ cố định. Mỗi request tiêu thụ một token. Nếu xô trống → request bị từ chối hoặc phải chờ.

Hình 2.1: So sánh cơ chế hoạt động của Token Bucket vs. Leaky Bucket — Token Bucket (bên trái) cho phép lưu lượng truyền tải tăng đột biến ngắn hạn (Burst) dựa trên lượng token tích lũy, trong khi Leaky Bucket (bên phải) xếp các request vào hàng đợi và xử lý với tốc độ đầu ra hoàn toàn ổn định (Traffic Shaping).

Token Bucket — Cơ bản

★ Cơ bản

Triển khai Token Bucket đơn giản bằng tay. Hiểu rõ cơ chế bên trong: tích lũy token theo thời gian, giới hạn burst bởi capacity.

GO token_bucket_basic.go 
// token_bucket_basic.go — Token Bucket tự implement, không dùng thư viện ngoài
// Go 1.18+ | import: sync, time

package ratelimit

import (
    "sync"
    "time"
)

// TokenBucket là rate limiter theo thuật toán Token Bucket.
// Thread-safe nhờ mutex nội bộ.
type TokenBucket struct {
    mu       sync.Mutex
    capacity float64   // dung lượng tối đa của xô (burst size)
    tokens   float64   // số token hiện tại
    rate     float64   // tốc độ nạp token/giây
    lastTime time.Time  // thời điểm lần cuối được nạp
}

// NewTokenBucket tạo một TokenBucket mới.
// rate: token được nạp mỗi giây (vd: 10.0 = 10 req/s)
// capacity: số token tối đa có thể tích lũy (burst)
func NewTokenBucket(rate, capacity float64) *TokenBucket {
    return &TokenBucket{
        capacity: capacity,
        tokens:   capacity, // bắt đầu đầy
        rate:     rate,
        lastTime: time.Now(),
    }
}

// refill nạp token dựa trên thời gian đã trôi qua.
// Phải gọi trong vùng lock!
func (tb *TokenBucket) refill() {
    now := time.Now()
    elapsed := now.Sub(tb.lastTime).Seconds()
    tb.lastTime = now

    // Tính số token mới được nạp
    tb.tokens += elapsed * tb.rate

    // Không vượt quá capacity (giới hạn burst)
    if tb.tokens > tb.capacity {
        tb.tokens = tb.capacity
    }
}

// Allow kiểm tra xem có thể tiêu thụ n token không.
// Trả về true nếu được phép, false nếu bị rate limited.
func (tb *TokenBucket) Allow(n float64) bool {
    tb.mu.Lock()
    defer tb.mu.Unlock()

    tb.refill() // cập nhật token trước khi kiểm tra

    if tb.tokens >= n {
        tb.tokens -= n
        return true
    }
    return false
}

// Wait block cho đến khi có đủ n token.
// Trả về thời gian đã chờ.
func (tb *TokenBucket) Wait(n float64) time.Duration {
    for {
        tb.mu.Lock()
        tb.refill()
        if tb.tokens >= n {
            tb.tokens -= n
            tb.mu.Unlock()
            return 0
        }
        // Tính thời gian cần chờ cho đủ token
        needed := n - tb.tokens
        waitDuration := time.Duration(needed/tb.rate*float64(time.Second))
        tb.mu.Unlock()
        time.Sleep(waitDuration)
    }
}

// Stats trả về trạng thái hiện tại (dùng cho monitoring)
func (tb *TokenBucket) Stats() (tokens, capacity, rate float64) {
    tb.mu.Lock()
    defer tb.mu.Unlock()
    tb.refill()
    return tb.tokens, tb.capacity, tb.rate
}

// ── Ví dụ sử dụng ──────────────────────────────

func ExampleBasicUsage() {
    // Cho phép 5 request/giây, burst tối đa 10
    limiter := NewTokenBucket(5.0, 10.0)

    for i := 0; i < 15; i++ {
        if limiter.Allow(1.0) {
            // xử lý request
            fmt.Printf("request %d: ALLOWED\n", i)
        } else {
            fmt.Printf("request %d: RATE LIMITED\n", i)
        }
    }
}

Sử dụng golang.org/x/time/rate — Chuẩn Production

💡
Khuyến nghị: Trong production, hãy dùng golang.org/x/time/rate — thư viện chính thức của Go team, được test kỹ, thread-safe, hỗ trợ context cancellation.
GO token_bucket_stdlib.go 
// token_bucket_stdlib.go — Dùng golang.org/x/time/rate
// go get golang.org/x/time/rate

package main

import (
    "context"
    "fmt"
    "log"
    "time"
    "golang.org/x/time/rate"
)

func main() {
    // rate.NewLimiter(r, b):
    //   r = rate.Limit(10) → 10 token/s (= 10 req/s)
    //   b = 20             → burst size: tối đa 20 token tích lũy
    limiter := rate.NewLimiter(rate.Limit(10), 20)

    ctx := context.Background()

    // ── 1. Allow() — non-blocking, immediate decision
    if limiter.Allow() {
        fmt.Println("✅ Allow: request được xử lý")
    } else {
        fmt.Println("❌ Allow: bị từ chối")
    }

    // ── 2. Wait() — blocking, chờ đến khi có token
    // Respect context timeout/cancellation
    ctxTimeout, cancel := context.WithTimeout(ctx, 2*time.Second)
    defer cancel()

    if err := limiter.Wait(ctxTimeout); err != nil {
        log.Printf("Wait bị cancel/timeout: %v", err)
        return
    }
    fmt.Println("✅ Wait: request được xử lý sau khi chờ")

    // ── 3. Reserve() — advanced: lên lịch trước
    // Cho phép caller tự quyết định có chờ không
    r := limiter.Reserve()
    if !r.OK() {
        fmt.Println("❌ Reserve: không thể thực hiện trong tương lai")
        return
    }
    delay := r.Delay()
    fmt.Printf("⏳ Reserve: cần chờ %v trước khi xử lý\n", delay)
    time.Sleep(delay)
    fmt.Println("✅ Reserve: xử lý request")

    // ── 4. AllowN() — tiêu thụ N token một lúc (weighted request)
    // Ví dụ: bulk operation tiêu thụ nhiều token hơn
    if limiter.AllowN(time.Now(), 5) {
        fmt.Println("✅ AllowN(5): bulk request được xử lý")
    }

    // ── 5. SetLimit/SetBurst — dynamic adjustment at runtime
    limiter.SetLimit(rate.Limit(100))  // tăng lên 100 req/s
    limiter.SetBurst(200)             // tăng burst lên 200
    fmt.Printf("Dynamic limit: %.0f req/s, burst: %d\n",
        float64(limiter.Limit()), limiter.Burst())
}

Token Bucket — Per-Client Rate Limiter

◆ Advanced

Trong thực tế, mỗi client/user/API-key cần rate limiter riêng. Triển khai LimiterStore với cleanup goroutine để tránh memory leak.

GO per_client_limiter.go 
// per_client_limiter.go — Per-client Token Bucket với TTL cleanup
// Pattern: lưu limiter trong map, cleanup định kỳ để tránh memory leak

package ratelimit

import (
    "sync"
    "time"
    "golang.org/x/time/rate"
)

// clientEntry lưu limiter + thời điểm truy cập cuối
type clientEntry struct {
    limiter  *rate.Limiter
    lastSeen time.Time
}

// LimiterStore quản lý per-client rate limiters.
// Thread-safe, tự động cleanup entries không hoạt động.
type LimiterStore struct {
    mu      sync.RWMutex
    entries map[string]*clientEntry
    rate    rate.Limit
    burst   int
    ttl     time.Duration // thời gian inactive trước khi xóa
}

// NewLimiterStore tạo store với cleanup goroutine
func NewLimiterStore(r rate.Limit, burst int, ttl time.Duration) *LimiterStore {
    s := &LimiterStore{
        entries: make(map[string]*clientEntry),
        rate:    r,
        burst:   burst,
        ttl:     ttl,
    }
    // Goroutine cleanup định kỳ — tránh memory leak
    go s.cleanupLoop()
    return s
}

// GetOrCreate lấy hoặc tạo limiter cho clientID
func (s *LimiterStore) GetOrCreate(clientID string) *rate.Limiter {
    // Fast path: read lock
    s.mu.RLock()
    entry, ok := s.entries[clientID]
    s.mu.RUnlock()

    if ok {
        s.mu.Lock()
        entry.lastSeen = time.Now() // cập nhật thời gian truy cập
        s.mu.Unlock()
        return entry.limiter
    }

    // Slow path: write lock, tạo mới
    s.mu.Lock()
    defer s.mu.Unlock()

    // Double-check sau khi lấy write lock (race condition)
    if entry, ok = s.entries[clientID]; ok {
        return entry.limiter
    }

    newEntry := &clientEntry{
        limiter:  rate.NewLimiter(s.rate, s.burst),
        lastSeen: time.Now(),
    }
    s.entries[clientID] = newEntry
    return newEntry.limiter
}

// cleanupLoop xóa các entries không hoạt động sau TTL
func (s *LimiterStore) cleanupLoop() {
    ticker := time.NewTicker(s.ttl / 2)
    defer ticker.Stop()

    for range ticker.C {
        s.cleanup()
    }
}

func (s *LimiterStore) cleanup() {
    cutoff := time.Now().Add(-s.ttl)

    s.mu.Lock()
    defer s.mu.Unlock()

    for id, entry := range s.entries {
        if entry.lastSeen.Before(cutoff) {
            delete(s.entries, id)
        }
    }
}

// Len trả về số clients đang được track
func (s *LimiterStore) Len() int {
    s.mu.RLock()
    defer s.mu.RUnlock()
    return len(s.entries)
}

// ── HTTP Middleware sử dụng LimiterStore ──────────────────

import "net/http"

func PerClientRateLimitMiddleware(store *LimiterStore) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            // Lấy client identifier (IP hoặc API key)
            clientID := getClientID(r)
            limiter := store.GetOrCreate(clientID)

            if !limiter.Allow() {
                http.Error(w,
                    `{"error":"rate limit exceeded","code":429}`,
                    http.StatusTooManyRequests)
                w.Header().Set("Retry-After", "1")
                return
            }
            next.ServeHTTP(w, r)
        })
    }
}

func getClientID(r *http.Request) string {
    // Ưu tiên API key, fallback về IP
    if key := r.Header.Get("X-API-Key"); key != "" {
        return "key:" + key
    }
    return "ip:" + r.RemoteAddr
}

Generic Token Bucket (Go 1.18+)

◆ Advanced

Sử dụng Go Generics để tạo type-safe LimiterStore cho bất kỳ kiểu key nào (string, int64, UUID...).

GO generic_limiter.go 
// generic_limiter.go — Generic rate limiter store (Go 1.18+)
// Hỗ trợ bất kỳ kiểu key nào: string, int64, UUID, custom struct

package ratelimit

import (
    "sync"
    "time"
    "golang.org/x/time/rate"
)

// Comparable constraint: key phải comparable (có thể dùng làm map key)
type GenericStore[K comparable] struct {
    mu      sync.RWMutex
    entries map[K]*clientEntry
    factory func() *rate.Limiter // factory function tạo limiter
    ttl     time.Duration
}

// NewGenericStore tạo store với factory function linh hoạt
func NewGenericStore[K comparable](
    factory func() *rate.Limiter,
    ttl time.Duration,
) *GenericStore[K] {
    s := &GenericStore[K]{
        entries: make(map[K]*clientEntry),
        factory: factory,
        ttl:     ttl,
    }
    go s.cleanupLoop()
    return s
}

func (s *GenericStore[K]) Get(key K) *rate.Limiter {
    s.mu.RLock()
    e, ok := s.entries[key]
    s.mu.RUnlock()
    if ok {
        s.mu.Lock()
        e.lastSeen = time.Now()
        s.mu.Unlock()
        return e.limiter
    }

    s.mu.Lock()
    defer s.mu.Unlock()
    if e, ok = s.entries[key]; ok {
        return e.limiter
    }
    entry := &clientEntry{limiter: s.factory(), lastSeen: time.Now()}
    s.entries[key] = entry
    return entry.limiter
}

func (s *GenericStore[K]) cleanupLoop() {
    t := time.NewTicker(s.ttl)
    defer t.Stop()
    for range t.C {
        cutoff := time.Now().Add(-s.ttl)
        s.mu.Lock()
        for k, e := range s.entries {
            if e.lastSeen.Before(cutoff) {
                delete(s.entries, k)
            }
        }
        s.mu.Unlock()
    }
}

// ── Ví dụ sử dụng với nhiều kiểu key ──────────────────

type UserID int64
type TenantKey struct { TenantID, UserID string }

func ExampleGenericStore() {
    // Store dùng int64 UserID làm key
    userStore := NewGenericStore[UserID](
        func() *rate.Limiter {
            return rate.NewLimiter(10, 20)
        },
        5*time.Minute,
    )
    userStore.Get(UserID(42)).Allow()

    // Store dùng composite key cho multi-tenant
    tenantStore := NewGenericStore[TenantKey](
        func() *rate.Limiter {
            return rate.NewLimiter(100, 500)
        },
        10*time.Minute,
    )
    key := TenantKey{TenantID: "acme", UserID: "user-001"}
    tenantStore.Get(key).Allow()
}

Distributed Rate Limiting với Redis

★ Expert

Khi chạy nhiều instance service, rate limiter local không đủ — các instance không biết về nhau. Cần triển khai distributed rate limiting với Redis sử dụng Lua script để đảm bảo atomicity.

⚠️
Tại sao cần Lua script? Các thao tác GET → compare → SET không phải atomic trong Redis. Nếu dùng riêng lẻ sẽ có race condition giữa các instance. Lua script chạy atomically trong single-threaded Redis.
GO redis_token_bucket.go 
// redis_token_bucket.go — Distributed Token Bucket với Redis + Lua
// go get github.com/redis/go-redis/v9

package ratelimit

import (
    "context"
    "fmt"
    "time"
    "github.com/redis/go-redis/v9"
)

// Lua script: Token Bucket atomic operation
// KEYS[1] = key prefix
// ARGV[1] = capacity, ARGV[2] = rate (token/ms), ARGV[3] = now (ms), ARGV[4] = cost
var tokenBucketScript = redis.NewScript(`
local key_tokens    = KEYS[1] .. ":tokens"
local key_last_time = KEYS[1] .. ":ts"
local capacity  = tonumber(ARGV[1])
local rate      = tonumber(ARGV[2])
local now       = tonumber(ARGV[3])
local cost      = tonumber(ARGV[4])

local last_time = tonumber(redis.call("GET", key_last_time) or 0)
local tokens    = tonumber(redis.call("GET", key_tokens) or capacity)

-- Nạp token dựa trên thời gian trôi qua
local elapsed = math.max(0, now - last_time)
tokens = math.min(capacity, tokens + elapsed * rate)

if tokens >= cost then
    tokens = tokens - cost
    redis.call("SET", key_tokens,    tokens, "PX", 60000)
    redis.call("SET", key_last_time, now,    "PX", 60000)
    return {1, math.floor(tokens)}  -- allowed, remaining
else
    local wait_ms = math.ceil((cost - tokens) / rate)
    redis.call("SET", key_last_time, now, "PX", 60000)
    return {0, math.floor(wait_ms)} -- denied, wait_ms
end
`)

// RedisTokenBucket là distributed rate limiter dùng Redis
type RedisTokenBucket struct {
    client   *redis.Client
    capacity float64
    ratePerMs float64 // token/millisecond
    keyPrefix string
}

func NewRedisTokenBucket(
    client *redis.Client,
    keyPrefix string,
    ratePerSec float64,
    capacity float64,
) *RedisTokenBucket {
    return &RedisTokenBucket{
        client:    client,
        capacity:  capacity,
        ratePerMs: ratePerSec / 1000.0, // convert sang ms
        keyPrefix: keyPrefix,
    }
}

// AllowResult chứa kết quả từ rate limiter
type AllowResult struct {
    Allowed   bool
    Remaining int64
    RetryAfter time.Duration // chỉ set khi Allowed=false
}

// Allow kiểm tra và tiêu thụ cost token cho clientID
func (r *RedisTokenBucket) Allow(ctx context.Context, clientID string, cost float64) (AllowResult, error) {
    key := fmt.Sprintf("%s:%s", r.keyPrefix, clientID)
    nowMs := float64(time.Now().UnixMilli())

    result, err := tokenBucketScript.Run(ctx, r.client,
        []string{key},
        r.capacity, r.ratePerMs, nowMs, cost,
    ).Slice()

    if err != nil {
        // Fail open: nếu Redis lỗi, cho phép request đi qua
        return AllowResult{Allowed: true}, fmt.Errorf("redis rate limit: %w", err)
    }

    allowed := result[0].(int64) == 1
    val := result[1].(int64)

    if allowed {
        return AllowResult{Allowed: true, Remaining: val}, nil
    }
    return AllowResult{
        Allowed:    false,
        RetryAfter: time.Duration(val) * time.Millisecond,
    }, nil
}

Leaky Bucket — Cơ bản

★ Cơ bản

Leaky Bucket đảm bảo output rate cố định. Phù hợp cho traffic shaping — ví dụ: gửi email notification tối đa 10 emails/giây, bất kể đầu vào spike bao nhiêu.

GO leaky_bucket_basic.go 
// leaky_bucket_basic.go — Leaky Bucket với ticker-based drain
// Ý tưởng: queue nhận request; ticker drain đều đặn

package ratelimit

import (
    "context"
    "errors"
    "time"
)

// ErrBucketFull trả về khi queue đầy
var ErrBucketFull = errors.New("leaky bucket: queue is full")

// Request đại diện cho một request cần xử lý
type Request[T any] struct {
    Payload T
    Done    chan error // báo kết quả về cho caller
}

// LeakyBucket xử lý requests với output rate cố định.
// Input: có thể là burst
// Output: luôn đều đặn — đúng rate
type LeakyBucket[T any] struct {
    queue    chan Request[T]
    handler  func(ctx context.Context, payload T) error
    interval time.Duration
}

// NewLeakyBucket tạo bucket mới.
// ratePerSec: số request/giây được xử lý
// capacity:   kích thước queue (burst buffer)
func NewLeakyBucket[T any](
    ratePerSec int,
    capacity int,
    handler func(context.Context, T) error,
) *LeakyBucket[T] {
    interval := time.Second / time.Duration(ratePerSec)
    lb := &LeakyBucket[T]{
        queue:    make(chan Request[T], capacity),
        handler:  handler,
        interval: interval,
    }
    return lb
}

// Start bắt đầu drain loop — chạy trong goroutine riêng
func (lb *LeakyBucket[T]) Start(ctx context.Context) {
    ticker := time.NewTicker(lb.interval)
    go func() {
        defer ticker.Stop()
        for {
            select {
            case <-ctx.Done():
                // drain phần còn lại trước khi thoát
                for {
                    select {
                    case req := <-lb.queue:
                        req.Done <- lb.handler(ctx, req.Payload)
                    default:
                        return
                    }
                }
            case <-ticker.C:
                // Mỗi tick xử lý đúng 1 request
                select {
                case req := <-lb.queue:
                    req.Done <- lb.handler(ctx, req.Payload)
                default:
                    // queue trống, bỏ qua tick này
                }
            }
        }
    }()
}

// Submit gửi payload vào queue.
// Trả về ErrBucketFull ngay lập tức nếu queue đầy (non-blocking).
func (lb *LeakyBucket[T]) Submit(payload T) (<-chan error, error) {
    done := make(chan error, 1)
    req := Request[T]{Payload: payload, Done: done}

    select {
    case lb.queue <- req:
        return done, nil
    default:
        return nil, ErrBucketFull
    }
}

// ── Ví dụ: Email notification với rate 10/s ──────────

func ExampleEmailRateLimiting() {
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    // Tối đa 10 email/giây, queue tối đa 100
    bucket := NewLeakyBucket(10, 100,
        func(ctx context.Context, email string) error {
            fmt.Printf("📧 Sending to: %s\n", email)
            return sendEmail(ctx, email)
        })
    bucket.Start(ctx)

    // Gửi 50 emails cùng lúc — chỉ 10/s được xử lý
    emails := []string{"a@x.com", "b@x.com", /* ... 48 more */}
    for _, email := range emails {
        done, err := bucket.Submit(email)
        if err != nil {
            fmt.Printf("⚠️  Queue full, dropping: %s\n", email)
            continue
        }
        go func() {
            if err := <-done; err != nil {
                fmt.Printf("❌ Send failed: %v\n", err)
            }
        }()
    }
}

// Mỗi goroutine phải "lấy" từ channel trước khi làm việc.
// Channel được fill lại theo rate cố định.

func NewChannelRateLimiter(ratePerSec int) <-chan struct{} {
    throttle := make(chan struct{}, ratePerSec)

    // Filler goroutine: nạp token định kỳ
    go func() {
        interval := time.Second / time.Duration(ratePerSec)
        ticker := time.NewTicker(interval)
        defer ticker.Stop()
        for range ticker.C {
            select {
            case throttle <- struct{}{}:
            default:
                // channel đầy — bỏ qua (không block)
            }
        }
    }()
    return throttle
}

func SimpleRateLimit(ctx context.Context, ratePerSec int, work func()) {
    ticker := time.NewTicker(time.Second / time.Duration(ratePerSec))
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            go work() // mỗi tick xử lý 1 unit of work
        }
    }

// ── Approach 4: Semaphore + Queue (concurrent with bound) ────
// Kết hợp rate limiting + concurrency limiting

type BoundedWorkerPool struct {
    sem     chan struct{}  // semaphore: giới hạn concurrent workers
    queue   chan func()    // task queue
    limiter <-chan struct{} // rate limiter channel
}

func NewBoundedWorkerPool(maxConcurrent, ratePerSec, queueSize int) *BoundedWorkerPool {
    p := &BoundedWorkerPool{
        sem:     make(chan struct{}, maxConcurrent),
        queue:   make(chan func(), queueSize),
        limiter: NewChannelRateLimiter(ratePerSec),
    }
    go p.dispatch()
    return p
}

func (p *BoundedWorkerPool) dispatch() {
    for task := range p.queue {
        <-p.limiter     // chờ rate limiter cho phép
        p.sem <- struct{}{}  // acquire semaphore
        go func(t func()) {
            defer func() { <-p.sem }() // release semaphore
            t()
        }(task)
    }
}

func (p *BoundedWorkerPool) Submit(task func()) bool {
    select {
    case p.queue <- task:
        return true
    default:
        return false // queue đầy
    }
}

◆ Advanced

// leaky_backpressure.go — Leaky Bucket với backpressure signaling
// Notify producer khi queue sắp đầy để chậm lại

package ratelimit

import (
    "context"
    "sync/atomic"
    "time"
)

// QueueStats thống kê real-time
type QueueStats struct {
    Submitted  atomic.Int64
    Dropped    atomic.Int64
    Processed  atomic.Int64
    QueueDepth atomic.Int64
}

type BackpressureLeakyBucket[T any] struct {
    queue       chan T
    processedCh chan T     // processed output (cho consumer)
    highWaterMark int     // % đầy khi bắt đầu backpressure
    capacity    int
    interval    time.Duration
    Stats       QueueStats

    // Backpressure: notify channel, producer đọc để chậm lại
    pressure chan float64 // 0.0-1.0: mức áp lực hiện tại
}

func NewBPLeakyBucket[T any](ratePerSec, capacity int) *BackpressureLeakyBucket[T] {
    lb := &BackpressureLeakyBucket[T]{
        queue:         make(chan T, capacity),
        processedCh:   make(chan T, capacity/2),
        highWaterMark: capacity * 80 / 100, // 80% threshold
        capacity:      capacity,
        interval:      time.Second / time.Duration(ratePerSec),
        pressure:      make(chan float64, 1),
    }
    return lb
}

// Start chạy drain loop
func (lb *BackpressureLeakyBucket[T]) Start(ctx context.Context) {
    ticker := time.NewTicker(lb.interval)
    go func() {
        defer ticker.Stop()
        for {
            select {
            case <-ctx.Done():
                return
            case <-ticker.C:
                select {
                case item := <-lb.queue:
                    lb.Stats.Processed.Add(1)
                    lb.Stats.QueueDepth.Add(-1)
                    lb.processedCh <- item

                    // Tính pressure level và signal
                    depth := lb.Stats.QueueDepth.Load()
                    pressure := float64(depth) / float64(lb.capacity)
                    select {
                    case lb.pressure <- pressure:
                    default:
                        // drain cũ, đặt mới
                        <-lb.pressure
                        lb.pressure <- pressure
                    }
                default:
                }
            }
        }
    }()
}

// Submit gửi item, trả về backpressure signal
func (lb *BackpressureLeakyBucket[T]) Submit(item T) (dropped bool, pressureLevel float64) {
    lb.Stats.Submitted.Add(1)

    select {
    case lb.queue <- item:
        depth := lb.Stats.QueueDepth.Add(1)
        level := float64(depth) / float64(lb.capacity)
        return false, level
    default:
        lb.Stats.Dropped.Add(1)
        return true, 1.0
    }
}

// Output trả về channel của processed items
func (lb *BackpressureLeakyBucket[T]) Output() <-chan T {
    return lb.processedCh
}

// Pressure trả về channel backpressure signal (0.0-1.0)
func (lb *BackpressureLeakyBucket[T]) Pressure() <-chan float64 {
    return lb.pressure
}

// ── Producer sử dụng backpressure để tự điều chỉnh tốc độ ──

func AdaptiveProducer(ctx context.Context, lb *BackpressureLeakyBucket[string]) {
    baseInterval := 1*time.Millisecond
    interval := baseInterval

    for {
        select {
        case <-ctx.Done():
            return
        case pressure := <-lb.Pressure():
            // Điều chỉnh tốc độ dựa trên áp lực
            switch {
            case pressure > 0.9:
                interval = baseInterval * 10 // rất chậm
            case pressure > 0.7:
                interval = baseInterval * 5  // chậm lại
            case pressure > 0.5:
                interval = baseInterval * 2  // hơi chậm
            default:
                interval = baseInterval          // bình thường
            }
        case <-time.After(interval):
            dropped, level := lb.Submit("item")
            if dropped {
                fmt.Printf("⚠️  Dropped! Pressure: %.0f%%\n", level*100)
            }
        }
    }
}

// errorResponse chuẩn RFC 7807 Problem Details
type errorResponse struct {
    Type     string `json:"type"`
    Title    string `json:"title"`
    Status   int    `json:"status"`
    Detail   string `json:"detail,omitempty"`
    RetryAfter int `json:"retryAfter,omitempty"`
}

func (m *RateLimitMiddleware) Handler(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // Tier 1: Global rate limit
        if !m.global.Allow() {
            m.rejectWith(w, "server-overloaded",
                "Server is overloaded, please try again later", 5)
            return
        }

        // Tier 2: Per-client rate limit
        clientIP := realIP(r)
        clientLimiter := m.perClient.GetOrCreate(clientIP)
        if !clientLimiter.Allow() {
            m.rejectWith(w, "rate-limit-exceeded",
                "Too many requests from your IP", 1)
            return
        }

        // Set RateLimit headers (IETF draft standard)
        r2 := m.perClient.GetOrCreate(clientIP)
        _ = r2
        w.Header().Set("X-RateLimit-Limit",
            strconv.Itoa(m.cfg.PerIPBurst))
        w.Header().Set("X-RateLimit-Policy",
            strconv.FormatFloat(float64(m.cfg.PerIPRate), 'f', 0, 64)+";w=1")

        next.ServeHTTP(w, r)
    })
}

func (m *RateLimitMiddleware) rejectWith(w http.ResponseWriter, typ, detail string, retryAfter int) {
    w.Header().Set("Content-Type", "application/problem+json")
    w.Header().Set("Retry-After", strconv.Itoa(retryAfter))
    w.WriteHeader(http.StatusTooManyRequests)
    json.NewEncoder(w).Encode(errorResponse{
        Type:       "https://example.com/errors/" + typ,
        Title:      "Too Many Requests",
        Status:     http.StatusTooManyRequests,
        Detail:     detail,
        RetryAfter: retryAfter,
    })
}

// realIP lấy real IP từ proxy headers
func realIP(r *http.Request) string {
    if ip := r.Header.Get("X-Real-IP"); ip != "" {
        return ip
    }
    if ip := r.Header.Get("X-Forwarded-For"); ip != "" {
        return ip
    }
    return r.RemoteAddr
}

Sliding Window Rate Limiter

★ Expert

Sliding Window giải quyết vấn đề "boundary burst" của Fixed Window. Chính xác hơn Token Bucket cho việc đếm requests trong khoảng thời gian.

GO sliding_window.go 
// sliding_window.go — Sliding Window với Redis sorted set
// Chính xác, distributed, nhưng tốn O(n) memory

package ratelimit

import (
    "context"
    "fmt"
    "time"
    "github.com/redis/go-redis/v9"
)

// Lua script: sliding window với sorted set
// Atomic: ZREMRANGEBYSCORE + ZADD + ZCARD trong 1 lần
var slidingWindowScript = redis.NewScript(`
local key      = KEYS[1]
local now      = tonumber(ARGV[1])
local window   = tonumber(ARGV[2])
local limit    = tonumber(ARGV[3])
local req_id   = ARGV[4]

-- Xóa requests ngoài window
redis.call("ZREMRANGEBYSCORE", key, 0, now - window)

-- Đếm requests trong window
local count = redis.call("ZCARD", key)

if count < limit then
    -- Thêm request hiện tại
    redis.call("ZADD", key, now, req_id)
    redis.call("PEXPIRE", key, window + 1000)
    return {1, limit - count - 1}  -- allowed, remaining
else
    -- Tìm thời điểm request cũ nhất sẽ expire
    local oldest = redis.call("ZRANGE", key, 0, 0, "WITHSCORES")
    local reset_at = tonumber(oldest[2]) + window
    return {0, reset_at - now}  -- denied, wait_ms
end
`)

type SlidingWindowLimiter struct {
    client    *redis.Client
    keyPrefix string
    limit     int64
    window    time.Duration
}

func NewSlidingWindowLimiter(
    client *redis.Client,
    keyPrefix string,
    limit int64,
    window time.Duration,
) *SlidingWindowLimiter {
    return &SlidingWindowLimiter{
        client:    client,
        keyPrefix: keyPrefix,
        limit:     limit,
        window:    window,
    }
}

func (s *SlidingWindowLimiter) Allow(ctx context.Context, clientID string) (AllowResult, error) {
    key := fmt.Sprintf("%s:%s", s.keyPrefix, clientID)
    nowMs := time.Now().UnixMilli()
    windowMs := s.window.Milliseconds()
    reqID := fmt.Sprintf("%d-%s", nowMs, clientID)

    result, err := slidingWindowScript.Run(ctx, s.client,
        []string{key},
        nowMs, windowMs, s.limit, reqID,
    ).Slice()

    if err != nil {
        return AllowResult{Allowed: true}, fmt.Errorf("sliding window: %w", err)
    }

    allowed := result[0].(int64) == 1
    val := result[1].(int64)

    if allowed {
        return AllowResult{Allowed: true, Remaining: val}, nil
    }
    return AllowResult{
        Allowed:    false,
        RetryAfter: time.Duration(val) * time.Millisecond,
    }, nil
}

Adaptive Rate Limiter

★ Expert

Adaptive Rate Limiter tự động điều chỉnh giới hạn dựa trên metrics hệ thống (error rate, latency, CPU). Khi hệ thống bị tải nặng → giảm limit; khi khoẻ → tăng dần.

GO adaptive_limiter.go 
// adaptive_limiter.go — Adaptive Rate Limiter dựa trên system health
// Tự động tăng/giảm rate dựa trên error rate và latency

package ratelimit

import (
    "context"
    "sync/atomic"
    "time"
    "golang.org/x/time/rate"
)

// SystemHealth đo lường sức khoẻ hệ thống
type SystemHealth struct {
    totalRequests atomic.Int64
    errorCount    atomic.Int64
    totalLatencyMs atomic.Int64
    window        time.Duration
}

func (h *SystemHealth) RecordSuccess(latencyMs int64) {
    h.totalRequests.Add(1)
    h.totalLatencyMs.Add(latencyMs)
}

func (h *SystemHealth) RecordError() {
    h.totalRequests.Add(1)
    h.errorCount.Add(1)
}

func (h *SystemHealth) ErrorRate() float64 {
    total := h.totalRequests.Load()
    if total == 0 {
        return 0
    }
    return float64(h.errorCount.Load()) / float64(total)
}

func (h *SystemHealth) AvgLatencyMs() float64 {
    total := h.totalRequests.Load()
    if total == 0 {
        return 0
    }
    return float64(h.totalLatencyMs.Load()) / float64(total)
}

// AdaptiveLimiter điều chỉnh rate theo health của hệ thống
type AdaptiveLimiter struct {
    limiter    *rate.Limiter
    health     *SystemHealth

    minRate    rate.Limit
    maxRate    rate.Limit
    currentRate rate.Limit

    // Thresholds để kích hoạt điều chỉnh
    errorRateThreshold  float64 // vd: 0.05 (5%)
    latencyThresholdMs  float64 // vd: 200ms
    adjustmentFactor    float64 // vd: 0.8 giảm 20% mỗi lần
    recoveryFactor      float64 // vd: 1.05 tăng 5% mỗi lần
}

func NewAdaptiveLimiter(minRate, maxRate rate.Limit) *AdaptiveLimiter {
    initialRate := (minRate + maxRate) / 2
    al := &AdaptiveLimiter{
        limiter:            rate.NewLimiter(initialRate, int(initialRate*2)),
        health:             &SystemHealth{},
        minRate:            minRate,
        maxRate:            maxRate,
        currentRate:        initialRate,
        errorRateThreshold: 0.05,
        latencyThresholdMs: 200,
        adjustmentFactor:   0.8,
        recoveryFactor:     1.05,
    }
    return al
}

// StartAdaptation chạy vòng lặp điều chỉnh định kỳ
func (al *AdaptiveLimiter) StartAdaptation(ctx context.Context, interval time.Duration) {
    ticker := time.NewTicker(interval)
    go func() {
        defer ticker.Stop()
        for {
            select {
            case <-ctx.Done():
                return
            case <-ticker.C:
                al.adapt()
            }
        }
    }()
}

func (al *AdaptiveLimiter) adapt() {
    errRate := al.health.ErrorRate()
    avgLatency := al.health.AvgLatencyMs()

    shouldDecrease := errRate > al.errorRateThreshold ||
        avgLatency > al.latencyThresholdMs

    var newRate rate.Limit
    if shouldDecrease {
        // Hệ thống đang stress → giảm rate
        newRate = rate.Limit(float64(al.currentRate) * al.adjustmentFactor)
        if newRate < al.minRate {
            newRate = al.minRate
        }
    } else {
        // Hệ thống khoẻ → tăng rate dần
        newRate = rate.Limit(float64(al.currentRate) * al.recoveryFactor)
        if newRate > al.maxRate {
            newRate = al.maxRate
        }
    }

    if newRate != al.currentRate {
        al.currentRate = newRate
        al.limiter.SetLimit(newRate)
        al.limiter.SetBurst(int(newRate * 2))
        fmt.Printf("🎚 Adaptive: rate=%.0f err=%.1f%% lat=%.0fms\n",
            float64(newRate), errRate*100, avgLatency)
    }
}

func (al *AdaptiveLimiter) Allow() bool {
    return al.limiter.Allow()
}

func (al *AdaptiveLimiter) Health() *SystemHealth {
    return al.health
}

// composite_limiter.go — Composite (chained) Rate Limiters
// Áp dụng nhiều limiters theo thứ tự; từ chối nếu BẤT KỲ limiter nào deny

package ratelimit

import "context"

// Limiter interface — đơn giản, consumer-defined (Go idiom)
type Limiter interface {
    Allow(ctx context.Context, key string) (bool, error)
    Name() string
}

// LimitResult chứa thông tin từ composite limiter
type LimitResult struct {
    Allowed    bool
    DeniedBy   string   // tên limiter đã deny
    Reasons    []string // tất cả lý do từ chối
}

// CompositeLimiter áp dụng nhiều limiters theo thứ tự
type CompositeLimiter struct {
    limiters []Limiter
    mode     string // "fail_fast" hoặc "check_all"
}

func NewCompositeLimiter(mode string, limiters ...Limiter) *CompositeLimiter {
    return &CompositeLimiter{limiters: limiters, mode: mode}
}

func (c *CompositeLimiter) Allow(ctx context.Context, key string) (LimitResult, error) {
    var reasons []string
    deniedBy := ""

    for _, l := range c.limiters {
        allowed, err := l.Allow(ctx, key)
        if err != nil {
            // Fail open: lỗi từ limiter không block request
            continue
        }
        if !allowed {
            reasons = append(reasons, l.Name())
            if deniedBy == "" {
                deniedBy = l.Name()
            }
            if c.mode == "fail_fast" {
                return LimitResult{Allowed: false, DeniedBy: deniedBy, Reasons: reasons}, nil
            }
        }
    }

    return LimitResult{
        Allowed:  len(reasons) == 0,
        DeniedBy: deniedBy,
        Reasons:  reasons,
    }, nil
}

// UnaryRateLimitInterceptor cho gRPC unary calls
func UnaryRateLimitInterceptor(store *LimiterStore) grpc.UnaryServerInterceptor {
    return func(

        handler grpc.UnaryHandler,

        clientID := extractClientID(ctx)

        return handler(ctx, req)
    }

    limiter *rate.Limiter
}

func (s *rateLimitedStream) RecvMsg(m interface{}) error {
    if !s.limiter.Allow() {
        return errRateLimited
    }
    return s.ServerStream.RecvMsg(m)
}

// extractClientID lấy client identifier từ gRPC metadata
func extractClientID(ctx context.Context) string {
    md, ok := metadata.FromIncomingContext(ctx)
    if !ok {
        return "unknown"
    }
    if keys := md.Get("x-api-key"); len(keys) > 0 {
        return "key:" + keys[0]
    }
    if peers := md.Get("x-peer-address"); len(peers) > 0 {
        return "ip:" + peers[0]
    }
    return "unknown"
}

    return grpc.NewServer(
        grpc.ChainUnaryInterceptor(
            UnaryRateLimitInterceptor(store),
            // ... other interceptors (logging, auth, tracing)
        ),
        grpc.ChainStreamInterceptor(
            StreamRateLimitInterceptor(store),
        ),
    )
}

                }

            }

// ── Test concurrency safety ────────────────────────────

func TestTokenBucket_Concurrent(t *testing.T) {
    tb := NewTokenBucket(100, 100)

    var (
        wg      sync.WaitGroup
        allowed atomic.Int64
        denied  atomic.Int64
    )

    const goroutines = 50
    const reqs = 10

    wg.Add(goroutines)
    for i := 0; i < goroutines; i++ {
        go func() {
            defer wg.Done()
            for j := 0; j < reqs; j++ {
                if tb.Allow(1) {
                    allowed.Add(1)
                } else {
                    denied.Add(1)
                }
            }
        }()
    }
    wg.Wait()

    total := allowed.Load() + denied.Load()
    if total != goroutines*reqs {
        t.Errorf("lost requests: got %d, want %d", total, goroutines*reqs)
    }
    // Không được phép nhiều hơn capacity
    if allowed.Load() > 100 {
        t.Errorf("allowed %d > capacity 100", allowed.Load())
    }
}

    }, []string{"limiter"})

    }, []string{"limiter", "client"})

        Help:    "Time spent waiting for rate limit",
        Buckets: prometheus.ExponentialBuckets(0.001, 2, 10),
    }, []string{"limiter"})
)

// InstrumentedLimiter bọc Limiter với Prometheus metrics
type InstrumentedLimiter struct {
    inner      Limiter
    name       string
    tracer     trace.Tracer
}

func Instrument(inner Limiter, name string, tracer trace.Tracer) *InstrumentedLimiter {
    return &InstrumentedLimiter{inner: inner, name: name, tracer: tracer}
}

func (il *InstrumentedLimiter) Allow(ctx context.Context, key string) (bool, error) {
    start := time.Now()

    // OTel span
    ctx, span := il.tracer.Start(ctx, "rate_limit.check")
    defer span.End()
    span.SetAttributes(
        attribute.String("limiter.name", il.name),
        attribute.String("client.key", key),
    )

    allowed, err := il.inner.Allow(ctx, key)

    // Record metrics
    result := "allowed"
    if !allowed { result = "denied" }
    rateLimitAllowed.WithLabelValues(il.name, result, "ip").Inc()

    elapsed := time.Since(start)
    rateLimitWaitDuration.WithLabelValues(il.name).Observe(elapsed.Seconds())

    span.SetAttributes(attribute.Bool("rate_limit.allowed", allowed))
    return allowed, err
}

func (il *InstrumentedLimiter) Name() string { return il.name }

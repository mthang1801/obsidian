01
Data Conversion
Type casting, JSON/encoding, generic converters, safe type coercion
ADVANCED
📖 Khái niệm
Chuyển đổi dữ liệu an toàn trong Go

Go là ngôn ngữ statically typed — không có implicit conversion. Mọi chuyển đổi kiểu phải tường minh. Data Conversion bao gồm: type assertion, type casting, encoding (JSON/XML/CSV), format transformation. Sử dụng generics (Go 1.18+) để viết converter tái sử dụng.

VÍ DỤ 01 Generic Type Converter — chuyển đổi kiểu an toàn với generics ADVANCED
⚠️ Problem Statement

Trong một dịch vụ xử lý dữ liệu đầu vào từ nhiều nguồn (API, form, CSV), bạn nhận được các giá trị dạng any và cần chuyển đổi sang kiểu cụ thể một cách an toàn, không panic, có fallback giá trị mặc định.

✅ Solution

Sử dụng generic converter với type constraint, kết hợp type assertion và reflect để xử lý mọi trường hợp.

goconverter/generic.go
package converter

import (
    "encoding/json"
    "fmt"
    "reflect"
    "strconv"
    "strings"
)

// Ordered — các kiểu có thể so sánh và chuyển đổi số học
type Number interface {
    int | int8 | int16 | int32 | int64 |
    uint | uint8 | uint16 | uint32 | uint64 |
    float32 | float64
}

// To — generic converter an toàn, không panic
func To[T any](v any, defaults ...T) (T, error) {
    var zero T
    if v == nil {
        if len(defaults) > 0 { return defaults[0], nil }
        return zero, fmt.Errorf("cannot convert nil to %T", zero)
    }

    // Direct type assertion — fastest path
    if result, ok := v.(T); ok {
        return result, nil
    }

    // Fallback: JSON round-trip (handles struct/map conversions)
    data, err := json.Marshal(v)
    if err != nil {
        return zero, fmt.Errorf("marshal failed: %w", err)
    }
    var result T
    if err := json.Unmarshal(data, &result); err != nil {
        if len(defaults) > 0 { return defaults[0], nil }
        return zero, fmt.Errorf("unmarshal %T to %T failed: %w", v, zero, err)
    }
    return result, nil
}

// MustTo — panic nếu conversion thất bại (dùng trong init/test)
func MustTo[T any](v any) T {
    result, err := To[T](v)
    if err != nil { panic(err) }
    return result
}

// ToNumber — chuyển đổi string/any sang numeric type
func ToNumber[T Number](v any, defaults ...T) (T, error) {
    var zero T
    switch val := v.(type) {
    case string:
        f, err := strconv.ParseFloat(strings.TrimSpace(val), 64)
        if err != nil {
            if len(defaults) > 0 { return defaults[0], nil }
            return zero, fmt.Errorf("invalid number: %q", val)
        }
        return T(f), nil
    case float64: return T(val), nil
    case int:     return T(val), nil
    case int64:   return T(val), nil
    default:
        rv := reflect.ValueOf(v)
        switch rv.Kind() {
        case reflect.Int, reflect.Int8, reflect.Int16, reflect.Int32, reflect.Int64:
            return T(rv.Int()), nil
        case reflect.Float32, reflect.Float64:
            return T(rv.Float()), nil
        }
    }
    if len(defaults) > 0 { return defaults[0], nil }
    return zero, fmt.Errorf("cannot convert %T to number", v)
}

// StructToMap — chuyển struct sang map[string]any via JSON tags
func StructToMap(s any) (map[string]any, error) {
    data, err := json.Marshal(s)
    if err != nil { return nil, err }
    var m map[string]any
    return m, json.Unmarshal(data, &m)
}

// MapToStruct — ngược lại, từ map sang struct
func MapToStruct[T any](m map[string]any) (*T, error) {
    data, err := json.Marshal(m)
    if err != nil { return nil, err }
    var result T
    return &result, json.Unmarshal(data, &result)
}

// === USAGE ===
type UserDTO struct {
    ID    int64  `json:"id"`
    Name  string `json:"name"`
    Score float64 `json:"score"`
}

func Example() {
    // Từ map[string]any (thường gặp khi parse JSON thuần)
    raw := map[string]any{
        "id": float64(42), "name": "Alice", "score": 98.5,
    }
    user, _ := MapToStruct[UserDTO](raw)
    fmt.Printf("User: %+v\n", user) // {ID:42 Name:Alice Score:98.5}

    // Convert string sang int64 với default
    id, _ := ToNumber[int64]("invalid", 0)
    fmt.Println(id) // 0 (default)
}
VÍ DỤ 02 Multi-format Encoder/Decoder — JSON, CSV, Protobuf-like EXPERT
⚠️ Problem Statement

Service cần export dữ liệu sang nhiều định dạng (JSON, CSV, TSV) tùy theo yêu cầu của client mà không duplicate business logic.

goconverter/encoder.go
package converter

import (
    "bytes"
    "encoding/csv"
    "encoding/json"
    "fmt"
    "io"
    "reflect"
    "strings"
)

type Format string
const (
    FormatJSON Format = "json"
    FormatCSV  Format = "csv"
    FormatTSV  Format = "tsv"
)

type Encoder interface {
    Encode(w io.Writer, data any) error
    ContentType() string
}

// JSONEncoder
type JSONEncoder struct{ Indent bool }
func (e JSONEncoder) ContentType() string { return "application/json" }
func (e JSONEncoder) Encode(w io.Writer, data any) error {
    enc := json.NewEncoder(w)
    if e.Indent { enc.SetIndent("", "  ") }
    return enc.Encode(data)
}

// CSVEncoder — encode slice of structs via reflection + csv tags
type CSVEncoder struct{ Delimiter rune }
func (e CSVEncoder) ContentType() string { return "text/csv" }
func (e CSVEncoder) Encode(w io.Writer, data any) error {
    rv := reflect.ValueOf(data)
    if rv.Kind() != reflect.Slice { return fmt.Errorf("CSV requires slice, got %T", data) }
    cw := csv.NewWriter(w)
    if e.Delimiter != 0 { cw.Comma = e.Delimiter }
    defer cw.Flush()

    if rv.Len() == 0 { return nil }

    elem := rv.Index(0).Type()
    // Write header from field names / csv tags
    headers := make([]string, elem.NumField())
    for i := range headers {
        field := elem.Field(i)
        tag := field.Tag.Get("csv")
        if tag == "" { tag = strings.ToLower(field.Name) }
        headers[i] = tag
    }
    cw.Write(headers)

    // Write rows
    for i := 0; i < rv.Len(); i++ {
        row := make([]string, elem.NumField())
        for j := 0; j < elem.NumField(); j++ {
            row[j] = fmt.Sprintf("%v", rv.Index(i).Field(j).Interface())
        }
        cw.Write(row)
    }
    return cw.Error()
}

// EncoderFactory — factory pattern cho multi-format
func NewEncoder(fmt Format) (Encoder, error) {
    switch fmt {
    case FormatJSON: return JSONEncoder{Indent: true}, nil
    case FormatCSV:  return CSVEncoder{}, nil
    case FormatTSV:  return CSVEncoder{Delimiter: '\t'}, nil
    }
    return nil, fmt.Errorf("unsupported format: %s", fmt)
}

// Export — one API, multiple formats
func Export(data any, format Format) ([]byte, string, error) {
    enc, err := NewEncoder(format)
    if err != nil { return nil, "", err }
    var buf bytes.Buffer
    if err := enc.Encode(&buf, data); err != nil { return nil, "", err }
    return buf.Bytes(), enc.ContentType(), nil
}

  • ✓To[T]() generic an toàn hơn type assertion vì không panic, luôn trả về error.
  • ⚠JSON round-trip có overhead — dùng khi không biết kiểu chính xác, không dùng trong hot path.
  • ✓Factory pattern cho Encoder giúp thêm format mới không cần sửa business code.

Pipeline Pattern — xử lý data theo chuỗi stages

Pipeline pattern tổ chức xử lý dữ liệu theo chuỗi stages kết nối với nhau. Mỗi stage nhận input, transform, và truyền output sang stage tiếp theo. Go hỗ trợ hai dạng: Functional pipeline (slice-based, eager) và Concurrent pipeline (channel-based, lazy evaluation).

VÍ DỤ 01 Generic Functional Pipeline — Map, Filter, Reduce với generics ADVANCED
⚠️ Problem Statement

Xử lý danh sách đơn hàng: lọc đơn chưa thanh toán, tính tổng giá trị, nhóm theo khách hàng — code dễ đọc, tái sử dụng, không lặp lại vòng for.

gopipeline/functional.go
package pipeline

import "golang.org/x/exp/constraints"

// Map — transform each element
func Map[T, U any](slice []T, fn func(T) U) []U {
    out := make([]U, len(slice))
    for i, v := range slice { out[i] = fn(v) }
    return out
}

// MapWithIndex — like Map but with index
func MapWithIndex[T, U any](slice []T, fn func(int, T) U) []U {
    out := make([]U, len(slice))
    for i, v := range slice { out[i] = fn(i, v) }
    return out
}

// Filter — keep elements matching predicate
func Filter[T any](slice []T, pred func(T) bool) []T {
    out := make([]T, 0, len(slice)/2)
    for _, v := range slice {
        if pred(v) { out = append(out, v) }
    }
    return out
}

// Reduce — aggregate to single value
func Reduce[T, Acc any](slice []T, initial Acc, fn func(Acc, T) Acc) Acc {
    acc := initial
    for _, v := range slice { acc = fn(acc, v) }
    return acc
}

// GroupBy — group elements by key function
func GroupBy[T any, K comparable](slice []T, key func(T) K) map[K][]T {
    result := make(map[K][]T)
    for _, v := range slice { k := key(v); result[k] = append(result[k], v) }
    return result
}

// FlatMap — map then flatten one level
func FlatMap[T, U any](slice []T, fn func(T) []U) []U {
    var out []U
    for _, v := range slice { out = append(out, fn(v)...) }
    return out
}

// Sum — generic summation
func Sum[T constraints.Ordered](slice []T) T {
    var total T
    for _, v := range slice { total += v }
    return total
}

// Pipeline — builder pattern cho chain operations
type Pipeline[T any] struct{ data []T }

func From[T any](data []T) *Pipeline[T] { return &Pipeline[T]{data: data} }
func (p *Pipeline[T]) Filter(pred func(T) bool) *Pipeline[T] {
    p.data = Filter(p.data, pred); return p
}
func (p *Pipeline[T]) Result() []T { return p.data }

// === REAL WORLD USAGE ===
type Order struct {
    ID         string
    CustomerID string
    Amount     float64
    Paid       bool
}

func ProcessOrders(orders []Order) {
    // Lọc unpaid → tổng tiền → nhóm theo customer
    unpaid := Filter(orders, func(o Order) bool { return !o.Paid })
    total  := Reduce(unpaid, 0.0, func(acc float64, o Order) float64 {
        return acc + o.Amount
    })
    byCustomer := GroupBy(unpaid, func(o Order) string { return o.CustomerID })
    ids := Map(unpaid, func(o Order) string { return o.ID })

    _ = total; _ = byCustomer; _ = ids
}
VÍ DỤ 02 Concurrent Pipeline — channel-based với fan-out/fan-in EXPERT
⚠️ Problem Statement

Xử lý hàng nghìn records từ database: đọc từ DB, enrich từ API ngoài (concurrent), validate, ghi kết quả — cần kiểm soát concurrency và memory.

gopipeline/concurrent.go
package pipeline

import (
    "context"
    "sync"
)

// Stage — function that transforms channel
type Stage[T any] func(ctx context.Context, in <-chan T) <-chan T

// Generator — source of pipeline
func Generator[T any](ctx context.Context, items ...T) <-chan T {
    out := make(chan T, len(items))
    go func() {
        defer close(out)
        for _, item := range items {
            select {
            case <-ctx.Done(): return
            case out <- item:
            }
        }
    }()
    return out
}

// FanOut — distribute to N workers
func FanOut[T any](ctx context.Context, in <-chan T, workers int,
    process func(context.Context, T) T) <-chan T {

    channels := make([]<-chan T, workers)
    for i := 0; i < workers; i++ {
        out := make(chan T)
        channels[i] = out
        go func(ch chan T) {
            defer close(ch)
            for item := range in {
                select {
                case <-ctx.Done(): return
                case ch <- process(ctx, item):
                }
            }
        }(out)
    }
    return FanIn(ctx, channels...)
}

// FanIn — merge N channels into one
func FanIn[T any](ctx context.Context, channels ...<-chan T) <-chan T {
    merged := make(chan T)
    var wg sync.WaitGroup
    wg.Add(len(channels))
    for _, ch := range channels {
        go func(c <-chan T) {
            defer wg.Done()
            for v := range c {
                select {
                case <-ctx.Done(): return
                case merged <- v:
                }
            }
        }(ch)
    }
    go func() { wg.Wait(); close(merged) }()
    return merged
}

// Batch — collect N items then process together
func Batch[T any](ctx context.Context, in <-chan T, size int) <-chan []T {
    out := make(chan []T)
    go func() {
        defer close(out)
        batch := make([]T, 0, size)
        for item := range in {
            batch = append(batch, item)
            if len(batch) >= size {
                select {
                case <-ctx.Done(): return
                case out <- batch:
                }
                batch = make([]T, 0, size)
            }
        }
        if len(batch) > 0 { out <- batch }
    }()
    return out
}

// === EXAMPLE: ETL pipeline ===
type Record struct{ ID int; Data string; Enriched bool }

func ETLPipeline(ctx context.Context, records []Record) []Record {
    // Stage 1: generate
    src := Generator(ctx, records...)

    // Stage 2: enrich with 4 concurrent workers
    enriched := FanOut(ctx, src, 4, func(ctx context.Context, r Record) Record {
        r.Enriched = true // simulate API call
        return r
    })

    // Stage 3: collect results
    var results []Record
    for r := range enriched { results = append(results, r) }
    return results
}
Concurrent Pipeline — Fan-out / Fan-in Flow
📌 Note & Conclusion
  • ✓Functional pipeline (eager) tốt cho small-medium datasets, code đơn giản hơn.
  • ✓Concurrent pipeline (lazy) tốt cho streaming data, IO-bound tasks (API calls, DB reads).
  • ⚠Luôn truyền context.Context để hủy pipeline khi cần.
  • ⚠Fan-out workers chia sẻ cùng một input channel — cẩn thận với close channel race.
  • ✓Dùng Batch() trước sink để giảm số round-trips (bulk insert vs row-by-row).

Object Map Utils
Merge, diff, deep clone, flatten, path access cho map và struct
ADVANCED
📖 Khái niệm
Thao tác nâng cao trên Map và Struct

Trong thực tế, map[string]any xuất hiện khắp nơi: config, JSON response, metadata. Go 1.21+ có package maps nhưng các use-case nâng cao như deep merge, diff, flatten, nested access cần tự implement. Generics giúp viết utilities type-safe và tái sử dụng.

VÍ DỤ 01 Deep Merge & Diff — so sánh và hợp nhất cấu hình ADVANCED
⚠️ Problem Statement

Hệ thống có default config và user config. Cần merge user config vào default (user override), đồng thời detect những key nào đã thay đổi để audit log.

gomaputils/merge.go
package maputils

import (
    "encoding/json"
    "fmt"
    "reflect"
    "strings"
)

// DeepMerge — src overrides dst recursively
func DeepMerge(dst, src map[string]any) map[string]any {
    result := make(map[string]any, len(dst))
    for k, v := range dst { result[k] = v }
    for k, srcVal := range src {
        dstVal, exists := result[k]
        if exists {
            srcMap, srcIsMap := srcVal.(map[string]any)
            dstMap, dstIsMap := dstVal.(map[string]any)
            if srcIsMap && dstIsMap {
                result[k] = DeepMerge(dstMap, srcMap)
                continue
            }
        }
        result[k] = srcVal
    }
    return result
}

// DiffResult — kết quả so sánh hai map
type DiffResult struct {
    Added   map[string]any
    Removed map[string]any
    Changed map[string][2]any // [old, new]
}

func (d DiffResult) HasChanges() bool {
    return len(d.Added)+len(d.Removed)+len(d.Changed) > 0
}

// Diff — compare two maps, returns changes
func Diff(old, new map[string]any) DiffResult {
    result := DiffResult{
        Added:   make(map[string]any),
        Removed: make(map[string]any),
        Changed: make(map[string][2]any),
    }
    for k, oldVal := range old {
        newVal, exists := new[k]
        if !exists { result.Removed[k] = oldVal; continue }
        if !reflect.DeepEqual(oldVal, newVal) {
            result.Changed[k] = [2]any{oldVal, newVal}
        }
    }
    for k, v := range new {
        if _, exists := old[k]; !exists { result.Added[k] = v }
    }
    return result
}

// Flatten — {"a": {"b": 1}} → {"a.b": 1}
func Flatten(m map[string]any, prefix string) map[string]any {
    result := make(map[string]any)
    for k, v := range m {
        key := k
        if prefix != "" { key = prefix + "." + k }
        switch nested := v.(type) {
        case map[string]any:
            for fk, fv := range Flatten(nested, key) { result[fk] = fv }
        default:
            result[key] = v
        }
    }
    return result
}

// Get — access nested value: Get(m, "server.host.port")
func Get(m map[string]any, path string) (any, bool) {
    parts := strings.SplitN(path, ".", 2)
    val, ok := m[parts[0]]
    if !ok || len(parts) == 1 { return val, ok }
    nested, ok := val.(map[string]any)
    if !ok { return nil, false }
    return Get(nested, parts[1])
}

// Set — set nested value: Set(m, "server.port", 8080)
func Set(m map[string]any, path string, value any) {
    parts := strings.SplitN(path, ".", 2)
    if len(parts) == 1 { m[path] = value; return }
    nested, ok := m[parts[0]].(map[string]any)
    if !ok { nested = make(map[string]any); m[parts[0]] = nested }
    Set(nested, parts[1], value)
}

// DeepClone — deep copy via JSON
func DeepClone[T any](v T) (T, error) {
    data, err := json.Marshal(v)
    var out T
    if err != nil { return out, err }
    return out, json.Unmarshal(data, &out)
}

// Keys — all keys sorted
func Keys[K comparable, V any](m map[K]V) []K {
    keys := make([]K, 0, len(m))
    for k := range m { keys = append(keys, k) }
    return keys
}

// Pick — extract subset of keys
func Pick[K comparable, V any](m map[K]V, keys ...K) map[K]V {
    result := make(map[K]V)
    for _, k := range keys {
        if v, ok := m[k]; ok { result[k] = v }
    }
    return result
}

// === REAL WORLD: config management ===
func ExampleConfigMerge() {
    defaultCfg := map[string]any{
        "server": map[string]any{ "host": "localhost", "port": 8080 },
        "db":     map[string]any{ "pool": 10, "timeout": 30 },
    }
    userCfg := map[string]any{
        "server": map[string]any{ "port": 9090 },
    }
    merged := DeepMerge(defaultCfg, userCfg)
    port, _ := Get(merged, "server.port")
    fmt.Println("port:", port) // 9090 (overridden)

    diff := Diff(defaultCfg, merged)
    fmt.Println("changed:", diff.Changed) // server: ...
}

  • ✓Get/Set với dot-path rất hữu ích khi làm việc với nested config hay JSON payloads.
  • ⚠DeepClone qua JSON marshal đủ dùng cho hầu hết cases; dùng reflect nếu cần clone type không marshal được.
  • ✓Diff pattern thiết yếu cho audit log và change detection trong CRUD APIs.

Promise / Async
Future pattern, concurrent tasks, errgroup, timeout & retry
EXPERT
📖 Khái niệm
Go không có Promise — nhưng có tools mạnh hơn

Go không có Promise/async/await như JS/Kotlin. Thay vào đó, Go dùng goroutines + channels và package golang.org/x/sync/errgroup. Có thể tự build Future[T] generic pattern để đạt trải nghiệm tương tự Promise, kết hợp với context timeout và retry.

VÍ DỤ 01 Generic Future[T] — Promise-like pattern EXPERT
⚠️ Problem Statement

Gọi 3 external APIs song song (user info, order history, recommendations). Cần đợi tất cả hoàn thành, collect lỗi riêng biệt, timeout toàn bộ sau 2s.

goasync/future.go
package async

import (
    "context"
    "sync"
    "time"
)

// Future — represents an async computation
type Future[T any] struct {
    done   chan struct{}
    result T
    err    error
    once   sync.Once
}

// Go — start async computation, return Future
func Go[T any](fn func() (T, error)) *Future[T] {
    f := &Future[T]{done: make(chan struct{})}
    go func() {
        defer close(f.done)
        f.result, f.err = fn()
    }()
    return f
}

// Await — block until done or context cancelled
func (f *Future[T]) Await(ctx context.Context) (T, error) {
    select {
    case <-f.done: return f.result, f.err
    case <-ctx.Done(): var zero T; return zero, ctx.Err()
    }
}

// Then — chain transformation (like .then() in JS)
func Then[T, U any](f *Future[T], fn func(T) (U, error)) *Future[U] {
    return Go[U](func() (U, error) {
        v, err := f.Await(context.Background())
        if err != nil { var zero U; return zero, err }
        return fn(v)
    })
}

// All — wait for all futures (like Promise.all)
func All[T any](ctx context.Context, futures ...*Future[T]) ([]T, []error) {
    results := make([]T, len(futures))
    errs    := make([]error, len(futures))

            defer wg.Done()
            results[idx], errs[idx] = fut.Await(ctx)
        }(i, f)
    }
    wg.Wait()
    return results, errs
}

func Parse(s string, formats ...string) (time.Time, error) {
    if len(formats) == 0 {
        formats = []string{ISO8601, DateOnly, time.RFC3339, time.RFC1123}
    }
    for _, f := range formats {
        t, err := time.Parse(f, s)
        if err == nil { return t.UTC(), nil }
    }
    return time.Time{}, fmt.Errorf("cannot parse %q with any known format", s)
}

// InZone — convert to named timezone
func InZone(t time.Time, zone string) (time.Time, error) {
    loc, err := time.LoadLocation(zone)
    if err != nil { return t, fmt.Errorf("unknown timezone %q: %w", zone, err) }
    return t.In(loc), nil
}

// StartOfDay / EndOfDay — calendar helpers

    y, m, d := t.Date()
    return time.Date(y, m, d, 0, 0, 0, 0, t.Location())
}

    return StartOfDay(t).Add(24*time.Hour - time.Nanosecond)
}

// StartOfMonth / EndOfMonth

    y, m, _ := t.Date()
    return time.Date(y, m, 1, 0, 0, 0, 0, t.Location())
}

    return StartOfMonth(t).AddDate(0, 1, 0).Add(-time.Nanosecond)
}

// Age — compute age in years
func Age(birthDate, now time.Time) int {
    years := now.Year() - birthDate.Year()
    if now.Month() < birthDate.Month() ||
        (now.Month() == birthDate.Month() && now.Day() < birthDate.Day()) {
        years--
    }
    return years
}

// BusinessDays — count weekdays between two dates
func BusinessDays(from, to time.Time) int {
    days := 0
    current := from
    for !current.After(to) {
        wd := current.Weekday()
        if wd != time.Saturday && wd != time.Sunday { days++ }
        current = current.AddDate(0, 0, 1)
    }
    return days
}

func HumanDuration(d time.Duration) string {
    switch {
    case d < time.Minute:  return fmt.Sprintf("%.0f giây trước", d.Seconds())
    case d < time.Hour:    return fmt.Sprintf("%.0f phút trước", d.Minutes())
    case d < 24*time.Hour: return fmt.Sprintf("%.0f giờ trước", d.Hours())
    default:               return fmt.Sprintf("%.0f ngày trước", d.Hours()/24)
    }
}

// Measure — time a function execution

    start := time.Now()
    fn()
    return time.Since(start)
}

func TickerWithContext(ctx context.Context, interval time.Duration, fn func()) {
    ticker := time.NewTicker(interval)
    defer ticker.Stop()
    for {
        select {
        case <-ticker.C: fn()
        case <-ctx.Done(): return
        }
    }
}

    OrderConfirmed OrderStatus = "confirmed"
    OrderShipping  OrderStatus = "shipping"
    OrderDelivered OrderStatus = "delivered"
    OrderCancelled OrderStatus = "cancelled"

var validOrderStatuses = map[OrderStatus]bool{
    OrderPending: true, OrderConfirmed: true,
    OrderShipping: true, OrderDelivered: true, OrderCancelled: true,
}

func (s OrderStatus) Valid() bool  { return validOrderStatuses[s] }
func (s OrderStatus) String() string { return string(s) }
func (s OrderStatus) IsFinal() bool {
    return s == OrderDelivered || s == OrderCancelled
}

// Transition rules (state machine)
var allowedTransitions = map[OrderStatus][]OrderStatus{
    OrderPending:   {OrderConfirmed, OrderCancelled},
    OrderConfirmed: {OrderShipping, OrderCancelled},
    OrderShipping:  {OrderDelivered},
}

func (s OrderStatus) CanTransitionTo(next OrderStatus) bool {
    for _, allowed := range allowedTransitions[s] {
        if allowed == next { return true }
    }
    return false
}

    var raw string

    candidate := OrderStatus(raw)

    return nil
}

    Perimeter() float64

func (r Rectangle) isShape() {}
func (t Triangle)  isShape() {}

func (c Circle) Perimeter() float64 { return 2 * 3.14159 * c.Radius }

func (r Rectangle) Area() float64      { return r.Width * r.Height }
func (r Rectangle) Perimeter() float64 { return 2 * (r.Width + r.Height) }

func (t Triangle) Area() float64 {
    s := (t.A + t.B + t.C) / 2
    return math.Sqrt(s*(s-t.A)*(s-t.B)*(s-t.C))
}
func (t Triangle) Perimeter() float64 { return t.A + t.B + t.C }

// Exhaustive switch — Go 1.21+ trick: add compile check
func Describe(s Shape) string {
    switch v := s.(type) {
    case Circle:    return fmt.Sprintf("Circle r=%.2f", v.Radius)
    case Rectangle: return fmt.Sprintf("Rect %.2fx%.2f", v.Width, v.Height)
    case Triangle:  return fmt.Sprintf("Triangle %.2f+%.2f+%.2f", v.A, v.B, v.C)
    default: panic(fmt.Sprintf("unknown shape: %T", s))
    }

// Result[T] — Either monad: Ok(value) or Err(error)
type Result[T any] struct {
    value T
    err   error
}

func Ok[T any](v T) Result[T]        { return Result[T]{value: v} }
func Err[T any](e error) Result[T]    { return Result[T]{err: e} }
func Errf[T any](f string, a ...any) Result[T] {
    return Result[T]{err: fmt.Errorf(f, a...)}
}
func From[T any](v T, err error) Result[T] { return Result[T]{value: v, err: err} }

func (r Result[T]) IsOk() bool      { return r.err == nil }
func (r Result[T]) IsErr() bool     { return r.err != nil }
func (r Result[T]) Unwrap() T       { return r.value }
func (r Result[T]) Error() error    { return r.err }
func (r Result[T]) Must() T {
    if r.err != nil { panic(r.err) }
    return r.value
}
func (r Result[T]) OrDefault(d T) T {
    if r.err != nil { return d }
    return r.value
}
func (r Result[T]) Unpack() (T, error) { return r.value, r.err }

// Map — transform value if Ok
func Map[T, U any](r Result[T], fn func(T) U) Result[U] {
    if r.IsErr() { return Err[U](r.err) }
    return Ok(fn(r.value))
}

// FlatMap — transform that returns Result (monadic bind)
func FlatMap[T, U any](r Result[T], fn func(T) Result[U]) Result[U] {
    if r.IsErr() { return Err[U](r.err) }
    return fn(r.value)
}

func WithStack(err error) error {
    if err == nil { return nil }
    stack := make([]string, 0, 8)
    for i := 1; i < 16; i++ {
        pc, file, line, ok := runtime.Caller(i)
        if !ok { break }
        fn := runtime.FuncForPC(pc)
        if fn != nil && !strings.Contains(fn.Name(), "runtime.") {
            stack = append(stack, fmt.Sprintf("%s:%d %s", file, line, fn.Name()))
        }
    }
    return &ErrorWithStack{cause: err, stack: stack}
}

func (e *ErrorWithStack) Unwrap() error { return e.cause }
func (e *ErrorWithStack) Stack() []string { return e.stack }

// MatchNamed — extract named groups as map
func MatchNamed(re *regexp.Regexp, s string) (map[string]string, bool) {
    match := re.FindStringSubmatch(s)
    if match == nil { return nil, false }
    result := make(map[string]string)
    for i, name := range re.SubexpNames() {
        if i != 0 && name != "" { result[name] = match[i] }
    }
    return result, true
}

// ParseLogLine — structured log parser
type LogEntry struct {
    Timestamp string
    Level     string
    Message   string
    Fields    string
}

func ParseLogLine(line string) (*LogEntry, error) {
    groups, ok := MatchNamed(reLogLine, line)
    if !ok { return nil, fmt.Errorf("invalid log format") }
    return &LogEntry{
        Timestamp: groups["timestamp"],
        Level:     groups["level"],
        Message:   groups["message"],
        Fields:    groups["fields"],
    }, nil

var regexCache struct {
    sync.RWMutex
    m map[string]*regexp.Regexp
}

func Compile(pattern string) (*regexp.Regexp, error) {
    regexCache.RLock()
    re, ok := regexCache.m[pattern]
    regexCache.RUnlock()
    if ok { return re, nil }

    regexCache.Lock()
    defer regexCache.Unlock()
    if regexCache.m == nil { regexCache.m = make(map[string]*regexp.Regexp) }
    re, err := regexp.Compile(pattern)
    if err != nil { return nil, err }
    regexCache.m[pattern] = re
    return re, nil
}

    e.mu.RLock()

    e.mu.RUnlock()
    if !ok { return "", fmt.Errorf("template %q not found", name) }

    var buf bytes.Buffer
    if err := t.Execute(&buf, data); err != nil {
        return "", fmt.Errorf("render %q: %w", name, err)
    }
    return buf.String(), nil
}

// Set — generic set backed by map
type Set[T comparable] struct{ m map[T]struct{} }

func New[T comparable](items ...T) *Set[T] {
    s := &Set[T]{m: make(map[T]struct{}, len(items))}
    for _, item := range items { s.Add(item) }
    return s
}

func (s *Set[T]) Add(items ...T)          { for _, v := range items { s.m[v] = struct{}{} } }
func (s *Set[T]) Remove(items ...T)       { for _, v := range items { delete(s.m, v) } }
func (s *Set[T]) Contains(item T) bool   { _, ok := s.m[item]; return ok }
func (s *Set[T]) Len() int                { return len(s.m) }
func (s *Set[T]) IsEmpty() bool           { return len(s.m) == 0 }

func (s *Set[T]) Items() []T {
    out := make([]T, 0, len(s.m))
    for k := range s.m { out = append(out, k) }
    return out
}

// Union — A ∪ B
func (s *Set[T]) Union(other *Set[T]) *Set[T] {
    result := New(s.Items()...)
    result.Add(other.Items()...)
    return result
}

// Intersection — A ∩ B
func (s *Set[T]) Intersection(other *Set[T]) *Set[T] {
    result := New[T]()
    for k := range s.m {
        if other.Contains(k) { result.Add(k) }
    }
    return result
}

// Difference — A \ B (in A but not in B)
func (s *Set[T]) Difference(other *Set[T]) *Set[T] {
    result := New[T]()
    for k := range s.m {
        if !other.Contains(k) { result.Add(k) }
    }
    return result
}

func (s *Set[T]) IsSubset(other *Set[T]) bool {
    for k := range s.m {
        if !other.Contains(k) { return false }
    }
    return true
}

VÍ DỤ 02 Sharded ConcurrentMap — high-throughput thread-safe map EXPERT
⚠️ Problem Statement

Cache session data cho 10,000+ concurrent users. sync.Mutex trên một map tạo bottleneck. sync.Map không có generic API. Cần giải pháp high-throughput.

goconcurrent/shardedmap.go
package concurrent

import (
    "hash/fnv"
    "sync"
)

const DefaultShards = 32

// shard — one partition of the map
type shard[K comparable, V any] struct {
    sync.RWMutex
    m map[K]V
}

// ShardedMap — reduces lock contention via sharding
type ShardedMap[V any] struct {
    shards [DefaultShards]*shard[string, V]
}

func NewShardedMap[V any]() *ShardedMap[V] {
    sm := &ShardedMap[V]{}
    for i := range sm.shards {
        sm.shards[i] = &shard[string, V]{m: make(map[string]V)}
    }
    return sm
}

func (sm *ShardedMap[V]) shardFor(key string) *shard[string, V] {
    h := fnv.New32a()
    h.Write([]byte(key))
    return sm.shards[h.Sum32()%DefaultShards]
}

func (sm *ShardedMap[V]) Set(key string, val V) {
    sh := sm.shardFor(key)
    sh.Lock(); defer sh.Unlock()
    sh.m[key] = val
}

func (sm *ShardedMap[V]) Get(key string) (V, bool) {
    sh := sm.shardFor(key)
    sh.RLock(); defer sh.RUnlock()
    v, ok := sh.m[key]
    return v, ok
}

func (sm *ShardedMap[V]) Delete(key string) {
    sh := sm.shardFor(key)
    sh.Lock(); defer sh.Unlock()
    delete(sh.m, key)
}

func (sm *ShardedMap[V]) GetOrSet(key string, create func() V) V {
    sh := sm.shardFor(key)
    sh.Lock(); defer sh.Unlock()
    if v, ok := sh.m[key]; ok { return v }
    v := create()
    sh.m[key] = v
    return v
}

func (sm *ShardedMap[V]) Len() int {
    total := 0
    for _, sh := range sm.shards {
        sh.RLock(); total += len(sh.m); sh.RUnlock()
    }
    return total
}

// ForEach — iterate all entries (reads under per-shard lock)
func (sm *ShardedMap[V]) ForEach(fn func(string, V)) {
    for _, sh := range sm.shards {
        sh.RLock()
        for k, v := range sh.m { fn(k, v) }
        sh.RUnlock()
    }
}

// === Session cache usage ===
type Session struct{ UserID int64; Token string }

var sessionCache = NewShardedMap[Session]()

func GetSession(token string) (Session, bool) { return sessionCache.Get(token) }
func SetSession(token string, s Session)     { sessionCache.Set(token, s) }

VÍ DỤ 01 Go 1.23 Range-over-func Iterator — lazy database pagination EXPERT
⚠️ Problem Statement

Xử lý 1 triệu records từ database theo từng trang (page), mỗi page 1000 records. Load tất cả vào memory cùng lúc sẽ OOM. Cần iterator lazy chỉ fetch page khi cần.

goiterator/paginator.go
package iterator

import (
    "context"
    "iter"      // Go 1.23+
    "database/sql"
)

// Seq[V] — Go 1.23 iterator type (yield function)
type Seq[V any] func(yield func(V) bool)

// Seq2[K,V] — key-value iterator
type Seq2[K, V any] func(yield func(K, V) bool)

// Pages — lazy paginator iterator (Go 1.23 style)
func Pages[T any](
    ctx context.Context,
    pageSize int,
    fetch func(ctx context.Context, offset, limit int) ([]T, error),
) iter.Seq2[[]T, error] {
    return func(yield func([]T, error) bool) {
        offset := 0
        for {
            select {
            case <-ctx.Done():
                yield(nil, ctx.Err())
                return
            default:
            }

            page, err := fetch(ctx, offset, pageSize)
            if err != nil {
                yield(nil, err)
                return
            }
            if len(page) == 0 { return } // no more data

            if !yield(page, nil) { return } // caller broke loop

            offset += len(page)
            if len(page) < pageSize { return } // last page
        }
    }
}

// Flatten — Seq2 of pages → Seq of items
func Flatten[T any](pages iter.Seq2[[]T, error]) iter.Seq2[T, error] {
    return func(yield func(T, error) bool) {
        for page, err := range pages {
            if err != nil {
                var zero T
                yield(zero, err)
                return
            }
            for _, v := range page {
                if !yield(v, nil) { return }
            }
        }
    }
}

// Take — limit iterator to first N items
func Take[T any](seq iter.Seq[T], n int) iter.Seq[T] {
    return func(yield func(T) bool) {
        count := 0
        for v := range seq {
            if count >= n || !yield(v) { return }
            count++
        }
    }
}

// Filter over iterator
func Filter[T any](seq iter.Seq[T], pred func(T) bool) iter.Seq[T] {
    return func(yield func(T) bool) {
        for v := range seq {
            if pred(v) && !yield(v) { return }
        }
    }
}

// Collect — materialize iterator to slice
func Collect[T any](seq iter.Seq[T]) []T {
    var out []T
    for v := range seq { out = append(out, v) }
    return out
}

// === Pre-1.23 pattern: closure cursor ===
type Cursor[T any] struct {
    next  func() (T, bool)
    close func()
}

func (c *Cursor[T]) Next() (T, bool) { return c.next() }
func (c *Cursor[T]) Close()        { if c.close != nil { c.close() } }

func NewDBCursor[T any](rows *sql.Rows, scan func(*sql.Rows) (T, error)) *Cursor[T] {
    return &Cursor[T]{
        next: func() (T, bool) {
            var zero T
            if !rows.Next() { return zero, false }
            v, err := scan(rows)
            if err != nil { return zero, false }
            return v, true
        },
        close: func() { rows.Close() },
    }
}

goiterator/tree.go
package iterator

import "iter"

type TreeNode[T any] struct {
    Val   T
    Left  *TreeNode[T]
    Right *TreeNode[T]
}

// InOrder — iterative in-order traversal (no recursion)
func InOrder[T any](root *TreeNode[T]) iter.Seq[T] {
    return func(yield func(T) bool) {
        stack := make([]*TreeNode[T], 0, 32)
        curr := root
        for curr != nil || len(stack) > 0 {
            for curr != nil { stack = append(stack, curr); curr = curr.Left }
            curr = stack[len(stack)-1]
            stack = stack[:len(stack)-1]
            if !yield(curr.Val) { return }
            curr = curr.Right
        }
    }
}

// LevelOrder — BFS level-by-level
func LevelOrder[T any](root *TreeNode[T]) iter.Seq2[int, T] {
    return func(yield func(int, T) bool) {
        if root == nil { return }
        type item struct{ n *TreeNode[T]; level int }
        queue := []item{{root, 0}}
        for len(queue) > 0 {
            cur := queue[0]; queue = queue[1:]
            if !yield(cur.level, cur.n.Val) { return }
            if cur.n.Left  != nil { queue = append(queue, item{cur.n.Left,  cur.level+1}) }
            if cur.n.Right != nil { queue = append(queue, item{cur.n.Right, cur.level+1}) }
        }
    }
}

// Chain — concatenate multiple iterators
func Chain[T any](seqs ...iter.Seq[T]) iter.Seq[T] {
    return func(yield func(T) bool) {
        for _, seq := range seqs {
            for v := range seq { if !yield(v) { return } }
        }
    }
}

// Zip — pair two iterators
func Zip[A, B any](a iter.Seq[A], b iter.Seq[B]) iter.Seq2[A, B] {
    return func(yield func(A, B) bool) {
        nextB, stopB := iter.Pull(b)
        defer stopB()
        for va := range a {
            vb, ok := nextB()
            if !ok || !yield(va, vb) { return }
        }
    }
}

package optional

import (
    "database/sql"
    "encoding/json"
    "fmt"

// Option[T] — represents a value that may or may not be present
type Option[T any] struct {
    value   T
    present bool
}

// Constructors
func Some[T any](v T) Option[T]  { return Option[T]{value: v, present: true} }
func None[T any]() Option[T]      { return Option[T]{} }
func FromPtr[T any](p *T) Option[T] {
    if p == nil { return None[T]() }
    return Some(*p)
}

// Queries
func (o Option[T]) IsPresent() bool { return o.present }
func (o Option[T]) IsEmpty() bool   { return !o.present }

// Unwrap variants
func (o Option[T]) Get() (T, bool) { return o.value, o.present }
func (o Option[T]) Must() T {
    if !o.present { panic("Option.Must: value is absent") }
    return o.value
}
func (o Option[T]) OrDefault(d T) T {
    if !o.present { return d }
    return o.value
}
func (o Option[T]) OrElse(fn func() T) T {
    if !o.present { return fn() }
    return o.value
}
func (o Option[T]) Ptr() *T {
    if !o.present { return nil }
    return &o.value
}

// Map — transform if present
func MapOption[T, U any](o Option[T], fn func(T) U) Option[U] {
    if !o.present { return None[U]() }
    return Some(fn(o.value))
}

// FlatMap — chain optional operations
func FlatMapOption[T, U any](o Option[T], fn func(T) Option[U]) Option[U] {
    if !o.present { return None[U]() }
    return fn(o.value)
}

// JSON marshaling: Some → value, None → null
func (o Option[T]) MarshalJSON() ([]byte, error) {
    if !o.present { return []byte("null"), nil }
    return json.Marshal(o.value)
}
func (o *Option[T]) UnmarshalJSON(data []byte) error {
    if string(data) == "null" { *o = None[T](); return nil }
    o.present = true
    return json.Unmarshal(data, &o.value)
}

type Nullable[T any] struct {
    Val   T
    Valid bool
}

func NullOf[T any](v T, valid bool) Nullable[T] { return Nullable[T]{v, valid} }
func Null[T any]() Nullable[T]                  { return Nullable[T]{} }

func (n *Nullable[T]) Scan(src any) error {

    if ns.Valid { fmt.Sscan(ns.String, &n.Val) }
    return nil
}

    UpdatedAt time.Time  `db:"updated_at" json:"updated_at"`
}
func (t *Timestamped) Touch() { t.UpdatedAt = time.Now().UTC() }

// SoftDeletable — embedded for soft delete support
type SoftDeletable struct {
    DeletedAt *time.Time `db:"deleted_at" json:"deleted_at,omitempty"`
}
func (s *SoftDeletable) Delete()          { now := time.Now().UTC(); s.DeletedAt = &now }
func (s *SoftDeletable) IsDeleted() bool  { return s.DeletedAt != nil }
func (s *SoftDeletable) Restore()         { s.DeletedAt = nil }

// Identifiable — embedded for ID management
type Identifiable struct {
    ID int64 `db:"id" json:"id"`
}
func (e Identifiable) IsNew() bool { return e.ID == 0 }

// === Domain entities via composition ===
type User struct {
    Identifiable                     // embeds ID, IsNew()
    Timestamped                      // embeds CreatedAt, UpdatedAt, Touch()
    SoftDeletable                    // embeds DeletedAt, Delete(), IsDeleted()
    Name  string  `db:"name"  json:"name"`
    Email string  `db:"email" json:"email"`
    Role  string  `db:"role"  json:"role"`
}

// === Generic Repository interface ===
type Repository[T any, ID comparable] interface {
    FindByID(ctx context.Context, id ID) (*T, error)
    FindAll(ctx context.Context, opts QueryOpts) ([]*T, int64, error)
    Save(ctx context.Context, entity *T) error
    Delete(ctx context.Context, id ID) error
}

type QueryOpts struct {
    Page     int
    PageSize int
    OrderBy  string
    Filters  map[string]any
}

// BaseRepository — reusable base with audit logging via decorator
type BaseRepository[T any] struct {
    tableName string
    logger    Logger
}

type Logger interface {
    Info(msg string, args ...any)
    Error(msg string, args ...any)
}

func (r *BaseRepository[T]) logOp(op string, args ...any) {
    r.logger.Info(fmt.Sprintf("[%s] %s", r.tableName, op), args...)
}

// UserRepository — concrete, embeds base
type UserRepository struct {
    BaseRepository[User]
}

func NewUserRepository(logger Logger) *UserRepository {
    return &UserRepository{
        BaseRepository: BaseRepository[User]{tableName: "users", logger: logger},
    }
}

func (r *UserRepository) FindByID(ctx context.Context, id int64) (*User, error) {
    r.logOp("FindByID", "id", id)
    // db.QueryRowContext(ctx, "SELECT * FROM users WHERE id=$1", id)
    return &User{Identifiable: Identifiable{ID: id}, Name: "Alice"}, nil
}

func (r *UserRepository) FindByEmail(ctx context.Context, email string) (*User, error) {
    r.logOp("FindByEmail", "email", email)
    return nil, nil
}

func (r *UserRepository) SoftDelete(ctx context.Context, id int64) error {
    user, err := r.FindByID(ctx, id)
    if err != nil { return err }
    user.Delete() // uses embedded SoftDeletable.Delete()
    r.logOp("SoftDelete", "id", id, "at", user.DeletedAt)
    return nil // UPDATE users SET deleted_at=NOW() WHERE id=$1
}
VÍ DỤ 03 Decorator Pattern — middleware chain cho HTTP handlers EXPERT
⚠️ Problem Statement

Các HTTP handlers cần cross-cutting concerns: logging, auth, rate limiting, metrics, recovery. Không muốn nhét tất cả vào từng handler — cần compose middleware linh hoạt.

gostruct/decorator.go
package middleware

import (
    "context"
    "fmt"
    "log/slog"
    "time"
)

// Handler — core abstraction
type Handler[Req, Resp any] interface {
    Handle(ctx context.Context, req Req) (Resp, error)
}

// HandlerFunc — func adapter (like http.HandlerFunc)
type HandlerFunc[Req, Resp any] func(context.Context, Req) (Resp, error)

func (f HandlerFunc[Req, Resp]) Handle(ctx context.Context, req Req) (Resp, error) {
    return f(ctx, req)
}

// Middleware — wraps a handler
type Middleware[Req, Resp any] func(Handler[Req, Resp]) Handler[Req, Resp]

// Chain — compose middlewares right-to-left
func Chain[Req, Resp any](h Handler[Req, Resp], mws ...Middleware[Req, Resp]) Handler[Req, Resp] {
    for i := len(mws) - 1; i >= 0; i-- { h = mws[i](h) }
    return h
}

// LoggingMiddleware — logs request/response timing
func Logging[Req, Resp any](logger *slog.Logger) Middleware[Req, Resp] {
    return func(next Handler[Req, Resp]) Handler[Req, Resp] {
        return HandlerFunc[Req, Resp](func(ctx context.Context, req Req) (Resp, error) {
            start := time.Now()
            resp, err := next.Handle(ctx, req)
            logger.Info("handled",
                "duration", time.Since(start),
                "error", err,
            )
            return resp, err
        })
    }
}

// RecoveryMiddleware — catches panics
func Recovery[Req, Resp any]() Middleware[Req, Resp] {
    return func(next Handler[Req, Resp]) Handler[Req, Resp] {
        return HandlerFunc[Req, Resp](func(ctx context.Context, req Req) (resp Resp, err error) {
            defer func() {
                if r := recover(); r != nil {
                    err = fmt.Errorf("panic recovered: %v", r)
                }
            }()
            return next.Handle(ctx, req)
        })
    }
}

// TimeoutMiddleware
func Timeout[Req, Resp any](d time.Duration) Middleware[Req, Resp] {
    return func(next Handler[Req, Resp]) Handler[Req, Resp] {
        return HandlerFunc[Req, Resp](func(ctx context.Context, req Req) (Resp, error) {
            ctx2, cancel := context.WithTimeout(ctx, d)
            defer cancel()
            return next.Handle(ctx2, req)
        })
    }
}

// === USAGE: compose handlers ===
type CreateOrderReq struct{ UserID int64; Items []string }
type CreateOrderResp struct{ OrderID string }

func createOrderHandler(ctx context.Context, req CreateOrderReq) (CreateOrderResp, error) {
    return CreateOrderResp{OrderID: "ORD-001"}, nil
}

func RegisterHandlers(logger *slog.Logger) Handler[CreateOrderReq, CreateOrderResp] {
    base := HandlerFunc[CreateOrderReq, CreateOrderResp](createOrderHandler)
    return Chain(base,
        Recovery[CreateOrderReq, CreateOrderResp](),
        Logging[CreateOrderReq, CreateOrderResp](logger),
        Timeout[CreateOrderReq, CreateOrderResp](5 * time.Second),
    )
}

    // Call VNPay API...
    return fmt.Sprintf("vnpay_tx_%f", amount), nil
}
func (v *VNPayStrategy) Refund(ctx context.Context, txID string, amount float64) error {
    return nil
}

// MoMoStrategy
type MoMoStrategy struct{ PartnerCode, AccessKey string }
func (m *MoMoStrategy) Name() string { return "momo" }
func (m *MoMoStrategy) Charge(ctx context.Context, amount float64, meta map[string]string) (string, error) {
    return fmt.Sprintf("momo_tx_%f", amount), nil
}
func (m *MoMoStrategy) Refund(ctx context.Context, txID string, amount float64) error {
    return nil
}

// PaymentService — context using a strategy
type PaymentService struct {
    strategies map[string]PaymentStrategy
}

func NewPaymentService(strategies ...PaymentStrategy) *PaymentService {
    m := make(map[string]PaymentStrategy)
    for _, s := range strategies { m[s.Name()] = s }
    return &PaymentService{strategies: m}
}

func (svc *PaymentService) Pay(ctx context.Context, method string, amount float64, meta map[string]string) (string, error) {
    s, ok := svc.strategies[method]
    if !ok { return "", fmt.Errorf("unsupported payment method: %q", method) }
    return s.Charge(ctx, amount, meta)
}

📌 Note & Conclusion
  • ✓Embedding là composition không phải inheritance — chỉ promote methods/fields, không tạo is-a relationship.
  • ✓Builder pattern giải quyết constructor explosion — MustBuild() cho init-time, Build() với error cho runtime.
  • ✓Strategy + interface cho phép thay đổi behavior runtime mà không sửa calling code — Open/Closed Principle.
  • ⚠Tránh embedding khi không muốn expose methods của embedded type ra ngoài — dùng field thay thế.
  • ✓Decorator middleware chain generic — viết một lần, dùng cho mọi handler type. Compose không cần framework.

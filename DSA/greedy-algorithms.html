Greedy Algorithm là gì?

Greedy Algorithm (thuật toán tham lam) là phương pháp giải bài toán tối ưu hóa: ở mỗi bước, luôn chọn lựa chọn tốt nhất tại thời điểm hiện tại (locally optimal) với hy vọng dẫn đến giải pháp tối ưu toàn cục (globally optimal).

Khác với Dynamic Programming (tính tất cả trạng thái), Greedy ra quyết định ngay và không xem xét lại — thường có độ phức tạp thấp hơn, nhưng không phải bài toán nào cũng dùng được Greedy.

So sánh: Greedy vs Dynamic Programming vs Brute Force

Hai điều kiện để áp dụng Greedy

🏆 1. Greedy Choice Property

Lựa chọn tham lam cục bộ tại mỗi bước có thể đưa vào giải pháp tối ưu toàn cục. Việc chọn "tốt nhất hiện tại" không bao giờ làm mất giải pháp tối ưu.

🔗 2. Optimal Substructure

Giải pháp tối ưu của bài toán chứa giải pháp tối ưu của các bài toán con. Cấu trúc này cho phép xây dựng giải pháp từng bước.

Template Greedy chuẩn trong Go

gogreedy_template.go
package main

import (
    "fmt"
    "sort"
)

// Template Greedy chuẩn:
// 1. Sắp xếp theo tiêu chí tham lam (greedy key)
// 2. Duyệt và đưa ra lựa chọn tham lam tại mỗi bước
// 3. Cập nhật trạng thái
// 4. Trả kết quả

type Item struct {
    Start, End, Value int
}

// GreedyTemplate — O(n log n): sort rồi scan một lần
func GreedyTemplate(items []Item) []Item {
    // Bước 1: định nghĩa tiêu chí tham lam (ví dụ: sort theo End tăng dần)
    sort.Slice(items, func(i, j int) bool {
        return items[i].End < items[j].End
    })

    result := []Item{}
    lastEnd := -1

    for _, item := range items {
        // Bước 2: lựa chọn tham lam — chọn nếu không chồng lắp
        if item.Start >= lastEnd {
            result = append(result, item)
            lastEnd = item.End // Bước 3: cập nhật trạng thái
        }
    }
    return result // Bước 4: kết quả
}

func main() {
    items := []Item{{0,2,10}, {1,4,5}, {3,5,8}, {6,9,3}}
    chosen := GreedyTemplate(items)
    fmt.Println(chosen) // [{0 2 10} {3 5 8} {6 9 3}]
}
⚠️

Lưu ý: Greedy không yêu cầu chứng minh hình thức trong mọi trường hợp, nhưng cần trực giác tốt. Hãy luôn thử counterexample trước khi implement. Câu hỏi kiểm tra: "Nếu tôi chọn A thay vì B, liệu có giải pháp tốt hơn không?"

Interval Scheduling

Cho n khoảng thời gian, mỗi khoảng có [start, end]. Tìm số lượng tối đa các khoảng không chồng lắp nhau (non-overlapping intervals) có thể chọn.

Greedy insight: Luôn chọn khoảng kết thúc sớm nhất (earliest finish time). Bằng cách này ta "giải phóng" thời gian sớm nhất cho các khoảng tiếp theo — đây là Exchange Argument chứng minh tính đúng đắn.

Time
O(n log n)
Space
O(1)
Chiến lược
Sort by End
Minh họa: Sort theo End Time → Greedy chọn

Bước thực hiện

1
Sort theo End Time tăng dần
Đây là bước quan trọng nhất — greedy key
2
Chọn interval đầu tiên (end nhỏ nhất)
Đặt lastEnd = intervals[0].end, count = 1
3
Duyệt các intervals còn lại
Nếu intervals[i].start >= lastEnd: chọn, cập nhật lastEnd
4
Trả về count hoặc danh sách đã chọn
Số lượng intervals không chồng lắp tối đa
✅

Tại sao sort theo End Time? Exchange Argument: giả sử có giải pháp tối ưu chọn X thay vì interval kết thúc sớm nhất Y. Ta có thể hoán đổi X → Y mà không làm mất bất kỳ interval nào khác, vì Y kết thúc không muộn hơn X. Vậy Y luôn an toàn để chọn.

Sample 1: EraseOverlapIntervals — LeetCode 435

gointerval_basic.go
package main

import (
    "fmt"
    "sort"
)

// EraseOverlapIntervals — LeetCode 435
// Trả về số lượng intervals cần XÓA để không còn chồng lắp
// Greedy: giữ tối đa → xóa = n - max_kept
func EraseOverlapIntervals(intervals [][]int) int {
    if len(intervals) == 0 { return 0 }
    sort.Slice(intervals, func(i, j int) bool {
        return intervals[i][1] < intervals[j][1]
    })
    kept, lastEnd := 1, intervals[0][1]
    for i := 1; i < len(intervals); i++ {
        if intervals[i][0] >= lastEnd {
            kept++
            lastEnd = intervals[i][1]
        }
    }
    return len(intervals) - kept
}

// MaxNonOverlap — trả về số lượng chọn được + danh sách
func MaxNonOverlap(intervals [][]int) ([][]int, int) {
    if len(intervals) == 0 { return nil, 0 }
    sort.Slice(intervals, func(i, j int) bool {
        return intervals[i][1] < intervals[j][1]
    })
    result := [][]int{intervals[0]}
    lastEnd := intervals[0][1]
    for i := 1; i < len(intervals); i++ {
        if intervals[i][0] >= lastEnd {
            result = append(result, intervals[i])
            lastEnd = intervals[i][1]
        }
    }
    return result, len(result)
}

func main() {
    fmt.Println(EraseOverlapIntervals([][]int{{1,2},{2,3},{3,4},{1,3}})) // 1
    chosen, count := MaxNonOverlap([][]int{{0,2},{1,4},{3,5},{2,7},{6,9}})
    fmt.Println(count, chosen) // 3  [[0 2] [3 5] [6 9]]
}

Sample 2: Meeting Rooms II — Min Rooms với Min-Heap

gomeeting_rooms.go
package main

import (
    "container/heap"
    "fmt"
    "sort"
)

type MinHeap []int
func (h MinHeap) Len() int           { return len(h) }
func (h MinHeap) Less(i, j int) bool { return h[i] < h[j] }
func (h MinHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }
func (h *MinHeap) Push(x any)        { *h = append(*h, x.(int)) }
func (h *MinHeap) Pop() any {
    old := *h; n := len(old); x := old[n-1]; *h = old[:n-1]; return x
}

// MinMeetingRooms — O(n log n), Min-Heap: LeetCode 253
func MinMeetingRooms(intervals [][]int) int {
    if len(intervals) == 0 { return 0 }
    sort.Slice(intervals, func(i, j int) bool {
        return intervals[i][0] < intervals[j][0]
    })
    rooms := &MinHeap{intervals[0][1]}
    heap.Init(rooms)
    for i := 1; i < len(intervals); i++ {
        if (*rooms)[0] <= intervals[i][0] {
            heap.Pop(rooms) // tái sử dụng phòng kết thúc sớm nhất
        }
        heap.Push(rooms, intervals[i][1])
    }
    return rooms.Len()
}

// MinMeetingRoomsChronological — O(n log n), two separate sorted arrays
func MinMeetingRoomsChronological(intervals [][]int) int {
    n := len(intervals)
    starts, ends := make([]int, n), make([]int, n)
    for i, iv := range intervals { starts[i] = iv[0]; ends[i] = iv[1] }
    sort.Ints(starts); sort.Ints(ends)
    rooms, endPtr := 0, 0
    for i := 0; i < n; i++ {
        if starts[i] < ends[endPtr] { rooms++ } else { endPtr++ }
    }
    return rooms
}

func main() {
    ivs := [][]int{{0,30},{5,10},{15,20}}
    fmt.Println(MinMeetingRooms(ivs))               // 2
    fmt.Println(MinMeetingRoomsChronological(ivs))  // 2
}

Sample 3: Task Scheduler — LeetCode 621

gotask_scheduler.go
package main

import (
    "fmt"
    "sort"
)

// LeastInterval — LeetCode 621: Tasks với cooling n
// Greedy: luôn thực thi task phổ biến nhất còn lại
// Formula: max(len(tasks), (maxFreq-1)*(n+1)+maxCount)
func LeastInterval(tasks []byte, n int) int {
    freq := [26]int{}
    for _, t := range tasks { freq[t-'A']++ }
    sort.Ints(freq[:])
    maxFreq := freq[25]
    idleSlots := (maxFreq - 1) * n
    for i := 24; i >= 0 && idleSlots > 0; i-- {
        fill := freq[i]
        if fill > maxFreq-1 { fill = maxFreq - 1 }
        idleSlots -= fill
    }
    if idleSlots < 0 { idleSlots = 0 }
    return len(tasks) + idleSlots
}

// LeastIntervalFormula — O(1) sau bước đếm
func LeastIntervalFormula(tasks []byte, n int) int {
    freq := [26]int{}
    for _, t := range tasks { freq[t-'A']++ }
    maxFreq, maxCount := 0, 0
    for _, f := range freq { if f > maxFreq { maxFreq = f } }
    for _, f := range freq { if f == maxFreq { maxCount++ } }
    formula := (maxFreq-1)*(n+1) + maxCount
    if len(tasks) > formula { return len(tasks) }
    return formula
}

func main() {
    tasks := []byte("AABABC")
    fmt.Println(LeastInterval(tasks, 2))       // 8
    fmt.Println(LeastIntervalFormula(tasks, 2)) // 8
}

Expert Sample 1: Merge Intervals + Weighted Job Scheduling

gointerval_expert.go
package main

import (
    "fmt"
    "sort"
)

// MergeIntervals — LeetCode 56: O(n log n)
func MergeIntervals(intervals [][]int) [][]int {
    if len(intervals) == 0 { return nil }
    sort.Slice(intervals, func(i, j int) bool {
        return intervals[i][0] < intervals[j][0]
    })
    merged := [][]int{intervals[0]}
    for _, curr := range intervals[1:] {
        last := merged[len(merged)-1]
        if curr[0] <= last[1] {
            if curr[1] > last[1] { last[1] = curr[1] }
        } else {
            merged = append(merged, curr)
        }
    }
    return merged
}

type Job struct { Start, End, Weight int }

// WeightedJobScheduling — DP + Binary Search: O(n log n)
// Tìm tập intervals không chồng lắp có tổng Weight LỚN NHẤT
// Pattern: Greedy Sort + DP + BS
func WeightedJobScheduling(jobs []Job) int {
    n := len(jobs)
    if n == 0 { return 0 }
    sort.Slice(jobs, func(i, j int) bool { return jobs[i].End < jobs[j].End })
    dp := make([]int, n)
    dp[0] = jobs[0].Weight
    for i := 1; i < n; i++ {
        dp[i] = dp[i-1] // không chọn jobs[i]
        lo, hi, lastNO := 0, i-1, -1
        for lo <= hi {
            mid := (lo + hi) / 2
            if jobs[mid].End <= jobs[i].Start { lastNO = mid; lo = mid + 1 } else { hi = mid - 1 }
        }
        withCurr := jobs[i].Weight
        if lastNO >= 0 { withCurr += dp[lastNO] }
        if withCurr > dp[i] { dp[i] = withCurr }
    }
    return dp[n-1]
}

// SweepLine — đếm số overlap tối đa tại bất kỳ thời điểm: O(n log n)
func MaxOverlapAtAnyPoint(intervals [][]int) int {
    events := [][]int{}
    for _, iv := range intervals {
        events = append(events, []int{iv[0], 1})   // start: +1
        events = append(events, []int{iv[1], -1}) // end: -1
    }
    // Cùng thời điểm: end event (-1) trước start event (+1)
    sort.Slice(events, func(i, j int) bool {
        if events[i][0] == events[j][0] { return events[i][1] < events[j][1] }
        return events[i][0] < events[j][0]
    })
    maxOvlp, curr := 0, 0
    for _, e := range events {
        curr += e[1]
        if curr > maxOvlp { maxOvlp = curr }
    }
    return maxOvlp
}

// VideoStitching — LeetCode 1024: cover [0,T] với ít clips nhất
func VideoStitching(clips [][]int, T int) int {
    sort.Slice(clips, func(i, j int) bool {
        if clips[i][0] == clips[j][0] { return clips[i][1] > clips[j][1] }
        return clips[i][0] < clips[j][0]
    })
    count, curEnd, maxEnd, i := 0, 0, 0, 0
    for curEnd < T {
        for i < len(clips) && clips[i][0] <= curEnd {
            if clips[i][1] > maxEnd { maxEnd = clips[i][1] }
            i++
        }
        if maxEnd == curEnd { return -1 }
        count++; curEnd = maxEnd
    }
    return count
}

func main() {
    fmt.Println(MergeIntervals([][]int{{1,3},{2,6},{8,10},{15,18}}))
    // [[1 6] [8 10] [15 18]]

    jobs := []Job{{1,3,50},{2,5,20},{4,6,100},{6,19,200}}
    fmt.Println(WeightedJobScheduling(jobs)) // 350

    fmt.Println(MaxOverlapAtAnyPoint([][]int{{0,2},{1,3},{2,5},{4,7}})) // 2

    clips := [][]int{{0,2},{4,6},{8,10},{1,9},{1,5},{5,9}}
    fmt.Println(VideoStitching(clips, 10)) // 3
}
⚡

Kadane's Algorithm
Maximum Subarray Sum trong O(n) — Greedy one-pass
ADVANCED
💡 Định nghĩa bài toán
Maximum Subarray Sum

Cho mảng số nguyên nums, tìm subarray liên tục có tổng lớn nhất. Kadane (1984) giải quyết trong O(n) với quyết định Greedy đơn giản tại mỗi bước.

Greedy insight: Tại mỗi vị trí i: "Nên extend subarray hiện tại hay bắt đầu mới từ nums[i]?" → Nếu currentSum < 0, bắt đầu mới tốt hơn.

Time
O(n)
Space
O(1)
Chiến lược
One Pass Reset
Kadane: nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4]

Quy tắc Greedy của Kadane

1
Tại mỗi index i: quyết định tham lam
currentSum = max(nums[i], currentSum + nums[i]) — reset nếu currentSum âm
2
Cập nhật maxSum
maxSum = max(maxSum, currentSum) — theo dõi subarray tốt nhất
3
Tại sao Greedy đúng?
Nếu currentSum âm, thêm bất kỳ phần tử nào chỉ làm tổng xấu hơn → bắt đầu mới luôn tốt hơn. Đây là Greedy Choice Property.

gokadane_classic.go
package main

import "fmt"

// MaxSubArray — LeetCode 53: O(n) time, O(1) space
func MaxSubArray(nums []int) int {
    maxSum, curSum := nums[0], nums[0]
    for i := 1; i < len(nums); i++ {
        if curSum < 0 { curSum = nums[i] } else { curSum += nums[i] }
        if curSum > maxSum { maxSum = curSum }
    }
    return maxSum
}

// MaxSubArrayWithBounds — trả về [sum, startIdx, endIdx]
func MaxSubArrayWithBounds(nums []int) (int, int, int) {
    maxSum, curSum := nums[0], nums[0]
    start, end, tempStart := 0, 0, 0
    for i := 1; i < len(nums); i++ {
        if curSum < 0 { curSum = nums[i]; tempStart = i } else { curSum += nums[i] }
        if curSum > maxSum { maxSum = curSum; start = tempStart; end = i }
    }
    return maxSum, start, end
}

// MaxSumCircularSubarray — LeetCode 918
// max circular = max(max_normal, totalSum - min_normal)
func MaxSumCircularSubarray(nums []int) int {
    totalSum := 0
    maxSum, curMax := nums[0], nums[0]
    minSum, curMin := nums[0], nums[0]
    for i := 1; i < len(nums); i++ {
        totalSum += nums[i-1]
        if curMax < 0 { curMax = nums[i] } else { curMax += nums[i] }
        if curMax > maxSum { maxSum = curMax }
        if curMin > 0 { curMin = nums[i] } else { curMin += nums[i] }
        if curMin < minSum { minSum = curMin }
    }
    totalSum += nums[len(nums)-1]
    if maxSum < 0 { return maxSum } // tất cả âm
    circular := totalSum - minSum
    if maxSum > circular { return maxSum }
    return circular
}

func main() {
    nums := []int{-2,1,-3,4,-1,2,1,-5,4}
    fmt.Println(MaxSubArray(nums)) // 6

    sum, s, e := MaxSubArrayWithBounds(nums)
    fmt.Printf("sum=%d [%d,%d]\n", sum, s, e) // sum=6 [3,6]

    fmt.Println(MaxSumCircularSubarray([]int{5,-3,5})) // 10
}

Sample 2: Best Time to Buy Stock (Kadane variant)

gostock_profit.go
package main

import "fmt"

// MaxProfit — LeetCode 121: single transaction
// Equivalent: max subarray sum of price differences
func MaxProfit(prices []int) int {
    minPrice, maxProfit := prices[0], 0
    for _, p := range prices[1:] {
        profit := p - minPrice
        if profit > maxProfit { maxProfit = profit }
        if p < minPrice { minPrice = p }
    }
    return maxProfit
}

// MaxProfitMultiple — LeetCode 122: unlimited transactions
// Greedy: take every upward move
func MaxProfitMultiple(prices []int) int {
    profit := 0
    for i := 1; i < len(prices); i++ {
        if prices[i] > prices[i-1] { profit += prices[i] - prices[i-1] }
    }
    return profit
}

// MaxSumSubarrayExactlyK — sliding window: O(n)
func MaxSumSubarrayExactlyK(nums []int, k int) int {
    if len(nums) < k { return 0 }
    window := 0
    for i := 0; i < k; i++ { window += nums[i] }
    maxVal := window
    for i := k; i < len(nums); i++ {
        window += nums[i] - nums[i-k]
        if window > maxVal { maxVal = window }
    }
    return maxVal
}

func main() {
    fmt.Println(MaxProfit([]int{7,1,5,3,6,4}))         // 5
    fmt.Println(MaxProfitMultiple([]int{7,1,5,3,6,4})) // 7
    fmt.Println(MaxSumSubarrayExactlyK([]int{1,4,2,10,23,3}, 3)) // 35
}

package main

import (
    "fmt"
    "math"
)

// MaxSumRectangle — 2D Kadane: O(n²m)
// Fix 2 rows (top, bot) → nén thành 1D → chạy Kadane
func MaxSumRectangle(matrix [][]int) int {
    rows := len(matrix)
    if rows == 0 { return 0 }
    cols := len(matrix[0])
    maxSum := matrix[0][0]
    for top := 0; top < rows; top++ {
        temp := make([]int, cols)
        for bot := top; bot < rows; bot++ {
            for c := 0; c < cols; c++ { temp[c] += matrix[bot][c] }
            // Kadane on 1D temp
            curSum, localMax := temp[0], temp[0]
            for c := 1; c < cols; c++ {
                if curSum < 0 { curSum = temp[c] } else { curSum += temp[c] }
                if curSum > localMax { localMax = curSum }
            }
            if localMax > maxSum { maxSum = localMax }
        }
    }
    return maxSum
}

// MaxSubArrayMinLen — max subarray sum với chiều dài tối thiểu k: O(n)
// Dùng prefix sum: tại j, tìm min prefix[i] với i <= j-k
func MaxSubArrayMinLen(nums []int, k int) int {
    n := len(nums)
    if n < k { return math.MinInt32 }
    prefix := make([]int, n+1)
    for i, v := range nums { prefix[i+1] = prefix[i] + v }
    minPrefix := prefix[0]
    result := math.MinInt32
    for j := k; j <= n; j++ {
        val := prefix[j] - minPrefix
        if val > result { result = val }
        if prefix[j-k+1] < minPrefix { minPrefix = prefix[j-k+1] }
    }
    return result
}

// StreamingKadane — Kadane trên data stream (online algorithm)
// Không cần toàn bộ mảng trong bộ nhớ — O(1) per element
type KadaneStream struct {
    CurrentSum  int
    MaxSum      int
    initialized bool
}

func (ks *KadaneStream) Feed(val int) {
    if !ks.initialized { ks.CurrentSum = val; ks.MaxSum = val; ks.initialized = true; return }
    if ks.CurrentSum < 0 { ks.CurrentSum = val } else { ks.CurrentSum += val }
    if ks.CurrentSum > ks.MaxSum { ks.MaxSum = ks.CurrentSum }
}

func main() {
    matrix := [][]int{{1,-2,-1,4},{1,-1,1,1},{0,-1,1,1},{0,0,-1,0}}
    fmt.Println(MaxSumRectangle(matrix)) // 7

    fmt.Println(MaxSubArrayMinLen([]int{-1,2,3,-4,5}, 2)) // 6

    ks := &KadaneStream{}
    for _, v := range []int{-2,1,-3,4,-1,2,1} { ks.Feed(v) }
    fmt.Println(ks.MaxSum) // 6
}

Longest Increasing Subsequence (LIS)
Patience Sorting — O(n log n) với Greedy + Binary Search
ADVANCED
💡 Định nghĩa bài toán
Longest Increasing Subsequence

Cho mảng nums, tìm dãy con (subsequence) tăng nghiêm ngặt có độ dài lớn nhất. Subsequence không cần liên tục, chỉ giữ nguyên thứ tự tương đối.

Ví dụ: [10,9,2,5,3,7,101,18] → LIS = [2,3,7,101], độ dài = 4.

Greedy insight (Patience Sorting): Duy trì mảng tails với tails[i] là phần tử kết thúc nhỏ nhất của mọi IS có độ dài i+1. Với mỗi phần tử mới, dùng Binary Search tìm vị trí thay thế hoặc mở rộng.

DP naive
O(n²)
Patience Sort
O(n log n)
Space
O(n)
Patience Sorting: [3, 1, 4, 1, 5, 9, 2, 6] — Xây dựng mảng tails
⚠️

Quan trọng: Mảng tails trong Patience Sorting KHÔNG phải là actual LIS. Nó chỉ cho ta độ dài chính xác. Để reconstruct actual LIS, cần lưu thêm parent pointer array riêng.

package main

import "fmt"

// LengthOfLIS_DP — O(n²) time, O(n) space — LeetCode 300
func LengthOfLIS_DP(nums []int) int {
    n := len(nums)
    if n == 0 { return 0 }
    dp := make([]int, n)
    for i := range dp { dp[i] = 1 }
    result := 1
    for i := 1; i < n; i++ {
        for j := 0; j < i; j++ {
            if nums[j] < nums[i] && dp[j]+1 > dp[i] { dp[i] = dp[j] + 1 }
        }
        if dp[i] > result { result = dp[i] }
    }
    return result
}

// ReconstructLIS_DP — trả về actual LIS sequence: O(n²)
func ReconstructLIS_DP(nums []int) []int {
    n := len(nums)
    if n == 0 { return nil }
    dp := make([]int, n)
    parent := make([]int, n)
    for i := range dp { dp[i] = 1; parent[i] = -1 }
    maxLen, maxIdx := 1, 0
    for i := 1; i < n; i++ {
        for j := 0; j < i; j++ {
            if nums[j] < nums[i] && dp[j]+1 > dp[i] {
                dp[i] = dp[j] + 1
                parent[i] = j
            }
        }
        if dp[i] > maxLen { maxLen = dp[i]; maxIdx = i }
    }
    seq := make([]int, 0, maxLen)
    for maxIdx != -1 { seq = append(seq, nums[maxIdx]); maxIdx = parent[maxIdx] }
    // Đảo ngược
    for l, r := 0, len(seq)-1; l < r; l, r = l+1, r-1 { seq[l], seq[r] = seq[r], seq[l] }
    return seq
}

func main() {
    nums := []int{10, 9, 2, 5, 3, 7, 101, 18}
    fmt.Println(LengthOfLIS_DP(nums))    // 4
    fmt.Println(ReconstructLIS_DP(nums)) // [2 3 7 101]
}

Sample 2: Patience Sorting O(n log n) + Reconstruct

golis_patience.go
package main

import (
    "fmt"
    "sort"
)

// LengthOfLIS — O(n log n): Patience Sorting — LeetCode 300
func LengthOfLIS(nums []int) int {
    tails := []int{}
    for _, num := range nums {
        // Binary search: tìm vị trí đầu tiên trong tails >= num
        pos := sort.SearchInts(tails, num)
        if pos == len(tails) {
            tails = append(tails, num) // mở rộng LIS
        } else {
            tails[pos] = num // greedy replace: giữ tail nhỏ nhất có thể
        }
    }
    return len(tails)
}

// LengthOfLIS_NonDecreasing — cho phép bằng nhau
func LengthOfLIS_NonDecreasing(nums []int) int {
    tails := []int{}
    for _, num := range nums {
        pos := sort.Search(len(tails), func(i int) bool { return tails[i] > num })
        if pos == len(tails) { tails = append(tails, num) } else { tails[pos] = num }
    }
    return len(tails)
}

// LISWithReconstruct — O(n log n) + actual sequence
func LISWithReconstruct(nums []int) ([]int, int) {
    n := len(nums)
    tails := []int{}
    tailsIdx := []int{}    // index trong nums của mỗi tail
    parent := make([]int, n)
    for i := range parent { parent[i] = -1 }

    for i, num := range nums {
        pos := sort.SearchInts(tails, num)
        if pos == len(tails) {
            tails = append(tails, num)
            tailsIdx = append(tailsIdx, i)
        } else {
            tails[pos] = num
            tailsIdx[pos] = i
        }
        if pos > 0 { parent[i] = tailsIdx[pos-1] }
    }

    seq := make([]int, 0, len(tails))
    cur := tailsIdx[len(tailsIdx)-1]
    for cur != -1 { seq = append(seq, nums[cur]); cur = parent[cur] }
    for l, r := 0, len(seq)-1; l < r; l, r = l+1, r-1 { seq[l], seq[r] = seq[r], seq[l] }
    return seq, len(tails)
}

func main() {
    nums := []int{10, 9, 2, 5, 3, 7, 101, 18}
    fmt.Println(LengthOfLIS(nums)) // 4

    seq, length := LISWithReconstruct(nums)
    fmt.Printf("LIS: %v, length=%d\n", seq, length) // [2 3 7 101] / 4
}

package main

import (
    "fmt"
    "sort"
)

// FindNumberOfLIS — LeetCode 673: đếm số lượng LIS — O(n²)
func FindNumberOfLIS(nums []int) int {
    n := len(nums)
    if n == 0 { return 0 }
    dp := make([]int, n)  // dp[i] = length ending at i
    cnt := make([]int, n) // cnt[i] = số LIS có độ dài dp[i] ending at i
    for i := range dp { dp[i] = 1; cnt[i] = 1 }
    maxLen := 1
    for i := 1; i < n; i++ {
        for j := 0; j < i; j++ {
            if nums[j] < nums[i] {
                if dp[j]+1 > dp[i] { dp[i] = dp[j] + 1; cnt[i] = cnt[j] } else if dp[j]+1 == dp[i] { cnt[i] += cnt[j] }
            }
        }
        if dp[i] > maxLen { maxLen = dp[i] }
    }
    total := 0
    for i, l := range dp { if l == maxLen { total += cnt[i] } }
    return total
}

// MaxEnvelopes — LeetCode 354: Russian Doll Envelopes — O(n log n)
// Sort: width ASC, height DESC khi width bằng nhau
// → LIS strictly increasing trên heights
// Lý do sort height DESC: tránh pick 2 envelopes cùng width
func MaxEnvelopes(envelopes [][]int) int {
    sort.Slice(envelopes, func(i, j int) bool {
        if envelopes[i][0] == envelopes[j][0] { return envelopes[i][1] > envelopes[j][1] }
        return envelopes[i][0] < envelopes[j][0]
    })
    tails := []int{}
    for _, e := range envelopes {
        h := e[1]
        pos := sort.SearchInts(tails, h)
        if pos == len(tails) { tails = append(tails, h) } else { tails[pos] = h }
    }
    return len(tails)
}

// LongestChain — LeetCode 646: Maximum Length of Pair Chain
// Sort by end, Greedy pick if pair[0] > lastEnd
func FindLongestChain(pairs [][]int) int {
    sort.Slice(pairs, func(i, j int) bool { return pairs[i][1] < pairs[j][1] })
    count, lastEnd := 0, -1<<31
    for _, p := range pairs {
        if p[0] > lastEnd { count++; lastEnd = p[1] }
    }
    return count
}

func main() {
    fmt.Println(FindNumberOfLIS([]int{1, 3, 5, 4, 7})) // 2

    envs := [][]int{{5,4},{6,4},{6,7},{2,3}}
    fmt.Println(MaxEnvelopes(envs)) // 3: [2,3] -> [5,4] -> [6,7]

    pairs := [][]int{{1,2},{2,3},{3,4}}
    fmt.Println(FindLongestChain(pairs)) // 2
}

Jump Game
Track maxReach — Greedy one-pass cho bài toán nhảy
ADVANCED
💡 Định nghĩa bài toán
Jump Game I & II

Cho mảng nums, với nums[i] là số bước tối đa có thể nhảy từ vị trí i. Xuất phát từ index 0:

Jump Game I (LC 55): Có thể đến index cuối không?
Jump Game II (LC 45): Số bước nhảy tối thiểu để đến cuối.

Greedy insight: Không cần biết đường đi cụ thể. Chỉ cần track maxReach — vị trí xa nhất có thể đến. Đây là BFS level-order: khi vượt boundary của level hiện tại → buộc phải nhảy thêm một lần.

Jump Game I
O(n)
Jump Game II
O(n)
Space
O(1)
Jump Game II: nums=[2,3,1,1,4] — BFS Level Greedy

Bước thực hiện Jump Game II

1
Duyệt từ 0 đến n-2 (không cần index cuối)
Cập nhật maxReach = max(maxReach, i + nums[i])
2
Khi i đạt curEnd → hết bước của level hiện tại
jumps++; curEnd = maxReach — bắt đầu "level" nhảy tiếp theo
3
Tại sao Greedy đúng?
Ta luôn chọn điểm cuối của mỗi "level BFS" để nhảy đến. Tại level k, ta có thể đến mọi điểm trong [curEnd+1, maxReach]. Để minimize jumps, ta chọn maxReach làm boundary của level k+1.

package main

import "fmt"

// CanJump — LeetCode 55: Jump Game I — O(n) O(1)
func CanJump(nums []int) bool {
    maxReach := 0
    for i, v := range nums {
        if i > maxReach { return false } // không thể đến i
        if i+v > maxReach { maxReach = i + v }
    }
    return true
}

// Jump — LeetCode 45: Jump Game II — minimum jumps: O(n) O(1)
// Tương đương BFS level-order traversal
func Jump(nums []int) int {
    jumps, curEnd, maxReach := 0, 0, 0
    for i := 0; i < len(nums)-1; i++ {
        if i+nums[i] > maxReach { maxReach = i + nums[i] }
        if i == curEnd { // hết level hiện tại
            jumps++
            curEnd = maxReach
        }
    }
    return jumps
}

// JumpWithPath — trả về đường đi tối ưu: O(n) O(n)
func JumpWithPath(nums []int) ([]int, int) {
    n := len(nums)
    if n == 1 { return []int{0}, 0 }
    // DP để reconstruct: best[i] = index tốt nhất để đến i
    dist := make([]int, n)
    best := make([]int, n)
    for i := range dist { dist[i] = n + 1; best[i] = -1 }
    dist[0] = 0
    for i := 0; i < n; i++ {
        for j := i+1; j <= i+nums[i] && j < n; j++ {
            if dist[i]+1 < dist[j] { dist[j] = dist[i] + 1; best[j] = i }
        }
    }
    path := []int{}
    cur := n - 1
    for cur != -1 { path = append(path, cur); cur = best[cur] }
    for l, r := 0, len(path)-1; l < r; l, r = l+1, r-1 { path[l], path[r] = path[r], path[l] }
    return path, dist[n-1]
}

// JumpGameV — LeetCode 1340: Jump Game V (DP)
// Từ i nhảy đến j nếu |i-j| <= d và nums[j] < nums[i] (và không block)
// Đếm số vị trí tối đa có thể thăm
func MaxJumps(arr []int, d int) int {
    n := len(arr)
    memo := make([]int, n)
    var dfs func(i int) int
    dfs = func(i int) int {
        if memo[i] != 0 { return memo[i] }
        best := 1
        // nhảy sang phải
        for j := i+1; j <= i+d && j < n && arr[j] < arr[i]; j++ {
            if v := dfs(j) + 1; v > best { best = v }
        }
        // nhảy sang trái
        for j := i-1; j >= i-d && j >= 0 && arr[j] < arr[i]; j-- {
            if v := dfs(j) + 1; v > best { best = v }
        }
        memo[i] = best
        return best
    }
    result := 0
    for i := 0; i < n; i++ { if v := dfs(i); v > result { result = v } }
    return result
}

func main() {
    fmt.Println(CanJump([]int{2,3,1,1,4})) // true
    fmt.Println(CanJump([]int{3,2,1,0,4})) // false
    fmt.Println(Jump([]int{2,3,1,1,4}))    // 2

    path, jumps := JumpWithPath([]int{2,3,1,1,4})
    fmt.Printf("path=%v jumps=%d\n", path, jumps) // [0 1 4] 2

    fmt.Println(MaxJumps([]int{6,4,14,6,8,13,9,7,10,6,12}, 2)) // 4
}

gojump_expert.go
package main

import "fmt"

// MinJumps — LeetCode 1345: Jump Game IV — O(n)
// Từ i có thể nhảy i±1 hoặc bất kỳ j: arr[j]==arr[i]
// BFS + HashMap precompute
func MinJumps(arr []int) int {
    n := len(arr)
    if n == 1 { return 0 }
    graph := make(map[int][]int)
    for i, v := range arr { graph[v] = append(graph[v], i) }

    visited := make([]bool, n)
    visited[0] = true
    queue := []int{0}
    steps := 0

    for len(queue) > 0 {
        steps++
        next := []int{}
        for _, i := range queue {
            for _, j := range []int{i-1, i+1} {
                if j >= 0 && j < n && !visited[j] {
                    if j == n-1 { return steps }
                    visited[j] = true; next = append(next, j)
                }
            }
            if same, ok := graph[arr[i]]; ok {
                for _, j := range same {
                    if !visited[j] {
                        if j == n-1 { return steps }
                        visited[j] = true; next = append(next, j)
                    }
                }
                delete(graph, arr[i]) // KEY: xóa để tránh revisit O(n²) → O(n)
            }
        }
        queue = next
    }
    return -1
}

// MinTaps — LeetCode 1326: Minimum Number of Taps to Open to Water a Garden
// Tương đương Jump Game II / Video Stitching
// Taps[i] có thể tưới [i-taps[i], i+taps[i]]
// Greedy: convert thành intervals, dùng Jump II approach
func MinTaps(n int, ranges []int) int {
    // Chuyển đổi mỗi tap thành interval, lưu max reach tại start
    maxReachAt := make([]int, n+1)
    for i, r := range ranges {
        left := i - r
        if left < 0 { left = 0 }
        right := i + r
        if right > n { right = n }
        if right > maxReachAt[left] { maxReachAt[left] = right }
    }
    // Bây giờ dùng Jump II Greedy
    taps, curEnd, maxEnd := 0, 0, 0
    for i := 0; i < n; i++ {
        if maxReachAt[i] > maxEnd { maxEnd = maxReachAt[i] }
        if i == curEnd {
            if maxEnd == curEnd { return -1 } // không thể cover
            taps++; curEnd = maxEnd
        }
    }
    return taps
}

// FrogJump — LeetCode 403: k-1/k/k+1 jumps on stones — O(n²) DP
func FrogJump(stones []int) bool {
    stoneSet := make(map[int]map[int]bool)
    for _, s := range stones { stoneSet[s] = make(map[int]bool) }
    stoneSet[stones[0]][0] = true
    last := stones[len(stones)-1]
    for _, s := range stones {
        for k := range stoneSet[s] {
            for _, next := range []int{k-1, k, k+1} {
                if next <= 0 { continue }
                target := s + next
                if _, ok := stoneSet[target]; ok {
                    stoneSet[target][next] = true
                    if target == last { return true }
                }
            }
        }
    }
    return false
}

func main() {
    fmt.Println(MinJumps([]int{100,-23,-23,404,100,23,23,23,3,404})) // 3
    fmt.Println(MinTaps(5, []int{3,4,1,1,0,0})) // 1
    fmt.Println(FrogJump([]int{0,1,3,5,6,8,12,17})) // true
}

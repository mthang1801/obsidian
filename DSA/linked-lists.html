Khái niệm cơ bản về Linked List

Linked List là một cấu trúc dữ liệu tuyến tính trong đó mỗi phần tử (Node) chứa dữ liệu và một hoặc nhiều con trỏ (pointer) trỏ đến node tiếp theo (hoặc cả trước đó trong Doubly LL). Không giống mảng, các node không cần nằm liền kề trong bộ nhớ — chúng được liên kết thông qua địa chỉ.

Trong Go, không có kiểu Linked List built-in như Java/C++. Ta tự định nghĩa bằng struct và pointer. Thư viện chuẩn có container/list nhưng trong phỏng vấn và DSA ta thường tự implement.

Cấu trúc bộ nhớ — Singly vs Doubly Linked List
GO node.go — Định nghĩa cơ bản 
// ListNode — node cơ bản cho Singly Linked List
type ListNode struct {
    Val  int
    Next *ListNode
}

// DListNode — node cho Doubly Linked List
type DListNode struct {
    Val        int
    Prev, Next *DListNode
}

// NewNode tạo một node mới với giá trị val
func NewNode(val int) *ListNode {
    return &ListNode{Val: val}
}

// Build tạo linked list từ slice — tiện cho testing
func Build(vals ...int) *ListNode {
    dummy := &ListNode{}
    cur := dummy
    for _, v := range vals {
        cur.Next = &ListNode{Val: v}
        cur = cur.Next
    }
    return dummy.Next
}

// ToSlice chuyển linked list thành []int để so sánh trong test
func ToSlice(head *ListNode) []int {
    var res []int
    for head != nil {
        res = append(res, head.Val)
        head = head.Next
    }
    return res
}

Reversal — Đảo Ngược Linked List Advanced
Đảo ngược một linked list bằng cách tái dây (rewire) con trỏ Next mà không cần thêm bộ nhớ phụ. Đây là bài toán nền tảng xuất hiện trong hầu hết mọi cuộc phỏng vấn hệ thống.

🔍 Định nghĩa & Tư duy

Reversal = thay đổi hướng của tất cả con trỏ Next. Node cuối trở thành HEAD mới, node HEAD cũ trỏ về nil.

Kỹ thuật cốt lõi: duy trì 3 con trỏ đồng thời — prev, cur, next. Mỗi bước: lưu next, đảo hướng, tiến một bước.

Quá trình đảo ngược từng bước

Giải pháp Iterative — Advanced

GO reverse_iterative.go 
// ReverseList đảo ngược toàn bộ linked list.
// Time: O(n) | Space: O(1) — rewire tại chỗ, không dùng bộ nhớ phụ.
func ReverseList(head *ListNode) *ListNode {
    var prev *ListNode // ban đầu prev = nil
    cur := head

    for cur != nil {
        next := cur.Next  // 1. lưu con trỏ next trước khi đứt
        cur.Next = prev    // 2. đảo chiều con trỏ
        prev = cur         // 3. tiến prev
        cur = next          // 4. tiến cur
    }
    return prev // prev bây giờ là HEAD mới
}

// ReverseListRecursive — giải pháp đệ quy.
// Time: O(n) | Space: O(n) call stack — dùng khi muốn giải thích elegance.
func ReverseListRecursive(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return head // base case: 0 hoặc 1 node
    }
    newHead := ReverseListRecursive(head.Next)
    // sau khi đệ quy: head.Next trỏ đến node cuối đã xử lý
    head.Next.Next = head // node kế trỏ ngược lại head
    head.Next = nil       // cắt con trỏ cũ
    return newHead
}

Đảo ngược đoạn [left, right] — Expert

GO reverse_between.go — LeetCode 92 
// ReverseBetween đảo ngược đoạn từ vị trí left đến right (1-indexed).
// Ví dụ: [1,2,3,4,5], left=2, right=4 → [1,4,3,2,5]
// Time: O(n) | Space: O(1)
func ReverseBetween(head *ListNode, left, right int) *ListNode {
    if head == nil || left == right {
        return head
    }

    dummy := &ListNode{Next: head}
    pre := dummy

    // tiến pre đến node ngay trước vị trí left
    for i := 0; i < left-1; i++ {
        pre = pre.Next
    }

    cur := pre.Next
    for i := 0; i < right-left; i++ {
        // "head insertion" — chèn nxt vào đầu đoạn đảo
        nxt := cur.Next
        cur.Next = nxt.Next
        nxt.Next = pre.Next
        pre.Next = nxt
    }
    return dummy.Next
}

// ReverseKGroup đảo ngược từng nhóm k node — Expert level.
// [1,2,3,4,5], k=2 → [2,1,4,3,5]
func ReverseKGroup(head *ListNode, k int) *ListNode {
    dummy := &ListNode{Next: head}
    groupPrev := dummy

    for {
        // kiểm tra còn đủ k node không
        kth := getKth(groupPrev, k)
        if kth == nil {
            break
        }
        groupNext := kth.Next

        // đảo ngược đoạn [groupPrev.Next .. kth]
        prev, cur := groupNext, groupPrev.Next
        for cur != groupNext {
            nxt := cur.Next
            cur.Next = prev
            prev = cur
            cur = nxt
        }
        tmp := groupPrev.Next
        groupPrev.Next = kth
        groupPrev = tmp
    }
    return dummy.Next
}

func getKth(cur *ListNode, k int) *ListNode {
    for cur != nil && k > 0 {
        cur = cur.Next; k--
    }
    return cur
}

Remove Kth From End — Xóa Node Thứ K Từ Cuối Advanced
Dùng kỹ thuật Two Pointers (Fast & Slow): con trỏ nhanh đi trước k bước, sau đó cả hai tiến cùng tốc độ. Khi fast đến cuối, slow đang ở node cần xóa. Một lần duyệt, O(n) time, O(1) space.
Two-Pointer: Fast đi trước K bước
GO remove_nth_from_end.go 
// RemoveNthFromEnd xóa node thứ n từ cuối danh sách.
// Sử dụng dummy node để xử lý edge case xóa head.
// Time: O(n) — một lần duyệt | Space: O(1)
func RemoveNthFromEnd(head *ListNode, n int) *ListNode {
    dummy := &ListNode{Next: head}
    fast, slow := dummy, dummy

    // fast đi trước n+1 bước để slow dừng ở node TRƯỚC node cần xóa
    for i := 0; i <= n; i++ {
        if fast == nil {
            return head // n lớn hơn độ dài list
        }
        fast = fast.Next
    }

    // tiến cả hai cho đến khi fast = nil
    for fast != nil {
        fast = fast.Next
        slow = slow.Next
    }

    // slow.Next là node cần xóa
    slow.Next = slow.Next.Next
    return dummy.Next
}

// FindKthFromEnd chỉ tìm, không xóa — dùng trong nhiều bài khác.
func FindKthFromEnd(head *ListNode, k int) *ListNode {
    fast, slow := head, head
    for i := 0; i < k; i++ {
        if fast == nil { return nil }
        fast = fast.Next
    }
    for fast != nil {
        fast = fast.Next
        slow = slow.Next
    }
    return slow
}

// RemoveNthTest — table-driven test
func TestRemoveNthFromEnd(t *testing.T) {
    cases := []struct{
        vals   []int
        n      int
        expect []int
    }{
        {[]int{1,2,3,4,5}, 2, []int{1,2,3,5}},
        {[]int{1},           1, []int{}},
        {[]int{1,2},         1, []int{1}},
    }
    for _, c := range cases {
        got := ToSlice(RemoveNthFromEnd(Build(c.vals...), c.n))
        if !reflect.DeepEqual(got, c.expect) {
            t.Errorf("got %v want %v", got, c.expect)
        }
    }
}

Intersection — Tìm Điểm Giao Nhau Advanced
Hai linked list có thể chia sẻ phần đuôi chung (cùng địa chỉ node, không chỉ cùng giá trị). Kỹ thuật Two-pointer alignment: khi pointer đến cuối, nhảy sang đầu list kia — hai pointer sẽ gặp nhau tại điểm giao sau đúng lenA + lenB bước.
Kỹ thuật chuyển đổi pointer qua list kia
GO intersection.go 
// GetIntersectionNode trả về node giao nhau của hai linked list.
// Nếu không giao: trả về nil.
// Trick: khi pA hết list A, chuyển sang head của B (và ngược lại).
// Sau lenA+lenB bước, hai pointer sẽ cùng ở điểm giao (hoặc cùng nil).
// Time: O(m+n) | Space: O(1)
func GetIntersectionNode(headA, headB *ListNode) *ListNode {
    if headA == nil || headB == nil {
        return nil
    }
    pA, pB := headA, headB
    for pA != pB {
        // nil thì chuyển sang đầu list kia, không phải nil thì tiến
        if pA == nil {
            pA = headB
        } else {
            pA = pA.Next
        }
        if pB == nil {
            pB = headA
        } else {
            pB = pB.Next
        }
    }
    return pA // hoặc nil nếu không giao
}

// GetIntersectionNodeWithLength — cách giải thay thế: đo độ dài, căn chỉnh.
// Rõ ràng hơn cho người đọc, cùng complexity.
func GetIntersectionNodeWithLength(headA, headB *ListNode) *ListNode {
    lenA, lenB := listLen(headA), listLen(headB)
    for lenA > lenB { headA = headA.Next; lenA-- }
    for lenB > lenA { headB = headB.Next; lenB-- }
    for headA != headB {
        headA, headB = headA.Next, headB.Next
    }
    return headA
}

func listLen(h *ListNode) int {
    n := 0
    for h != nil { n++; h = h.Next }
    return n
}

LRU Cache — Bộ Nhớ Đệm Least Recently Used Expert
LRU Cache kết hợp HashMap + Doubly Linked List: HashMap cho O(1) lookup, Doubly LL để di chuyển node lên đầu hoặc xóa khỏi đuôi trong O(1). Đây là bài toán kinh điển nhất về kết hợp cấu trúc dữ liệu.
Kiến trúc LRU Cache: HashMap + Doubly Linked List
GO lru_cache.go — Full Implementation 
package lrucache

// node là một node trong doubly linked list nội bộ
type node struct {
    key, val      int
    prev, next   *node
}

// LRUCache triển khai cache LRU với O(1) get/put.
// Sử dụng HashMap + Doubly Linked List với hai sentinel node (head, tail).
// Sentinel nodes đơn giản hóa việc insert/remove biên.
type LRUCache struct {
    cap        int
    cache      map[int]*node
    head, tail *node // dummy sentinels
}

// Constructor khởi tạo LRUCache với capacity cap.
func Constructor(cap int) LRUCache {
    head := &node{}
    tail := &node{}
    head.next = tail
    tail.prev = head
    return LRUCache{
        cap:   cap,
        cache: make(map[int]*node),
        head:  head, tail: tail,
    }
}

// Get trả về val của key, -1 nếu không tồn tại.
// Cập nhật vị trí lên MRU.
func (c *LRUCache) Get(key int) int {
    n, ok := c.cache[key]
    if !ok { return -1 }
    c.remove(n)
    c.insertFront(n)
    return n.val
}

// Put thêm hoặc cập nhật key-val.
// Nếu full: evict LRU (node sát tail).
func (c *LRUCache) Put(key, val int) {
    if n, ok := c.cache[key]; ok {
        n.val = val
        c.remove(n)
        c.insertFront(n)
        return
    }
    if len(c.cache) == c.cap {
        // xóa LRU: node ngay trước tail
        lru := c.tail.prev
        c.remove(lru)
        delete(c.cache, lru.key)
    }
    newNode := &node{key: key, val: val}
    c.insertFront(newNode)
    c.cache[key] = newNode
}

// remove tách node khỏi doubly list — O(1)
func (c *LRUCache) remove(n *node) {
    n.prev.next = n.next
    n.next.prev = n.prev
}

// insertFront chèn node vào vị trí MRU (ngay sau head) — O(1)
func (c *LRUCache) insertFront(n *node) {
    n.next = c.head.next
    n.prev = c.head
    c.head.next.prev = n
    c.head.next = n
}

// ─── Thread-safe LRU với sync.RWMutex ───────────────────────────────────────

import "sync"

type SafeLRUCache struct {
    LRUCache
    mu sync.RWMutex
}

func (c *SafeLRUCache) Get(key int) int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.LRUCache.Get(key)
}

func (c *SafeLRUCache) Put(key, val int) {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.LRUCache.Put(key, val)
}
💡 Mẹo thực tế: Go stdlib cung cấp container/list (doubly LL). Trong production dùng được, nhưng trong phỏng vấn nên tự implement để chứng minh hiểu nguyên lý. Package golang.org/x/exp/cache có generic LRU từ Go 1.18+.

Palindrome — Kiểm Tra Chuỗi Đối Xứng Advanced
Kiểm tra linked list có phải palindrome không trong O(n) time, O(1) space: tìm giữa → đảo nửa sau → so sánh từ hai đầu → khôi phục list. Kỹ thuật này kết hợp cả slow/fast pointer và reversal.
Chiến lược 3 pha: Tìm giữa → Đảo → So sánh
GO palindrome.go 
// IsPalindrome kiểm tra linked list có phải palindrome không.
// Time: O(n) | Space: O(1) — không dùng stack hay mảng.
func IsPalindrome(head *ListNode) bool {
    if head == nil || head.Next == nil {
        return true
    }

    // Pha 1: Tìm điểm giữa bằng slow/fast pointer
    slow, fast := head, head
    for fast != nil && fast.Next != nil {
        slow = slow.Next
        fast = fast.Next.Next
    }

    // Pha 2: Đảo ngược nửa sau (bắt đầu từ slow)
    secondHalf := reverseList(slow)
    savedTail := secondHalf // lưu lại để restore sau

    // Pha 3: So sánh hai nửa
    p1, p2 := head, secondHalf
    result := true
    for p2 != nil {
        if p1.Val != p2.Val {
            result = false
            break
        }
        p1, p2 = p1.Next, p2.Next
    }

    // Pha 4: Khôi phục list về trạng thái ban đầu (best practice)
    reverseList(savedTail)
    return result
}

func reverseList(head *ListNode) *ListNode {
    var prev *ListNode
    for head != nil {
        next := head.Next
        head.Next = prev
        prev = head
        head = next
    }
    return prev
}

// IsPalindromeWithStack — giải pháp O(n) space, rõ ràng hơn cho người đọc
func IsPalindromeWithStack(head *ListNode) bool {
    var stack []int
    for cur := head; cur != nil; cur = cur.Next {
        stack = append(stack, cur.Val)
    }
    cur := head
    for i := len(stack) - 1; i >= len(stack)/2; i-- {
        if cur.Val != stack[i] { return false }
        cur = cur.Next
    }
    return true
}

Flatten Multilevel — Phẳng Hóa Danh Sách Đa Cấp Expert
Mỗi node ngoài Next còn có con trỏ Child trỏ đến một danh sách con. Nhiệm vụ: nối tất cả level vào một danh sách phẳng theo thứ tự DFS (depth-first). Kỹ thuật: khi gặp Child, chèn danh sách con giữa node hiện tại và Next.
Cấu trúc Multilevel và kết quả sau flatten
GO flatten.go — LeetCode 430 
// MultilevelNode — node đặc biệt có cả Next và Child
type MultilevelNode struct {
    Val        int
    Prev, Next *MultilevelNode
    Child      *MultilevelNode
}

// Flatten phẳng hóa doubly linked list đa cấp.
// Thuật toán: với mỗi node có Child, chèn chuỗi child vào giữa node và Next.
// Time: O(n) mỗi node được duyệt đúng 1 lần | Space: O(1)
func Flatten(root *MultilevelNode) *MultilevelNode {
    if root == nil { return nil }

    cur := root
    for cur != nil {
        if cur.Child == nil {
            cur = cur.Next
            continue
        }

        // có child: chèn child list vào giữa cur và cur.Next
        child := cur.Child
        childTail := findTail(child) // tìm đuôi của child list

        // nối childTail → cur.Next
        if cur.Next != nil {
            childTail.Next = cur.Next
            cur.Next.Prev = childTail
        }
        // nối cur → child (đầu child list)
        cur.Next = child
        child.Prev = cur
        cur.Child = nil // xóa child pointer

        cur = cur.Next
    }
    return root
}

func findTail(node *MultilevelNode) *MultilevelNode {
    for node.Next != nil { node = node.Next }
    return node
}

// FlattenRecursive — giải pháp đệ quy cho từng node
// Mỗi lần gặp child, flatten nó, nối vào, tiếp tục.
// Space: O(depth) call stack
func FlattenRecursive(root *MultilevelNode) *MultilevelNode {
    var dfs func(head *MultilevelNode) (*MultilevelNode, *MultilevelNode)
    dfs = func(head *MultilevelNode) (*MultilevelNode, *MultilevelNode) {
        cur := head
        var tail *MultilevelNode
        for cur != nil {
            tail = cur
            if cur.Child != nil {
                childHead, childTail := dfs(cur.Child)
                savedNext := cur.Next
                cur.Next = childHead; childHead.Prev = cur
                if savedNext != nil {
                    childTail.Next = savedNext; savedNext.Prev = childTail
                }
                cur.Child = nil
                tail = childTail
                cur = savedNext
            } else {
                cur = cur.Next
            }
        }
        return head, tail
    }
    if root == nil { return nil }
    result, _ := dfs(root)
    return result
}

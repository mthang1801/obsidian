Two Pointers là gì?

Two Pointers (hai con trỏ) là kỹ thuật sử dụng hai biến index để duyệt cấu trúc dữ liệu tuyến tính (mảng, chuỗi) từ hai đầu hoặc theo tốc độ khác nhau. Mục tiêu là giảm độ phức tạp từ O(n²) (brute force với 2 vòng lặp lồng nhau) xuống còn O(n).

Sliding Window là biến thể của Two Pointers — duy trì một "cửa sổ" [left, right] đang mở rộng hoặc co lại, đảm bảo mọi phần tử chỉ được xét tối đa 2 lần (khi vào và khi ra cửa sổ).

Hình 1.1: Trực quan hóa Two Pointers (Opposite Direction) và Sliding Window (Variable Window)

Các biến thể của Two Pointers

↔️ Opposite Direction (Hướng ngược)

Left bắt đầu từ đầu, Right từ cuối. Dùng khi mảng đã sắp xếp và cần tìm cặp thỏa mãn điều kiện (VD: Two Sum II, Container With Most Water).

→→ Same Direction (Cùng chiều)

Cả hai con trỏ đi cùng chiều nhưng tốc độ khác nhau. Fast/Slow pointers phát hiện chu trình (Floyd's Algorithm), remove duplicates.

🪟 Fixed Window (Cửa sổ cố định)

Window size k không đổi. Dùng cho "maximum/minimum sum of k consecutive elements".

🔄 Variable Window (Cửa sổ động)

Window co dãn theo điều kiện. Dùng cho "longest substring without repeating", "minimum window substring".

Điều kiện tiên quyết: Two Pointers (opposite direction) chỉ hoạt động đúng khi mảng/chuỗi đã được sắp xếp hoặc có tính chất đơn điệu. Sliding Window không yêu cầu điều này.

// twoSum sử dụng opposite-direction two pointers.
// Vì mảng đã sắp xếp: nếu sum > target → right--, ngược lại left++.
func twoSum(numbers []int, target int) []int {
    left, right := 0, len(numbers)-1

    for left < right {
        sum := numbers[left] + numbers[right]
        switch {
        case sum == target:
            return []int{left + 1, right + 1} // 1-indexed
        case sum < target:
            left++ // tổng nhỏ quá → tăng left để lấy số lớn hơn
        default:
            right-- // tổng lớn quá → giảm right
        }
    }
    return nil
}

go longest_substring.go 
// lengthOfLongestSubstring dùng sliding window với map lưu vị trí
// cuối cùng của mỗi ký tự. Khi gặp duplicate, left nhảy thẳng
// (không phải tăng từng bước) → O(n) thực sự.
func lengthOfLongestSubstring(s string) int {
    // charIdx[c] = vị trí cuối cùng thấy ký tự c + 1
    charIdx := make(map[byte]int)
    maxLen, left := 0, 0

    for right := 0; right < len(s); right++ {
        c := s[right]
        // Nếu c đã trong window [left, right-1], dịch left
        if prev, ok := charIdx[c]; ok && prev >= left {
            left = prev + 1
        }
        charIdx[c] = right
        if right-left+1 > maxLen {
            maxLen = right - left + 1
        }
    }
    return maxLen
}

// Test cases
func main() {
    fmt.Println(lengthOfLongestSubstring("abcabcbb")) // 3: "abc"
    fmt.Println(lengthOfLongestSubstring("bbbbb"))    // 1: "b"
    fmt.Println(lengthOfLongestSubstring("pwwkew"))   // 3: "wke"
}

HARD

go min_window_substring.go 
// minWindow — Expert sliding window pattern.
// Key insight: dùng 'formed' để biết window hiện tại có đủ ký tự không.
// formed tăng khi count của một ký tự đạt đủ yêu cầu.
func minWindow(s, t string) string {
    if len(t) == 0 { return "" }

    // need[c] = số lần c cần có trong window
    need := make(map[byte]int)
    for i := 0; i < len(t); i++ {
        need[t[i]]++
    }

    window := make(map[byte]int) // count trong window hiện tại
    have, total := 0, len(need)   // have = số ký tự đã đủ; total = số unique cần
    best := [3]int{-1, 0, 0}        // {len, left, right}
    left := 0

    for right := 0; right < len(s); right++ {
        c := s[right]
        window[c]++

        // Kiểm tra c vừa thêm có khiến count đủ không
        if cnt, ok := need[c]; ok && window[c] == cnt {
            have++
        }

        // Thu hẹp window từ trái khi đã đủ tất cả ký tự
        for have == total {
            wLen := right - left + 1
            if best[0] == -1 || wLen < best[0] {
                best = [3]int{wLen, left, right}
            }

            lc := s[left]
            window[lc]--
            if cnt, ok := need[lc]; ok && window[lc] < cnt {
                have--
            }
            left++
        }
    }

    if best[0] == -1 { return "" }
    return s[best[1] : best[2]+1]
}

// trap — Two pointers O(1) space.
// Key insight: lượng nước tại vị trí i = min(maxLeft, maxRight) - height[i].
// Dùng two pointer thay vì precompute maxLeft/maxRight array.
func trap(height []int) int {
    left, right := 0, len(height)-1
    maxLeft, maxRight := 0, 0
    total := 0

    for left < right {
        if height[left] <= height[right] {
            // maxRight đảm bảo >= height[right] >= height[left]
            // nên bottleneck là maxLeft
            if height[left] >= maxLeft {
                maxLeft = height[left]
            } else {
                total += maxLeft - height[left]
            }
            left++
        } else {
            if height[right] >= maxRight {
                maxRight = height[right]
            } else {
                total += maxRight - height[right]
            }
            right--
        }
    }
    return total
}

// Variant: Generic sliding window template cho bài toán bất kỳ
func slidingWindowTemplate[T comparable](
    arr []T,
    valid func(window map[T]int) bool,
    update func(best, current int) int,
) int {
    window := make(map[T]int)
    best, left := 0, 0

    for right, v := range arr {
        window[v]++
        for !valid(window) {
            window[arr[left]]--
            if window[arr[left]] == 0 {
                delete(window, arr[left])
            }
            left++
        }
        best = update(best, right-left+1)
    }
    return best
}

Binary Search là gì?

Binary Search là thuật toán tìm kiếm trên không gian đã sắp xếp bằng cách liên tục chia đôi phạm vi tìm kiếm. Mỗi lần so sánh loại bỏ một nửa không gian → độ phức tạp O(log n).

Ý tưởng cốt lõi: tìm điểm biên (boundary) của một hàm đơn điệu (monotonic predicate). Đây là chìa khóa để áp dụng binary search cho các bài toán tối ưu hóa phức tạp.

Hình 2.1: Trực quan hóa thuật toán Binary Search và thu hẹp không gian tìm kiếm (O(log n))

Ba dạng Binary Search cần nắm

🎯 Exact Match

Tìm chính xác giá trị target. Classic binary search.

⬅️ Left Boundary

Tìm vị trí đầu tiên ≥ target (lower_bound). Khi có duplicate, tìm vị trí trái nhất.

➡️ Right Boundary

Tìm vị trí cuối cùng ≤ target (upper_bound). Khi có duplicate, tìm vị trí phải nhất.

💡

Insight quan trọng: Binary Search không chỉ tìm trong mảng — mà tìm trên không gian đáp án. Nếu có thể định nghĩa hàm canAchieve(x) bool đơn điệu (false...false...true...true), ta dùng binary search tìm ranh giới.

// === Template 1: Exact Match ===
func binarySearch(nums []int, target int) int {
    lo, hi := 0, len(nums)-1
    for lo <= hi {
        mid := lo + (hi-lo)/2 // tránh overflow thay vì (lo+hi)/2
        switch {
        case nums[mid] == target: return mid
        case nums[mid] < target:  lo = mid + 1
        default:                  hi = mid - 1
        }
    }
    return -1
}

// === Template 2: Left Boundary (lower_bound) ===
// Tìm vị trí đầu tiên mà nums[i] >= target
func lowerBound(nums []int, target int) int {
    lo, hi := 0, len(nums) // hi = len (không phải len-1)
    for lo < hi {             // điều kiện lo < hi (không phải <=)
        mid := lo + (hi-lo)/2
        if nums[mid] < target {
            lo = mid + 1
        } else {
            hi = mid // không phải mid-1! Giữ mid vì có thể là đáp án
        }
    }
    return lo // lo == hi tại điểm hội tụ
}

// === Template 3: Right Boundary (upper_bound - 1) ===
// Tìm vị trí cuối cùng mà nums[i] <= target
func upperBound(nums []int, target int) int {
    lo, hi := 0, len(nums)
    for lo < hi {
        mid := lo + (hi-lo)/2
        if nums[mid] <= target {
            lo = mid + 1
        } else {
            hi = mid
        }
    }
    return lo - 1 // vị trí phải nhất có nums[i] <= target
}

// === Template 4: Binary Search on Answer (Universal) ===
// pred(x) = true/false tạo thành dạng: F F F T T T
// Tìm x nhỏ nhất mà pred(x) = true
func binarySearchOnAnswer(lo, hi int, pred func(int) bool) int {
    for lo < hi {
        mid := lo + (hi-lo)/2
        if pred(mid) {
            hi = mid
        } else {
            lo = mid + 1
        }
    }
    return lo
}

// minEatingSpeed — Binary search on answer.
// canFinish(k) = có thể ăn hết trong h giờ với tốc độ k?
// Khi k tăng → canFinish đơn điệu từ false → true.
// Tìm k nhỏ nhất mà canFinish = true.
func minEatingSpeed(piles []int, h int) int {
    maxPile := slices.Max(piles)

    canFinish := func(k int) bool {
        hours := 0
        for _, p := range piles {
            hours += (p + k - 1) / k // ceiling division
        }
        return hours <= h
    }

    return binarySearchOnAnswer(1, maxPile, canFinish)
}

// Search in Rotated Sorted Array (LeetCode #33) — Tìm trong mảng xoay
func searchRotated(nums []int, target int) int {
    lo, hi := 0, len(nums)-1
    for lo <= hi {
        mid := lo + (hi-lo)/2
        if nums[mid] == target { return mid }

        // Xác định nửa nào được sắp xếp
        if nums[lo] <= nums[mid] { // nửa trái đã sắp xếp
            if nums[lo] <= target && target < nums[mid] {
                hi = mid - 1
            } else {
                lo = mid + 1
            }
        } else { // nửa phải đã sắp xếp
            if nums[mid] < target && target <= nums[hi] {
                lo = mid + 1
            } else {
                hi = mid - 1
            }
        }
    }
    return -1
}

// findMedianSortedArrays — Binary search tìm partition.
// Ý tưởng: tìm partition trên mảng nhỏ hơn sao cho phần tử
// phía trái của cả hai mảng đều nhỏ hơn phần tử phía phải.
func findMedianSortedArrays(nums1, nums2 []int) float64 {
    // Đảm bảo nums1 là mảng nhỏ hơn để binary search hiệu quả
    if len(nums1) > len(nums2) {
        nums1, nums2 = nums2, nums1
    }
    m, n := len(nums1), len(nums2)
    half := (m + n + 1) / 2

    lo, hi := 0, m
    for lo <= hi {
        i := lo + (hi-lo)/2 // partition trong nums1
        j := half - i          // partition trong nums2

        // Lấy các phần tử biên, xử lý edge case
        maxLeft1  := getOrInf(nums1, i-1, math.MinInt64)
        minRight1 := getOrInf(nums1, i,   math.MaxInt64)
        maxLeft2  := getOrInf(nums2, j-1, math.MinInt64)
        minRight2 := getOrInf(nums2, j,   math.MaxInt64)

        if maxLeft1 <= minRight2 && maxLeft2 <= minRight1 {
            // Tìm được partition đúng
            if (m+n)%2 == 1 {
                return float64(max(maxLeft1, maxLeft2))
            }
            return float64(max(maxLeft1, maxLeft2)+min(minRight1, minRight2)) / 2.0
        } else if maxLeft1 > minRight2 {
            hi = i - 1
        } else {
            lo = i + 1
        }
    }
    return 0
}

func getOrInf(arr []int, i, def int) int {
    if i < 0 || i >= len(arr) { return def }
    return arr[i]
}

Stack, Queue, Monotonic Stack

Stack (LIFO — Last In First Out): phần tử vào sau cùng ra trước. Trong Go implement bằng slice với append và slice[:len-1].

Queue (FIFO — First In First Out): phần tử vào trước ra trước. Dùng container/list hoặc slice với index.

Monotonic Stack: Stack duy trì tính đơn điệu (tăng hoặc giảm). Khi push phần tử mới, pop tất cả phần tử vi phạm tính đơn điệu. Ứng dụng: Next Greater Element, Largest Rectangle, Sliding Window Maximum.

// dailyTemperatures — Monotonic Decreasing Stack.
// Stack lưu INDEX (không phải giá trị) để tính khoảng cách.
func dailyTemperatures(temperatures []int) []int {
    n := len(temperatures)
    ans := make([]int, n) // default 0: chưa thấy ngày ấm hơn
    stack := make([]int, 0, n) // monotonic decreasing stack of indices

    for i, t := range temperatures {
        // Pop tất cả index có nhiệt độ < t (vi phạm monotonic decreasing)
        for len(stack) > 0 && temperatures[stack[len(stack)-1]] < t {
            j := stack[len(stack)-1]
            stack = stack[:len(stack)-1]
            ans[j] = i - j // i là ngày ấm hơn đầu tiên sau j
        }
        stack = append(stack, i)
    }
    return ans
}

// largestRectangleArea — Monotonic Increasing Stack.
// Mỗi lần pop, tính diện tích với heights[popped] là chiều cao,
// width = i - stack.top - 1.
func largestRectangleArea(heights []int) int {
    // Thêm sentinel 0 ở cuối để flush stack
    heights = append(heights, 0)
    stack := []int{-1} // sentinel index ở bottom
    maxArea := 0

    for i, h := range heights {
        for len(stack) > 1 && heights[stack[len(stack)-1]] > h {
            height := heights[stack[len(stack)-1]]
            stack = stack[:len(stack)-1]
            width := i - stack[len(stack)-1] - 1
            if area := height * width; area > maxArea {
                maxArea = area
            }
        }
        stack = append(stack, i)
    }
    return maxArea
}

// Sliding Window Maximum (LeetCode #239) — Monotonic Deque
func maxSlidingWindow(nums []int, k int) []int {
    deque := make([]int, 0, k) // decreasing deque of indices
    result := make([]int, 0, len(nums)-k+1)

    for i, v := range nums {
        // Loại bỏ index ngoài window
        for len(deque) > 0 && deque[0] < i-k+1 {
            deque = deque[1:]
        }
        // Duy trì deque giảm dần
        for len(deque) > 0 && nums[deque[len(deque)-1]] <= v {
            deque = deque[:len(deque)-1]
        }
        deque = append(deque, i)

        if i >= k-1 {
            result = append(result, nums[deque[0]])
        }
    }
    return result
}

type MinStack struct {
    stack    []int
    minStack []int // đồng bộ với stack, lưu min tại mỗi điểm
}

func (ms *MinStack) Push(val int) {
    ms.stack = append(ms.stack, val)
    curMin := val
    if n := len(ms.minStack); n > 0 {
        if ms.minStack[n-1] < curMin { curMin = ms.minStack[n-1] }
    }
    ms.minStack = append(ms.minStack, curMin)
}

func (ms *MinStack) Pop() {
    ms.stack = ms.stack[:len(ms.stack)-1]
    ms.minStack = ms.minStack[:len(ms.minStack)-1]
}

func (ms *MinStack) Top() int    { return ms.stack[len(ms.stack)-1] }
func (ms *MinStack) GetMin() int { return ms.minStack[len(ms.minStack)-1] }

    Val  int
    Next *ListNode
}

// reverseList — đảo ngược toàn bộ list. O(n) time, O(1) space.
func reverseList(head *ListNode) *ListNode {
    var prev *ListNode
    curr := head
    for curr != nil {
        next := curr.Next  // lưu next trước khi đổi chiều
        curr.Next = prev   // đảo chiều
        prev = curr        // tiến prev
        curr = next        // tiến curr
    }
    return prev // prev là head mới
}

// reverseBetween — đảo ngược từ vị trí left đến right (1-indexed).
func reverseBetween(head *ListNode, left, right int) *ListNode {
    dummy := &ListNode{Next: head}
    pre := dummy

    // Tiến đến node trước left
    for i := 1; i < left; i++ { pre = pre.Next }

    curr := pre.Next
    // "Insert after pre" n lần — kỹ thuật reverse trong O(1) space
    for i := 0; i < right-left; i++ {
        next := curr.Next
        curr.Next = next.Next
        next.Next = pre.Next
        pre.Next = next
    }
    return dummy.Next
}

// detectCycle — trả về node bắt đầu của cycle (Floyd's algorithm).
// Phase 1: Phát hiện cycle (fast/slow gặp nhau).
// Phase 2: Tìm entry point (reset slow về head, cả hai đi 1 bước/lần).
func detectCycle(head *ListNode) *ListNode {
    slow, fast := head, head
    // Phase 1: Detect
    for fast != nil && fast.Next != nil {
        slow = slow.Next
        fast = fast.Next.Next
        if slow == fast { break }
    }
    if fast == nil || fast.Next == nil { return nil }

    // Phase 2: Find entry
    // Khoảng cách từ head đến entry = từ meeting point đến entry
    slow = head
    for slow != fast {
        slow = slow.Next
        fast = fast.Next
    }
    return slow
}

// mergeKLists — Merge K sorted lists. Priority Queue approach.
import "container/heap"

type nodeHeap []*ListNode
func (h nodeHeap) Len() int            { return len(h) }
func (h nodeHeap) Less(i, j int) bool { return h[i].Val < h[j].Val }
func (h nodeHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }
func (h *nodeHeap) Push(x interface{}) { *h = append(*h, x.(*ListNode)) }
func (h *nodeHeap) Pop() interface{} {
    old := *h; n := len(old)
    x := old[n-1]; *h = old[:n-1]; return x
}

func mergeKLists(lists []*ListNode) *ListNode {
    h := &nodeHeap{}
    for _, node := range lists {
        if node != nil { heap.Push(h, node) }
    }
    dummy := &ListNode{}
    curr := dummy
    for h.Len() > 0 {
        node := heap.Pop(h).(*ListNode)
        curr.Next = node
        curr = curr.Next
        if node.Next != nil { heap.Push(h, node.Next) }
    }
    return dummy.Next
}

Tree Traversal trong Go

Tree Traversal (duyệt cây) là quá trình thăm tất cả các node trong cây. Có hai chiến lược chính: DFS (Depth-First Search) duyệt sâu theo nhánh, và BFS (Breadth-First Search) duyệt theo từng tầng.

Ba biến thể DFS: Preorder (Root → Left → Right), Inorder (Left → Root → Right), Postorder (Left → Right → Root). BST đặc biệt: Inorder cho output đã sắp xếp.

Hình 5.1: Các giải thuật duyệt cây nhị phân bao gồm DFS (Pre/In/Postorder) và BFS (Level-order)

Khi nào dùng DFS vs BFS?

DFS — Ưu tiên dùng khi
  • ✓
    Cần tính toán theo chiều sâu (Max Depth, Path Sum)
  • ✓
    Cần Inorder BST (sorted output)
  • ✓
    Backtracking trên cây
  • ✓
    Stack space = O(h) — tối ưu với cây cân bằng
BFS — Ưu tiên dùng khi
  • ✓
    Level-order traversal, Right Side View
  • ✓
    Đường đi ngắn nhất (unweighted)
  • ✓
    Zigzag Level Order
  • ✓
    Minimum depth (tìm node đầu tiên theo level)

type TreeNode struct {
    Val   int
    Left  *TreeNode
    Right *TreeNode
}

// === Recursive DFS === (ngắn gọn, dễ đọc)
func inorderRecursive(root *TreeNode) []int {
    if root == nil { return nil }
    left := inorderRecursive(root.Left)
    right := inorderRecursive(root.Right)
    return append(append(left, root.Val), right...)
}

// === Iterative Inorder === (tránh stack overflow với cây rất sâu)
func inorderIterative(root *TreeNode) []int {
    var result []int
    var stack []*TreeNode
    curr := root

    for curr != nil || len(stack) > 0 {
        // Đi hết sang trái
        for curr != nil {
            stack = append(stack, curr)
            curr = curr.Left
        }
        // Pop và process
        curr = stack[len(stack)-1]
        stack = stack[:len(stack)-1]
        result = append(result, curr.Val)
        curr = curr.Right
    }
    return result
}

// === BFS Level Order ===
func levelOrder(root *TreeNode) [][]int {
    if root == nil { return nil }
    var result [][]int
    queue := []*TreeNode{root}

    for len(queue) > 0 {
        n := len(queue)
        level := make([]int, n)
        for i := 0; i < n; i++ {
            node := queue[i]
            level[i] = node.Val
            if node.Left != nil  { queue = append(queue, node.Left) }
            if node.Right != nil { queue = append(queue, node.Right) }
        }
        queue = queue[n:]
        result = append(result, level)
    }
    return result
}

// maxPathSum — Postorder DFS.
// Key insight: mỗi node có 2 vai trò:
// 1. Là "đỉnh" của path (cộng cả 2 nhánh) → cập nhật global max
// 2. Là "cánh tay" cho parent (chỉ cộng nhánh tốt hơn)
func maxPathSum(root *TreeNode) int {
    maxSum := math.MinInt32

    var dfs func(*TreeNode) int
    dfs = func(node *TreeNode) int {
        if node == nil { return 0 }

        // Chỉ lấy nhánh có giá trị dương (nếu âm thì không lấy)
        leftGain  := max(dfs(node.Left), 0)
        rightGain := max(dfs(node.Right), 0)

        // Cập nhật global max khi node là đỉnh
        if pathSum := node.Val + leftGain + rightGain; pathSum > maxSum {
            maxSum = pathSum
        }

        // Trả về giá trị tốt nhất khi đóng vai "cánh tay"
        return node.Val + max(leftGain, rightGain)
    }
    dfs(root)
    return maxSum
}

// Serialize & Deserialize Binary Tree (LeetCode #297)
import (
    "strconv"
    "strings"

func serialize(root *TreeNode) string {
    var sb strings.Builder
    var dfs func(*TreeNode)
    dfs = func(node *TreeNode) {
        if node == nil { sb.WriteString("#,"); return }
        sb.WriteString(strconv.Itoa(node.Val) + ",")
        dfs(node.Left)
        dfs(node.Right)
    }
    dfs(root)
    return sb.String()
}

func deserialize(data string) *TreeNode {
    vals := strings.Split(data, ",")
    idx := 0
    var build func() *TreeNode
    build = func() *TreeNode {
        if idx >= len(vals) || vals[idx] == "#" { idx++; return nil }
        val, _ := strconv.Atoi(vals[idx]); idx++
        node := &TreeNode{Val: val}
        node.Left = build()
        node.Right = build()
        return node
    }
    return build()
}

// numIslands — DFS flood fill. Mỗi lần gặp '1', DFS sink (đánh dấu '0')
// toàn bộ island đó. Đếm số lần bắt đầu DFS.
func numIslands(grid [][]byte) int {
    rows, cols := len(grid), len(grid[0])
    count := 0

    dirs := [][2]int{{0,1},{0,-1},{1,0},{-1,0}}

    var dfs func(r, c int)
    dfs = func(r, c int) {
        if r < 0 || r >= rows || c < 0 || c >= cols || grid[r][c] != '1' {
            return
        }
        grid[r][c] = '0' // sink / mark visited
        for _, d := range dirs {
            dfs(r+d[0], c+d[1])
        }
    }

    for r := 0; r < rows; r++ {
        for c := 0; c < cols; c++ {
            if grid[r][c] == '1' {
                count++
                dfs(r, c)
            }
        }
    }
    return count
}

// UnionFind với path compression + union by rank.
// Complexity: O(α(n)) ≈ O(1) per operation.
type UnionFind struct {
    parent []int
    rank   []int
    count  int // số connected components
}

func NewUnionFind(n int) *UnionFind {
    parent := make([]int, n)
    rank := make([]int, n)
    for i := range parent { parent[i] = i }
    return &UnionFind{parent, rank, n}
}

func (uf *UnionFind) Find(x int) int {
    if uf.parent[x] != x {
        uf.parent[x] = uf.Find(uf.parent[x]) // path compression
    }
    return uf.parent[x]
}

func (uf *UnionFind) Union(x, y int) bool {
    px, py := uf.Find(x), uf.Find(y)
    if px == py { return false } // đã cùng component
    switch { // union by rank
    case uf.rank[px] < uf.rank[py]: px, py = py, px
    case uf.rank[px] == uf.rank[py]: uf.rank[px]++
    }
    uf.parent[py] = px
    uf.count--
    return true
}

func (uf *UnionFind) Connected(x, y int) bool {
    return uf.Find(x) == uf.Find(y)
}

// findOrder — Kahn's Algorithm (BFS-based topological sort).
// Bắt đầu từ node có in-degree = 0, lần lượt loại bỏ và giảm
// in-degree của neighbors. Nếu cuối cùng đã xử lý tất cả node
// → không có cycle → trả về order.
func findOrder(numCourses int, prerequisites [][2]int) []int {
    adj := make([][]int, numCourses)
    inDegree := make([]int, numCourses)

    for _, prereq := range prerequisites {
        course, pre := prereq[0], prereq[1]
        adj[pre] = append(adj[pre], course)
        inDegree[course]++
    }

    queue := make([]int, 0, numCourses)
    for i, d := range inDegree {
        if d == 0 { queue = append(queue, i) }
    }

    order := make([]int, 0, numCourses)
    for len(queue) > 0 {
        node := queue[0]; queue = queue[1:]
        order = append(order, node)
        for _, next := range adj[node] {
            inDegree[next]--
            if inDegree[next] == 0 {
                queue = append(queue, next)
            }
        }
    }

    if len(order) != numCourses { return nil } // cycle exists
    return order
}

// Dijkstra — Shortest path với weighted graph (LeetCode #743 Network Delay)
import "container/heap"

type Item struct { node, dist int }
type PQ []Item
func (pq PQ) Len() int             { return len(pq) }
func (pq PQ) Less(i, j int) bool   { return pq[i].dist < pq[j].dist }
func (pq PQ) Swap(i, j int)         { pq[i], pq[j] = pq[j], pq[i] }
func (pq *PQ) Push(x interface{})  { *pq = append(*pq, x.(Item)) }
func (pq *PQ) Pop() interface{} {
    old := *pq; n := len(old); x := old[n-1]; *pq = old[:n-1]; return x
}

func networkDelayTime(times [][3]int, n, k int) int {
    adj := make([][]Item, n+1)
    for _, t := range times {
        u, v, w := t[0], t[1], t[2]
        adj[u] = append(adj[u], Item{v, w})
    }

    dist := make([]int, n+1)
    for i := range dist { dist[i] = math.MaxInt32 }
    dist[k] = 0

    pq := &PQ{{k, 0}}
    heap.Init(pq)

    for pq.Len() > 0 {
        cur := heap.Pop(pq).(Item)
        if cur.dist > dist[cur.node] { continue } // stale entry
        for _, nb := range adj[cur.node] {
            if nd := cur.dist + nb.dist; nd < dist[nb.node] {
                dist[nb.node] = nd
                heap.Push(pq, Item{nb.node, nd})
            }
        }
    }

    maxDist := 0
    for i := 1; i <= n; i++ {
        if dist[i] == math.MaxInt32 { return -1 }
        if dist[i] > maxDist { maxDist = dist[i] }
    }
    return maxDist
}

// ladderLength — Bidirectional BFS (Expert optimization).
// Thay vì BFS từ một phía, dùng hai fronts đồng thời.
// Luôn expand front nhỏ hơn → giảm space đáng kể.
func ladderLength(beginWord, endWord string, wordList []string) int {
    wordSet := make(map[string]bool)
    for _, w := range wordList { wordSet[w] = true }
    if !wordSet[endWord] { return 0 }

    // Bidirectional BFS: 2 fronts gặp nhau ở giữa
    front1 := map[string]bool{beginWord: true}
    front2 := map[string]bool{endWord: true}
    steps := 1

    for len(front1) > 0 {
        steps++
        if len(front1) > len(front2) {
            front1, front2 = front2, front1 // expand smaller front
        }
        next := make(map[string]bool)
        for word := range front1 {
            bs := []byte(word)
            for i := range bs {
                orig := bs[i]
                for c := byte('a'); c <= 'z'; c++ {
                    if c == orig { continue }
                    bs[i] = c
                    nw := string(bs)
                    if front2[nw] { return steps } // fronts met!
                    if wordSet[nw] {
                        next[nw] = true
                        delete(wordSet, nw)
                    }
                }
                bs[i] = orig
            }
        }
        front1 = next
    }
    return 0
}
